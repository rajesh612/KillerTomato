 34 function MVInfoWindow(mpane, x, y, tabs, index, _width, _height,style, mouseOver, parent, screenLoc, parameters, domDiv, infoLayerId)
 35 {
 36   if(_width == null || _width<0) _width=0;
 37   if(_height == null || _height<0) _height=0;
 38   
 39   this.wininfopar=parent.parent;
 40   if(style!="MVInfoWindowStyle1" && style!="MVInfoWindowStyle4")
 41   {
 42     if(!_width)
 43       _width = MVGlobalVariables.infoWindowWidth;
 44     if(!_height)
 45       _height = MVGlobalVariables.infoWindowHeight;
 46     eval(style+".init()");
 47     this.winDiv=eval(style+".createWindow("+x+","+y+","+_width+","+ _height+","+parent.parent.mvId+")");
 48     mpane.appendChild(this.winDiv);
 49   }
 50   
 51   var close = function(){
 52     if(MVInfoWindowStyle1.redrawHandle)
 53     {
 54       clearTimeout(MVInfoWindowStyle1.redrawHandle) ;
 55       MVInfoWindowStyle1.redrawHandle = null ;
 56     }
 57     var id = parent.parent.mvId+"_"+infoLayerId;
 58     if(MVInfoWindowStyle4.redrawHandle && MVInfoWindowStyle4.redrawHandle[id])
 59     {
 60       clearTimeout(MVInfoWindowStyle4.redrawHandle[id]) ;
 61       MVInfoWindowStyle4.redrawHandle[id] = null ;
 62     }
 63     MVInfoWindowStyle1.clear() ;
 64     MVInfoWindowStyle4.clear() ;
 65     parent.deleteInfoWindow(mpane);
 66   }
 67   
 68   if(style=="MVInfoWindowStyle1")
 69   {
 70     var winDiv = this.winDiv ;
 71     var mvinfo = this;  
 72     var drawInfoWindow = function()
 73     {
 74       if(MVInfoWindowStyle1.sizeChanged(parent.parent.mvId))
 75       {
 76         MVUtil.removeAllChildren(mpane);
 77         MVInfoWindowStyle1.init(parameters) ;
 78         winDiv=MVInfoWindowStyle1.createWindow(x,y,_width,_height,tabs.length, parent.parent.mvId);
 79         MVInfoWindowStyle1.setTitleStyle(tabs,index) ;
 80         mpane.appendChild(winDiv);
 81         MVInfoWindowStyle1.addContent(mpane, tabs) ;
 82         var table = document.getElementById("infowindow3table_"+parent.parent.mvId);
 83         MVInfoWindowStyle1.getWindowSize(_width, _height, parent.parent.mvId);
 84         MVInfoWindowStyle1.removeInActiveTabs(index, parent.parent.mvId);
 85         if(table)
 86         {
 87           if(!isNaN(screenLoc.x))
 88           {
 89             var infoPos = mvinfo.checkInfoWindowOverLayer(parent, infoLayerId, table, screenLoc, MVInfoWindowStyle1.windowOffset);
 90             table.style.left = MVUtil.pixel(infoPos.x-screenLoc.x) ;
 91             table.style.top = MVUtil.pixel(infoPos.y-screenLoc.y) ;
 92             if(!mouseOver)
 93             {
 94               MVInfoWindowStyle1.addCloseButton(close);
 95               MVInfoWindowStyle1.renderCone(mpane, parent.parent.mvId) ;
 96             }
 97             //MVInfoWindowStyle1.getWindowSize();
 98           }
 99           else
100           {
101             if(!mouseOver)
102             {
103               MVInfoWindowStyle1.addCloseButton(close);
104               MVInfoWindowStyle1.renderCone(mpane, parent.parent.mvId) ;
105             }
106           }
107         }
108       }
109       MVInfoWindowStyle1.redrawHandle = setTimeout(drawInfoWindow, 500) ;
110     }
111     drawInfoWindow() ;
112     this.winDiv = winDiv ;
113   }
114   else if(style=="MVInfoWindowStyle4")
115   {
116     var winDiv = this.winDiv ;
117     MVInfoWindowStyle4.init(parameters) ;
118     this.mvId = parent.parent.mvId+"_"+infoLayerId;
119     var mvinfo = this;  
120     var autoResize = false;
121     if (_width == 0 || _height == 0)
122       autoResize = true;
123     var drawInfoWindow = function()
124     {
125       var mvId = mvinfo.mvId;
126       if (autoResize == true)
127       {
128         _width = 0;
129         _height = 0;
130       }
131       if(MVInfoWindowStyle4.sizeChanged(mvId))
132       {
133         if (domDiv && domDiv.parentNode)
134         {
135           domDiv.parentNode.removeChild(domDiv);
136         }
137         MVUtil.removeAllChildren(mpane);
138         winDiv = document.createElement("div");
139         winDiv.style.position = "absolute";
140         winDiv.style.left = "0px";
141         winDiv.style.top = "0px";
142         winDiv.style.zIndex = "200";
143         var contentDiv;
144         if (tabs == null && domDiv != null)
145         {
146           contentDiv = MVInfoWindowStyle4.createContentByDiv(mvId, domDiv, autoResize, _width, _height);
147           winDiv.appendChild(contentDiv);
148           mpane.appendChild(winDiv);
149         }
150         else if (tabs.length == 1)
151         {
152           contentDiv=MVInfoWindowStyle4.createTitleAndContent(mvId, tabs[0].getTitle(), tabs[0].getContent(), autoResize, _width, _height, close, mouseOver);
153           winDiv.appendChild(contentDiv);
154           mpane.appendChild(winDiv);
155           var titleCell = document.getElementById(mvId+"_title");
156           if (titleCell)
157           {
158             var titleHeight = titleCell.offsetHeight;
159             if (_height && _height-titleHeight-26>0)
160             {
161               var conDiv = document.getElementById(mvId+"_content");
162               conDiv.style.height = MVUtil.pixel(_height-titleHeight-26);
163             }
164           }
165         }
166         else
167         {
168           if (_width == 0)
169             _width = MVGlobalVariables.infoWindowWidth;
170           if (_height == 0)
171             _height = MVGlobalVariables.infoWindowHeight; 
172           mpane.appendChild(winDiv);
173           contentDiv = MVInfoWindowStyle4.createTabsAndContent(tabs,index, _width, _height,winDiv,close);
174         }  
175         
176         if (autoResize)
177         {
178           _width = contentDiv.offsetWidth;
179           if (_width < MVGlobalVariables.infoWindowWidth)
180           {
181             contentDiv.style.width = MVUtil.pixel(MVGlobalVariables.infoWindowWidth);
182             _width = MVGlobalVariables.infoWindowWidth;
183           }
184           _height = contentDiv.offsetHeight;
185           if (_height < MVGlobalVariables.infoWindowHeight)
186           {
187             contentDiv.style.height = MVUtil.pixel(MVGlobalVariables.infoWindowHeight);
188             _height = MVGlobalVariables.infoWindowHeight;
189           }
190         }
191         var dialogDiv=MVInfoWindowStyle4.createWindow(x,y,_width,_height,tabs, mvId, close, mouseOver);
192         winDiv.appendChild(dialogDiv);
193         var table = document.getElementById("infowindow4_"+mvId);
194         if(table)
195         {
196           MVInfoWindowStyle4.infoWindowWidth[mvId] = table.offsetWidth ;
197           MVInfoWindowStyle4.infoWindowHeight[mvId] = table.offsetHeight ;
198           if(!isNaN(screenLoc.x))
199           {
200             var infoPos = mvinfo.checkInfoWindowOverLayer(parent, infoLayerId, table, screenLoc, MVInfoWindowStyle4.windowOffset);
201             table.style.left = MVUtil.pixel(infoPos.x-screenLoc.x) ;
202             table.style.top = MVUtil.pixel(infoPos.y-screenLoc.y) ;
203             if (contentDiv != null)
204             {
205               contentDiv.style.left = MVUtil.pixel(infoPos.x-screenLoc.x) ;
206               contentDiv.style.top = MVUtil.pixel(infoPos.y-screenLoc.y) ;
207             }
208             if(!mouseOver)
209             {
210               MVInfoWindowStyle4.renderCone(mpane, mvId) ;
211             }
212           }
213           else
214           {
215             if(!mouseOver)
216             {
217               MVInfoWindowStyle4.renderCone(mpane, mvId) ;
218             }
219           }
220         }
221       }
222       MVInfoWindowStyle4.redrawHandle[mvId] = setTimeout(drawInfoWindow, 500) ;
223     }
224     drawInfoWindow() ;
225     this.winDiv = winDiv ;
226   }
227   else
228   {
229     div=this.createContentArea(tabs[0].getContent(), _width);
230     eval(style+".addContent(div)");
231     if(!mouseOver)
232     {
233       eval(style+".addCloseButton(close)");
234     }
235   }
236 
237 }

241 MVInfoWindow.prototype.createContentArea=function(infoStr, width)
242 {
243   var mDiv=window.document.createElement("DIV");
244   mDiv.className="infowindow";
245   mDiv.style.position="absolute";
246   
247   mDiv.style.zIndex=3;
248   mDiv.innerHTML = "<table width="+width+"px><tr><td > " + infoStr + "</td></tr></table>";
249   MVUtil.setCursor(mDiv,"auto");
250   return mDiv;
251 }

286 MVInfoWindow.prototype.checkInfoWindowOverLayer=function(parent, infoLayerId, table, screenLoc, offset)
287 {
288   var w=table.offsetWidth; 
289   var h=table.offsetHeight;
290   var divLeft = parent.parent.divLeft; 
291   var divTop = parent.parent.divTop ;
292   var mapWidth = parent.width; 
293   var mapHeight = parent.height ;
294   var infoX = screenLoc.x ;
295   var infoY = screenLoc.y ;
296   var right = true ; 
297   var top = true ;
298   if(divLeft+infoX > mapWidth/2) 
299     right = false ;
300   if(divTop+infoY < mapHeight/2) 
301     top = false ;
302   var vPortion = false ;
303   if (!parent.parent.displayedWindow)
304     parent.parent.displayedWindow = new Array();
305   infoLayerId = parseInt(infoLayerId);
306   if (infoLayerId==2 && parent.parent.displayedWindow[1] ||
307       infoLayerId==1 && parent.parent.displayedWindow[2])
308   {
309     if (right == parent.parent.displayedWindow[3-infoLayerId].right && 
310         top == parent.parent.displayedWindow[3-infoLayerId].top)
311     {
312       var tmp = divLeft+infoX-mapWidth/2 ;
313       tmp = tmp/mapWidth ;
314       var r1 = Math.abs(tmp/2);
315       tmp = divTop+infoY-mapHeight/2 ;
316       tmp = tmp/mapHeight ;
317       var t1 = Math.abs(tmp/2);
318       if (r1>t1)
319         top = !top;
320       else
321         right = !right;
322     }
323   }
324   parent.parent.displayedWindow[infoLayerId] = {right:right, top:top, x:screenLoc.x, y:screenLoc.y};
325   if(top)
326   {
327     if(divTop+infoY-offset-h >=0)
328       infoY = infoY-offset-h ;
329     else
330     {
331       infoY = -divTop+offset ;
332       vPortion = true ;
333     }
334   }
335   else
336   {
337     if(divTop+infoY+offset+h <= mapHeight)
338       infoY = infoY+offset;
339     else
340     {
341       infoY = mapHeight-divTop-offset-h;
342       vPortion = true ;
343     }
344   }
345 
346   if(right)
347   {
348     if(divLeft+infoX+offset+w <= mapWidth || vPortion)
349       infoX = infoX+offset ;
350     else
351       infoX = mapWidth-divLeft-offset-w ;
352   }
353   else
354   {
355     if(divLeft+infoX-offset-w >= 0 || vPortion)
356       infoX = infoX-offset-w ;
357     else
358       infoX = -divLeft+offset ;
359   }
360   return {x:infoX, y:infoY};
361 }
