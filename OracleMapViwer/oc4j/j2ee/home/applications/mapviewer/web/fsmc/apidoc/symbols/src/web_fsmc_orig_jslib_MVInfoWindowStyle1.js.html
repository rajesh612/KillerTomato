 41 /**@private*/
 42 MVInfoWindowStyle1.setStyleParameters = function(parameters)
 43 {
 44   if(parameters && parameters.hideCloseButton)
 45     this.hideCloseButton = true ;
 46   else
 47     this.hideCloseButton = false ;
 48   if(parameters && parameters.closeButtonImageURL)
 49   {
 50     if(!this.style_closeButtonImageURL || 
 51        this.style_closeButtonImageURL!=parameters.closeButtonImageURL)
 52     {
 53       this.closeButtonWidth = null;
 54       this.closeButtonHeight = null;
 55     }
 56     this.style_closeButtonImageURL = parameters.closeButtonImageURL ;
 57     if(parameters.closeButtonWidth)
 58       this.closeButtonWidth = parameters.closeButtonWidth;
 59     if(parameters.closeButtonHeight)
 60       this.closeButtonHeight = parameters.closeButtonHeight;
 61   }
 62   else
 63   {
 64     if(!this.style_closeButtonImageURL || 
 65        this.style_closeButtonImageURL!=MVInfoWindowStyle1.closeButtonImgURL)
 66     {
 67       this.closeButtonWidth = null;
 68       this.closeButtonHeight = null;
 69     }
 70     this.style_closeButtonImageURL = MVInfoWindowStyle1.closeButtonImgURL;
 71     this.closeButtonWidth = MVGlobalVariables.toolButtonW-2 ;
 72     this.closeButtonHeight = MVGlobalVariables.toolButtonW-2 ;
 73   }
 74   if(parameters && parameters.bodyStyle)
 75     this.style_bodyStyle = parameters.bodyStyle ;
 76   else
 77     this.style_bodyStyle = "border:1px;border-color:#8794A3;background-color:#FFFFFF" ;
 78     //this.style_bodyStyle = "border:1px;border-color:#8794A3;background-color:#FFFFFF;white-space:-moz-pre-wrap" ;
 79   if(parameters && parameters.titleBarStyle)
 80     this.style_titleBarStyle = parameters.titleBarStyle ;
 81   else
 82     this.style_titleBarStyle = "background-color:#CFDCEB;font-weight:bold" ;
 83   if(parameters && parameters.coneStyle)
 84   {
 85     this.style_coneStyle = parameters.coneStyle ;
 86     var words = parameters.coneStyle.split(/;|:/);
 87     if(words)
 88     {
 89       for(i=0; i<words.length; i++)
 90         if(words[i].indexOf("opacity")>=0 && i<words.length)
 91         {
 92           this.style_coneOpacity = parseInt(words[i+1]) ;
 93           break ;
 94         }
 95     }
 96   }
 97   else
 98   {
 99     this.style_coneStyle = "background-color:#99CC33;opacity:30" ;
100     this.style_coneOpacity = 30 ;
101   }
102   if(parameters && parameters.offset)
103     MVInfoWindowStyle1.windowOffset = parameters.offset ;
104   else
105     MVInfoWindowStyle1.windowOffset = 20 ;
106   if(parameters && parameters.activeTab)
107     this.style_activeTab = parameters.activeTab;
108   else
109     this.style_activeTab = "background-color:#FFFFFF;font-weight:bold";
110   
111   if(parameters && parameters.inactiveTab)
112     this.style_inactiveTab = parameters.inactiveTab;
113   else
114     this.style_inactiveTab = "background-color:#CFDCEB;font-weight:bold";
115   
116   if(parameters && parameters.tabBorder)
117     this.style_tabBorder = parameters.tabBorder;
118   else
119     this.style_tabBorder = "1px solid #8794A3";
120   
121   if(parameters && parameters.contentStyle)
122     this.style_contentStyle = parameters.contentStyle;
123   else
124     this.style_contentStyle = "padding:5px";
125 }

314 /**@private*/
315 MVInfoWindowStyle1.addContent=function(parent, tabs)
316 {
317   var titleIsEmpty = true ;
318   for(var i=0;i<tabs.length;i++){
319     // To combat IE+ADF issue, we need to first remove the table 
320     // cell, set innserHTML and then add it back.
321     var tc = this.contentBar[i].childNodes[0] ;
322     var parent = this.contentBar[i] ;
323     parent.removeChild(tc) ;
324     tc.innerHTML = tabs[i].getContent() ;
325     if(parent.childNodes.length>0)
326       parent.insertBefore(tc, parent.childNodes[0]) ;
327     else
328       parent.appendChild(tc) ;
329     
330     var title=tabs[i].getTitle();
331     if (!title)
332       title=" ";
333     else 
334       titleIsEmpty = false ;
335     parent = this.titleBar[i].parentNode ;
336     parent.removeChild(this.titleBar[i]) ;
337     this.titleBar[i].innerHTML = title ;
338     if(parent.childNodes.length>0)
339       parent.insertBefore(this.titleBar[i], parent.childNodes[0]) ;
340     else
341       parent.appendChild(this.titleBar[i]) ;
342   }
343   if(titleIsEmpty && this.hideCloseButton)
344   {
345     this.titleRow.parentNode.removeChild(this.titleRow) ;
346     MVUtil.clearNode(this.titleRow) ;
347     this.titleRow = null ;
348   }

479 MVInfoWindowStyle1.opaque = function(object)
480 {
481   if(MVGlobalVariables.browser=="IF"){

505 /**@private*/
506 MVInfoWindowStyle1.getWindowSize = function(_width, _height,mvId)
507 {
508   var infoTable = document.getElementById("infowindow3table_"+mvId) ;
509   if(infoTable)
510   {
511     var width=infoTable.offsetWidth;
512     if (width>(_width?_width:MVGlobalVariables.infoWindowWidth))
513       infoTable.style.width=MVUtil.pixel(width);
514     //get the max height of content row
515     var max=0;
516     var contentHight=this.contentBar[0].offsetHeight;
517     for(var i=1;i<this.contentBar.length;i++){
518       if (this.contentBar[max].offsetHeight<this.contentBar[i].offsetHeight)
519         max=i;
520       contentHight+=this.contentBar[i].offsetHeight;
521     }
522     var height=infoTable.offsetHeight-contentHight+this.contentBar[max].offsetHeight;
523     if (height>(_height?_height:MVGlobalVariables.infoWindowHeight))
524       infoTable.style.height = MVUtil.pixel(height);
525   }
526 }
