 47 /**
 48   @class MVOverviewMap defines an overview map that displays the miniature overview of the main map 
 49   as a small rectangle inside itself. The overview map is usually displayed inside a 
 50   small HTML DIV element, whose size and position can be customized. 
 51   The overview map can also be placed inside
 52   a collapsible map decoration. As the result, the user can hide or display the
 53   overview map by minimize or restore the map decoration. </p>
 54   <p>The overview map always displays maps that are zoomed out from the main map. 
 55   By default, the overview map and the first(bottom) main base map layer 
 56   are rendered by the same map tile layer. If different map tile layer 
 57   are used, they must be defined in the same coordinate system and have the 
 58   same zoom level definition. </p>
 59   
 60   The following code snippet adds an overview map inside a map decoration.
 61  <pre>
 62     md = new MVMapDecoration(new MVOverviewMap(4),null,null,200,150) ;
 63     md.setCollapsible(true, true);
 64     mapview.addMapDecoration(md);
 65  </pre>
 66 
 67   The following code snippet adds an overview map inside a DIV container outside the map.
 68  <pre>
 69   var overviewMapContainer = document.getElementById("overview_map");
 70   var over=new MVOverviewMap(overviewMapContainer,4);
 71   mapview.addOverviewMap(over) ;
 72  </pre>
 73 
 74   @param {DIV} overviewMapDIV the HTML DIV element 
 75                         inside which the overview map is displayed. 
 76                         This parameter should be set to null or omitted when 
 77                         the overview map is to be displayed inside a map decoration object.
 78   @param {int} zoomdiff an integer that specifies the zoom level difference 
 79                         between the overview map and the main map. 
 80   @param {String} overviewMapTileLayer an optional String parameter that 
 81                         specifies the map tile layer to be used by the 
 82                         overview map. If omitted, the first map tile layer
 83                         added to the MVMapView instance is used the overview map.
 84   @constructor
 85 */
 86 function MVOverviewMap(overviewMapDIV,zoomdiff,overviewMapTileLayer)
 87 {
 88   if(overviewMapDIV && !overviewMapDIV.style)
 89   {
 90     if(zoomdiff)
 91       overviewMapTileLayer = zoomdiff ;
 92     zoomdiff = overviewMapDIV ;
 93   }
 94   /**@private*/
 95   this.type = "MVOverviewMap" ;
 96   /**@private*/
 97   this.isMapControl = true ;
 98   /**@private*/
 99   this.overviewMapDIV = overviewMapDIV ? overviewMapDIV:null;
100   /**@private*/
101   this.mainmap = null;
102   /**@private*/
103   this.diff = zoomdiff ? zoomdiff : 4;     
104   /**@private*/
105   this.submap = null;
106   /**@private*/
107   this.overviewMapTileLayer = overviewMapTileLayer;
108   /**@private*/
109   this.moveflag = false;
110   /**@private*/
111   this.moveX = 0;
112   /**@private*/
113   this.moveY = 0;
114   /**@private*/
115   this.coords = null;
116   /**@private*/
117   this.moveOnSubMap = false;
118   /**@private*/
119   this.moveOnMainMap = false;
120   /**@private*/
121   this.borderStyle = "2px solid red";
122   /**@private*/
123   this.backgroundColor = "red";
124   /**@private*/
125   this.cursorStyleDragging = null ;
126   /**@private*/
127   this.cursorStyleDefault = null ;
128   /**@private*/
129   this.origDocumentMouseUpListener = null;
130 }

133   @private
134 */
135 MVOverviewMap.prototype.init=function(overviewMapDIV, mainmap)
136 {
137   mainmap.overviewmap = this;
138   if(overviewMapDIV)
139     this.overviewMapDIV = overviewMapDIV;
140   this.overviewMapDIV.overview = true;
141   this.mainmap = mainmap;
142   var subZoom = this.mainmap.getZoomLevel()-this.diff;  
143   if (subZoom<0)
144     subZoom=0;   
145   this.submap = new MVMapView(this.overviewMapDIV, this.mainmap.baseURL);
146   this.submap.isSubmap = true ;
147   
148   if(mainmap.wraparound)
149 	this.submap.wraparound = true;
150   
151   this.submap.enableLoadingIcon(false);
152   if(this.cursorStyleDefault)
153     this.submap.setMouseCursorStyle(this.cursorStyleDefault, "default");
154   if(this.cursorStyleDragging)
155     this.submap.setMouseCursorStyle(this.cursorStyleDragging, "dragging");
156   if (this.overviewMapTileLayer == null)
157   {
158     this.overviewMapTileLayer=this.mainmap.get1stMapTileLayer().clone();
159   }
160   
161   this.rectangle=document.createElement("div");
162   this.submap.base_div.appendChild(this.rectangle);
163   this.rectangle.style.zIndex=10000;
164   //add listener
165   var ovm = this;
166   var moveRectangle = function(x,y)
167   {
168     if (!ovm.moveOnMainMap)
169     {
170       ovm.rectangle.style.left = (parseInt(ovm.rectangle.style.left)-x)+"px";
171       ovm.rectangle.style.top = (parseInt(ovm.rectangle.style.top)-y)+"px";
172     }
173   }
174    
175   var zoomSubMap=function(beforeZoom,afterZoom)
176   {
177     var different=beforeZoom-afterZoom;
178     if (afterZoom>beforeZoom && ovm.mainmap.zoomflag)
179     {
180       ovm.submap.zoomflag=false;
181       ovm.mainmap.setZoomLevel(MVOverviewMap.validateZoomLevel(ovm.mainmap.getZoomLevel()-different));
182       ovm.resetRectangle();
183     
184     }
185     else if (afterZoom<beforeZoom && ovm.mainmap.zoomflag)
186     {
187       ovm.submap.zoomflag=false;
188       if (ovm.mainmap.getZoomLevel()-beforeZoom==ovm.diff){
189         ovm.mainmap.setZoomLevel(MVOverviewMap.validateZoomLevel(ovm.mainmap.getZoomLevel()-different));
190       }
191       else{
192       	ovm.mainmap.setZoomLevel(MVOverviewMap.validateZoomLevel(beforeZoom-different+ovm.diff));
193       }
194       ovm.resetRectangle();
195     }
196     ovm.mainmap.zoomflag=true;
197     ovm.resetRectangle();
198   }
199 
200   var refreshMainMap=function()
201   {
202     if (!ovm.moveOnMainMap)
203     {
204       ovm.moveOnSubMap=true;
205       ovm.mainmap.setCenter(ovm.submap.getCenter());
206       ovm.resetRectangle();
207     }
208     ovm.moveOnMainMap=false;
209   }
210   //this.submap.addEventListener("moving",moveRectangle);
211   this.submap.setEventListener(MVEvent.RECENTER,refreshMainMap);
212   this.submap.addEventListener(MVEvent.ZOOM_LEVEL_CHANGE,zoomSubMap);
213 
214   var mapTileLayerAdded = function()
215   {
216     var center = ovm.mainmap.getCenter();
217     ovm.submap.setCenter(center);   
218     ovm.submap.setZoomLevel(MVOverviewMap.validateZoomLevel(subZoom));
219     ovm.submap.overview=true;
220     function setRectangle()
221     {
222       ovm.resetRectangle();
223     }
224     ovm.submap.setEventListener(MVEvent.INITIALIZE, setRectangle);
225     ovm.submap.display();
226   }
227   var tl = this.overviewMapTileLayer ;
228   if(tl.getType && tl.getType()=='MVExternalAPIMapTileLayer')
229   {
230     if(tl.hideCopyRights)
231       tl.initializeListener = function(){tl.hideCopyRights();} ;
232   }
233   this.submap.addMapTileLayer(this.overviewMapTileLayer, mapTileLayerAdded);
234 }

240   this.submap.addMapTileLayer(this.overviewMapTileLayer);
241 }

248 /** @private */
249 MVOverviewMap.prototype.destroy = function()
250 {
251   if(this.mainmap)
252     this.mainmap.overviewmap = null ;
253   if(this.submap)
254     this.submap.destroy() ;
255   MVUtil.clearNode(this.overviewMapDIV, this.mainmap?this.mainmap.errorHandler:null) ;
256   this.submap = null ;
257 }
258 /**
259   @private
260 */
261 MVOverviewMap.prototype.resetRectangle = function()
262 {
263   var mainBBox = this.mainmap.getMapWindowBBox() ;
264   var subBBox = this.submap.getMapWindowBBox();
265   var ovm=this;
266   
267   var setRectSize = function (mBBox)
268   {
269     var mmbr = mBBox.getMBR() ;
270     var smbr = subBBox.getMBR() ;
271     ovm.xRatio = (mmbr[2]-mmbr[0])/(smbr[2]-smbr[0]) ;
272     ovm.yRatio = (mmbr[3]-mmbr[1])/(smbr[3]-smbr[1]) ;
273     var width=ovm.xRatio*ovm.submap.getPaneWidth();
274     var height=ovm.yRatio*ovm.submap.getPaneHeight();
275     ovm.rectangle.style.position = "absolute";
276     ovm.rectangle.style.overflow = "hidden";
277     ovm.rectangle.style.zIndex = 10000
278     ovm.rectangle.style.left = ((ovm.submap.getPaneWidth()-width)/2)+"px";
279     ovm.rectangle.style.top = ((ovm.submap.getPaneHeight()-height)/2)+"px";
280     ovm.rectangle.style.width = width+"px";
281     ovm.rectangle.style.height = height+"px";
282     ovm.rectangle.style.border = ovm.borderStyle;
283     
284     if (MVGlobalVariables.browser == 'NS')
285       ovm.rectangle.innerHTML = "<div  style='width: " + width + "px; height: " + height
286         + "px;opacity: 0.3;background-color:" + ovm.backgroundColor+";'></div>";
287     else if (MVGlobalVariables.browser == 'SF')
288       ovm.rectangle.innerHTML = "<div  style='width: " + width + "px; height: " + height
289         + "px;opacity: 0.3;background-color:" + ovm.backgroundColor+";'></div>";
290     else if (MVGlobalVariables.browser == 'IF')
291       ovm.rectangle.innerHTML = "<div  style='width: " + width + "px; height: " + height
292         + "px;filter:alpha(opacity=30);background-color:" + ovm.backgroundColor+";'></div>";
293     if (width>ovm.submap.getPaneWidth() || height>ovm.submap.getPaneHeight())
294       ovm.rectangle.style.visibility="hidden";
295     else
296       ovm.rectangle.style.visibility=ovm.submap.base_div.style.visibility;
297       
298     
299     MVUtil.attachEvent(ovm.rectangle, "mousedown", MVUtil.closureWrapper(ovm.rectangle,function(evt){
300       var center = ovm.mainmap.getCenter();
301       evt = (evt) ? evt : ((window.event) ? event : null); 
302       if (evt.button==2)
303       {
304         return;
305       }
306       ovm.coords=ovm.submap.getEventCoords(evt)
307       ovm.moveflag=true;
308       //evt.cancelBubble=true;
309       MVUtil.stopEvent(evt);
310       ovm.moveX=0;
311       ovm.moveY=0;
312     }), true) ;
313   
314     MVUtil.attachEvent(ovm.rectangle, "mouseup", MVUtil.closureWrapper(ovm.rectangle,function(evt)
315     {
316       if(ovm.moveflag)
317       {
318         evt = (evt) ? evt : ((window.event) ? event : null); 
319         ovm.moveflag=false;
320         evt.cancelBubble=true;
321         ovm.rectangle.endmove();
322       }
323     }), true) ;
324     
325     ovm.rectangle.endmove = MVUtil.closureWrapper(ovm.rectangle,function()
326     {
327       ovm.moveflag = false ;
328       var newCenterx_pix=parseInt(this.style.left)+parseInt(this.style.width)/2;
329       var newCentery_pix=parseInt(this.style.top)+parseInt(this.style.height)/2;
330   
331       // THe original complex statements are broken into simpler ones for 
332       // obfuscator. -- ji
333       var submap = ovm.submap;
334       var tx = newCenterx_pix - submap.getPaneWidth()/2 ;
335       tx = tx/submap.pixPerX;
336       var ty = newCentery_pix - submap.getPaneHeight()/2 ;
337       ty = ty/submap.pixPerY;
338       
339       var newCenterX=ovm.submap.getCenterLon()+tx;
340       var newCenterY=ovm.submap.getCenterLat()-ty;
341       
342       var point=MVSdoGeometry.createPoint(newCenterX,newCenterY,ovm.submap.srid);
343       ovm.submap.setCenter(point);
344     })
345     
346     if(!ovm.origDocumentMouseUpListener)
347       MVUtil.detachEvent(window.document, "mouseup", ovm.origDocumentMouseUpListener) ; 
348     ovm.origDocumentMouseUpListener = MVUtil.closureWrapper(window.document, function(evt)
349     {
350       if (ovm.moveflag){
351         ovm.moveflag=false;
352         ovm.rectangle.endmove();
353       }
354     });
355     MVUtil.attachEvent(window.document, "mouseup", ovm.origDocumentMouseUpListener) ; 
356   
357     MVUtil.attachEvent(ovm.rectangle, "mousemove", MVUtil.closureWrapper(ovm.rectangle,function(evt)
358     {
359       evt = (evt) ? evt : ((window.event) ? event : null); 
360       if (ovm.moveflag)
361       {
362         var coords2=ovm.submap.getEventCoords(evt);
363         x=coords2.left-ovm.coords.left;
364         y=coords2.top-ovm.coords.top;
365         ovm.coords=coords2;
366         this.style.left = (parseInt(this.style.left)+x)+"px";
367         this.style.top = (parseInt(this.style.top)+y)+"px";
368         
369         evt.cancelBubble=true;
370       }
371     }), true) ;
372   }
373   
374   if(mainBBox.srid!=subBBox.srid)
375   {
376     this.mainmap.transformGeom(mainBBox, subBBox.srid, null, setRectSize);
377   }
378   else
379     setRectSize(mainBBox) ;
380 }

382 /**
383   @private
384 */
385 MVOverviewMap.prototype.moveRectanglefromMainMap=function(x,y)
386 {
387   if(this.xRatio==undefined)
388     return ;
389   
390   //The Rectangle on submap should move less pixel than Main map
391   var moveX=Math.floor(this.xRatio*x * (this.submap.getPaneWidth()/this.mainmap.getPaneWidth()));
392   var moveY=Math.floor(this.yRatio*y * (this.submap.getPaneHeight()/this.mainmap.getPaneHeight()));
393 
394   if (moveX>=0)
395     moveX=Math.floor(moveX);
396   else
397     moveX=Math.ceil(moveX);
398   
399   if (moveY>=0)
400     moveY=Math.floor(moveY);
401   else
402     moveY=Math.ceil(moveY);
403   
404   if (!this.moveOnSubMap)
405   {
406     this.moveOnMainMap=true;
407     this.rectangle.style.left = (parseInt(this.rectangle.style.left)+moveX)+"px";
408     this.rectangle.style.top = (parseInt(this.rectangle.style.top)+moveY)+"px";
409   }
410 }

426 /**
427   @private
428 */
429 MVOverviewMap.prototype.zoomMainMap = function(beforeZoom,afterZoom,center)
430 {
431   var different = beforeZoom - afterZoom;
432   if (afterZoom >= beforeZoom && this.submap.zoomflag )
433   {
434     this.mainmap.zoomflag=false;
435     if (beforeZoom - this.submap.getZoomLevel() == this.diff){
436       if(center)
437         this.submap.setCenterAndZoomLevel(center, MVOverviewMap.validateZoomLevel(this.submap.getZoomLevel() - different));
438       else
439         this.submap.setZoomLevel(MVOverviewMap.validateZoomLevel(this.submap.getZoomLevel() - different)) ;
440     }
441     else {
442       if(center)
443         this.submap.setCenterAndZoomLevel(center, MVOverviewMap.validateZoomLevel(beforeZoom - different - this.diff));
444       else  
445         this.submap.setZoomLevel(MVOverviewMap.validateZoomLevel(beforeZoom - different - this.diff));
446     }
447     //this.resetRectangle();
448   }
449   else if (afterZoom < beforeZoom && this.submap.zoomflag)
450   {
451     this.mainmap.zoomflag = false;
452     if(center)
453       this.submap.setCenterAndZoomLevel(center, MVOverviewMap.validateZoomLevel(this.submap.getZoomLevel() - different));
454     else
455       this.submap.setZoomLevel(MVOverviewMap.validateZoomLevel(this.submap.getZoomLevel() - different));
456     //this.resetRectangle();
457   }
458   this.submap.zoomflag = true;
459 }
