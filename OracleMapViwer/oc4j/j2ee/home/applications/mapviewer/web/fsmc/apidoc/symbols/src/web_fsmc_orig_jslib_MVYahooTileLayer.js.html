119 /**@private*/
120 MVYahooTileLayer.prototype.init = function (div)
121 {
122   this.map = new YMap(div, this.mapType) ;
123   var center = null ;
124   if(this.centerX != null && this.centerY != null)
125   {
126     center = new YGeoPoint(this.centerY, this.centerX);
127     this.map.drawZoomAndCenter(center, 17-this.zoomLevel) ;
128   }
129   if(this.initializeListener)
130     this.initializeListener() ;
131 }
132 /**@private*/
133 MVYahooTileLayer.prototype.setCenterAndZoomLevel = function (x, y, zoomLevel)
134 {
135   this.centerX = x ;
136   this.centerY = y ;
137   this.zoomLevel = zoomLevel ;
138   if(this.map)
139     this.map.panToLatLon(new YGeoPoint(y, x), 17-this.zoomLevel) ;
140 }
141 /**@private*/
142 MVYahooTileLayer.prototype.getCenter = function ()
143 {
144   var center = this.map.getCenterLatLon() ;
145   return MVSdoGeometry.createPoint(center.Lon, center.Lat, this.srid) ;
146 }
147 /**@private*/
148 MVYahooTileLayer.prototype.move = function (xOffset, yOffset)
149 {
150   this.map.moveByXY(new YCoordPoint(xOffset, yOffset)) ;
151 }
152 /**@private*/
153 MVYahooTileLayer.prototype.setCenter = function (x, y)

159 }
