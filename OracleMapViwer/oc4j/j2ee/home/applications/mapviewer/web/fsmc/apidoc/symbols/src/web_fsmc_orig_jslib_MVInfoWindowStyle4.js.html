 48 MVInfoWindowStyle4.setStyleParameters = function(parameters)
 49 {
 50   if(parameters && parameters.hideCloseButton)
 51     this.hideCloseButton = true ;
 52   else
 53     this.hideCloseButton = false ;
 54   if(parameters && parameters.coneStyle)
 55   {
 56     this.style_coneStyle = parameters.coneStyle ;
 57     var words = parameters.coneStyle.split(/;|:/);
 58     if(words)
 59     {
 60       for(i=0; i<words.length; i++)
 61         if(words[i].indexOf("opacity")>=0 && i<words.length)
 62         {
 63           this.style_coneOpacity = parseInt(words[i+1]) ;
 64           break ;
 65         }
 66     }
 67   }
 68   else
 69   {
 70     this.style_coneStyle = "background-color:#a0a0a0;opacity:30" ;
 71     this.style_coneOpacity = 30 ;
 72   }
 73   if(parameters && parameters.offset)
 74     MVInfoWindowStyle4.windowOffset = parameters.offset ;
 75   else
 76     MVInfoWindowStyle4.windowOffset = 20;
 77   if(parameters && parameters.tabStyle != undefined && parameters.tabStyle.toLowerCase() == "f")
 78     this.tabStyle = "f";
 79   else
 80     this.tabStyle = "p";
 81 }

259   shadowDiv.style.left = MVUtil.pixel(3);
260   shadowDiv.style.top = MVUtil.pixel(3);
261   for(var i = 0; i<18; i++)
262   {
263     var sDiv=document.createElement("DIV");
264     sDiv.style.backgroundColor = "#000000";
265     sDiv.style.left = MVUtil.pixel(leftP);
266     sDiv.style.top = MVUtil.pixel(topP);
267     sDiv.style.position = "absolute";
268     sDiv.style.width = MVUtil.pixel(width-3);
269     sDiv.style.height = MVUtil.pixel(height-3);
270     var opacity = opacityArray[i];
271     MVInfoWindowStyle4.setOpaque(sDiv, opacity);
272     if (i%2 == 0)
273       leftP++;
274     else
275       topP++;
276     shadowDiv.appendChild(sDiv);
277   }
278   return shadowDiv;
