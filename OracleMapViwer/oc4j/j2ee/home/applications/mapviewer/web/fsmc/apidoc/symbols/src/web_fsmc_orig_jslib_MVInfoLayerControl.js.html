422  MVInfoLayerControl.prototype.calculateInfoWindowHtml = function(_id, _name, _attrs, _attrNames, _theme,nameAsTitle,style)
423  {
424     var infoArray = new Array(4);
425     if(!_name)
426       _name = "null" ;
427     _name = MVUtil.strReplaceAll(_name, "  ", "  ");
428     var nameLength = 0 ;
429     var lineNum = 0;
430     var result = ""; 
431     var charWidth = 12 ;
432     if (_name != "null")
433     {
434       if(nameAsTitle)
435         infoArray[3] = _name ;
436       else
437       {
438         nameLength = _name.length;
439         lineNum++;
440         result = "<tr><td>"+this.replaceWhiteSpace(_name)+"</td></tr>";
441       }
442     }  
443     
444     result = "<table id='mvinfotable_"+this.parent.mvId+"'>"+result;
445     
446     if (_attrs==null || _attrs.length==0)   //has  no attrs
447     {
448       infoArray[0] = result;
449       if(style=="MVInfoWindowStyle1")
450       {
451         infoArray[1] = 0 ;
452         infoArray[2] = 0 ;
453       }
454       else
455       {
456         infoArray[1] = _name.length * charWidth + 36;
457         infoArray[2] = 27 + 30;
458       }
459       return infoArray;
460     }
461     
462     var maxAttNameLen = 0;
463     var maxAttValueLen = 0;
464     
465     lineNum += _attrs.length;
466     
467     result += "<tr><td><table id='mvinfotable1_"+this.parent.mvId+"'>";  
468     for (var a = 0; a <_attrs.length; a++)
469     {
470        var color = "#bbbbbb";
471        if (a % 2 == 0)
472          color = "#dddddd";
473        if (_attrNames!=null && _attrNames[a]!=null && _attrNames[a].length > maxAttNameLen) 
474             maxAttNameLen = _attrNames[a].length;
475        if (_attrs[a]==null)
476             _attrs[a]="";
477        if (_attrs[a].length > maxAttValueLen) 
478            maxAttValueLen = _attrs[a].length;
479            
480        var attr = this.replaceWhiteSpace(_attrs[a]);
481        result = result + "<tr bgcolor=" + color + ">"; 
482        if(_attrNames!=null )
483        {
484          var attrName="";
485          if (_attrNames[a]==null)
486            attrName = " ";
487          else
488            attrName = this.replaceWhiteSpace(_attrNames[a])+":";
489          result = result + "<td>" + attrName + "</td>" ;
490        }
491 //       result = result + "<td align=left style='word-break:break-all;white-space:-moz-pre-wrap;'>" + attr + "</td></tr>";
492        result = result + "<td align=left>" + attr + "</td></tr>";
493        
494     }
495     var infoWidth;
496     if (nameLength > (maxAttNameLen + maxAttValueLen))
497      infoWidth = (nameLength + 1)* charWidth + 40;
498     else 
499      infoWidth = (maxAttNameLen + maxAttValueLen + 1)* charWidth + 40;
500     infoArray[0] = result+"</table></td></tr></table>";
501     if(style=="MVInfoWindowStyle1" || style=="MVInfoWindowStyle4")
502     {
503       infoArray[1] = 0 ;
504       infoArray[2] = 0 ;
505     }
506     else
507     {
508       infoArray[1] = infoWidth;
509       infoArray[2] = lineNum*11 + 80;
510     }
511     return infoArray;     
512  }

527    if (_width && _height)

533    else

539      {
540        this.showInfoWindow(_id, x, y, tabs, 0, result[1], result[2],style,time,mouseOver,parameters,true);
541      }
542      else
543        this.showInfoWindow(_id, x, y, tabs, 0, MVGlobalVariables.infoWindowWidth, 
544                                                  MVGlobalVariables.infoWindowHeight,style,time, mouseOver,parameters,true);
545    }
546  }

559  MVInfoLayerControl.prototype.deleteInfoWindow = function(win)
560  {
561    if(MVInfoWindowStyle1.redrawHandle)
562    {
563      clearTimeout(MVInfoWindowStyle1.redrawHandle) ;
564      MVInfoWindowStyle1.redrawHandle = null ;
565    }
566    var infoWinId = this.parent.mvId+"_"+this.id;;
567    if(MVInfoWindowStyle4.redrawHandle && MVInfoWindowStyle4.redrawHandle[infoWinId])
568    {
569      clearTimeout(MVInfoWindowStyle4.redrawHandle[infoWinId]) ;
570      MVInfoWindowStyle4.redrawHandle[infoWinId] = null ;
571    }
572    var i=0 ;
573    for(i=0;i < this.infoWindows.length; i++)
574    {
575      if(this.infoWindows[i] == win) // info window already exist?
576      {
577        this.infoWindows.splice(i,1) ;
578      }
579    }
580    MVUtil.freeMemory(win);
581    MVUtil.clearNode(win, this.parent.errorHandler);
582    this.div.removeChild(win);
583    this.mouseOnInfoWindow = false;
584  }

610 MVInfoLayerControl.prototype.scrollInfoLayer = function()
611 {
612   for(var i=0;i < this.infoWindows.length; i++)
613   {
614     if(this.infoWindows[i].winX < this.parent.mapWindowMinLon || this.infoWindows[i].winX > this.parent.mapWindowMaxLon ||
615        this.infoWindows[i].winY < this.parent.mapWindowMinLat || this.infoWindows[i].winY > this.parent.mapWindowMaxLat )
616       this.infoWindows[i].style.visibility = "hidden";       
617     else
618       this.infoWindows[i].style.visibility = "visible";       
619   }
620 }

622 MVInfoLayerControl.prototype.replaceWhiteSpace = function(str)
623 {
624   var i = str.indexOf("<") ;
625   if(i>=0 && str.indexOf(">",i)>=0) // possible html tag?
626     return str ;
627    
628   str = str.replace(" ", " ");
629   return MVUtil.strReplaceAll(str, "  ", "  ");
630 }
