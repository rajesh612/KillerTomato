227  MVUtil.offsetPositionAbsolute = function(mcontainer)
228  {
229    var b={"x":0,"y":0};
230    var mc = mcontainer ;
231    while(mcontainer)
232    {
233      b.x+=mcontainer.offsetLeft;
234      b.y+=mcontainer.offsetTop;
235      mcontainer=mcontainer.offsetParent;
236    }
237    mcontainer = mc ;
238    while(mcontainer && mcontainer.tagName!="BODY")
239    {
240      b.x-=mcontainer.scrollLeft;
241      b.y-=mcontainer.scrollTop;
242      mcontainer=mcontainer.parentNode;
243    }
244    return b;
245  }

340  /**@private 
341   * Get the mouse event location relative to the upper-left 
342   * corner of the document.
343  */
344  MVUtil.getMouseXY = function(evt)
345  {
346    evt=(evt) ? evt : ((window.event) ? event : null);
347    var x = 0;
348    var y = 0;
349    if (evt.touches)
350    {
351      var touchEvent = null ;
352      if(evt.touches.length == 0)
353      {
354        if(evt.changedTouches && evt.changedTouches.length>0)
355          touchEvent = evt.changedTouches[0] ;
356        else
357          return null ;
358      }
359      else
360        touchEvent = evt.touches[0] ;
361      x = touchEvent.pageX ;
362      y = touchEvent.pageY ;
363    }
364    if (evt.pageX) 
365    {
366      x = evt.pageX;
367      y = evt.pageY;
368    }
369    else if (evt.clientX)
370    {
371      x = evt.clientX + document.body.scrollLeft - document.body.clientLeft ;
372      y = evt.clientY + document.body.scrollTop - document.body.clientTop ;
373      if (document.body.parentElement && (document.body.parentElement.clientLeft
374          || document.body.parentElement.clientTop)) 
375      {
376        var bodParent = document.body.parentElement;
377        x += bodParent.scrollLeft - bodParent.clientLeft;
378        y += bodParent.scrollTop - bodParent.clientTop;
379      }
380    }
381    return MVSdoGeometry.createPoint(x,y).sdo_point;  
382  }

385   /**
386   When the map the map displayed in the wrap-around manner, keep 
387   the longitude value of a point in the original data coordinate system boundary
388   @param {Object} par the parent container,  MVMapView object.
389   @param {double} initial value of the longitude
390   @returns  {double} the longitude in the coordinate system boundary
391  */ 
392  MVUtil.keepInCSBoundary = function(par,x)
393  {
394     var width = par.msi.coverMaxX - par.msi.coverMinX;
395     while(x < par.msi.coverMinX)
396     {
397         x = x + width;
398     }
399     while(x > par.msi.coverMaxX)
400     {
401         x -= width;
402     }
403     return x;
404  }

658  MVUtil.closureWrapper=function(obj, func)
659  {  
660    if (MVGlobalVariables.browser != "IF")
661    {
662      return function()
663      {
664        return func.apply(obj, arguments);
665      }
666    }
667     if(!(MVUtil.closureObj))
668     {
669       MVUtil.closureObj=new Array();
670       MVUtil.closureFunc=new Array();
671       MVUtil.closureSlot=new Array();
672     }
673 
674     if(!(MVUtil.closureObjSpec))
675     {
676       MVUtil.closureObjSpec=new Array();
677       MVUtil.closureFuncSpec=new Array();
678       MVUtil.closureSlotSpec=new Array();
679     }       
680    
681     try
682     {       
683     var i=(MVUtil.closureObj.length);
684     if(obj.closureIndex)
685       {
686         i=(obj.closureIndex);
687       }
688       else
689         obj.closureIndex=-1;
690             
691       if((obj.closureIndex)==-1 && (MVUtil.closureSlot.length)>0)
692         i=MVUtil.closureSlot.pop();   
693     
694     MVUtil.closureObj[i]=obj;
695     MVUtil.closureObj[i].closureIndex=i;
696     
697     if(!(MVUtil.closureFunc[i]))
698       MVUtil.closureFunc[i]=new Array();
699     
700     var j=(MVUtil.closureFunc[i].length);
701     MVUtil.closureFunc[i][j]=func;
702       
703     obj=null;
704     func=null;  
705     
706     return function()
707     {
708       return MVUtil.closureFunc[i][j].apply(MVUtil.closureObj[i], arguments);
709     }
710   }
711   catch(e)
712   {
713     var i=(MVUtil.closureObjSpec.length);
714                   
715       if((MVUtil.closureSlotSpec.length)>0)
716         i=MVUtil.closureSlotSpec.pop();   
717     
718     MVUtil.closureObjSpec[i]=obj;
719     MVUtil.closureFuncSpec[i]=func;
720       
721     obj=null;
722     func=null;  
723     
724     return function()
725     {
726       return MVUtil.closureFuncSpec[i].apply(MVUtil.closureObjSpec[i], arguments);
727     }
728   }

1244 /**@private*/
1245 //APPLY LOCALE NUMBER FORMATTING ON number
1246 MVUtil.formatNum = function(num)
1247 {
1248   if(num == null ||num == "undefine")
1249     return;
1250   //Make sure we don't print out too many floating point digits
1251   if(Math.abs(num)>=1000)
1252     num = Math.round(num) ;
1253   else if(Math.abs(num)>=100)
1254     num = Math.round(num*10)/10.0 ;
1255   else if(Math.abs(num)>=10)
1256     num = Math.round(num*100)/100.0 ;
1257   else if(Math.abs(num)>=1)
1258     num = Math.round(num*1000)/1000.0 ;
1259   else if(Math.abs(num)>=0.1)
1260     num = Math.round(num*10000)/10000.0 ;
1261   else if(Math.abs(num)>=0.01)
1262     num = Math.round(num*100000)/100000.0 ;
1263   else if(Math.abs(num)>=0.001)
1264     num = Math.round(num*1000000)/1000000.0 ;
1265   else
1266     num = Math.round(num*10000000)/10000000.0 ;
1267   //Get the group seperator and decimal seperator.
1268   var groupSeperator = MVMessages["number_format"].charAt(1);
1269   var decimalSeperator = MVMessages["number_format"].charAt(5);
1270   num = new String(num);
1271   if(num.indexOf(".00") > 0)
1272   {
1273     num = num.substring(0,num.indexOf(".00"));
1274   }
1275   var numbers, intPart, decimalPart;
1276   if(num.indexOf(".")>0)
1277   {
1278     numbers = num.split(".");
1279     intPart = numbers[0];
1280     decimalPart = numbers[1];
1281   }
1282   else
1283   { intPart = num; 
1284     decimalPart = null;
1285   }
1286   //Format the integer part
1287   var i = intPart.length-1-3-3;
1288   var numInt;
1289   if(intPart.length <= 3)
1290     numInt = intPart;
1291   else{
1292     numInt = groupSeperator.concat(intPart.slice(-3));
1293     while(i>-1)
1294     {
1295       //Get three characters from the tail
1296       var sSubString = intPart.substr(i+1,3);
1297       numInt = groupSeperator.concat(sSubString, numInt);
1298       i= i - 3;
1299     }
1300     if(intPart.length%3 != 0)
1301       numInt = intPart.substr(0,(intPart.length%3)).concat(numInt);
1302     else 
1303       numInt = intPart.substr(0,3).concat(numInt);
1304   }
1305 
1306   //Format the decimal part
1307   if(decimalPart != null)
1308   {
1309     var j = 3;
1310     var numDeci;
1311     if(decimalPart.length <= 3)
1312       numDeci = decimalPart;
1313     else{
1314       numDeci = decimalPart.slice(0,3);
1315       while(j<=decimalPart.length-1-3)
1316       {
1317         //Get three characters from the head
1318         var sSubString = decimalPart.substr(j,3);
1319         numDeci = numDeci.concat(groupSeperator,sSubString);
1320         j= j + 3;
1321       }
1322       if(decimalPart.length%3 != 0)
1323         numDeci = numDeci.concat(groupSeperator,decimalPart.slice(-decimalPart.length%3));
1324       else 
1325         numDeci = numDeci.concat(groupSeperator,decimalPart.slice(-3));
1326     }
1327   }
1328   if(decimalPart == null )
1329     return numInt;
1330   else
1331     return numInt.concat(decimalSeperator,numDeci);
1332 }

1572       }
1573       distance = xOrdinate - centerX;
1574     }
1575     if( xOrdinate < boundMin)
1576       pointAreaCode = centerAreaCode - 1 + i;
1577     else if( xOrdinate > boundMax)
1578       pointAreaCode = centerAreaCode + 1 + i;
1579     else

1635          "tileImageWidth":250,"tileImageHeight":250

1637        ...
1638      ]
1639    }

1642 MVUtil.canBeCombined = function(layer1, layer2)
1643 {
1644   if(layer1.doNotCombine || layer2.doNotCombine)
1645     return false ;
1646   // must both be tile layer served by mapviewer
1647   if(layer1.getType()!="MVMapTileLayer" || layer2.getType()!="MVMapTileLayer" ||
1648      !layer1.layerControl || !layer2.layerControl ||
1649      !(layer1.layerControl instanceof MVMapTilesControl) ||
1650      !(layer2.layerControl instanceof MVMapTilesControl))
1651     return false ;
1652   // must be served by the same mapviewer
1653   if(layer1.mcsURL != layer2.mcsURL)
1654     return false ;
1655   // must have the same tile layer configuration
1656   var config1 = layer1.config ;
1657   var config2 = layer2.config ;
1658   if(config1.clientTileAccess || config2.clientTileAccess)
1659     return false ;
1660   if(config1.dataSource!=config1.dataSource)
1661     return false ;
1662   // must be PNG
1663   if(config1.format != "PNG" || config2.format != "PNG")
1664     return false ;
1665   if (layer1.opacity != layer2.opacity)
1666     return false;
1667   if(config1.coordSys.srid!=config2.coordSys.srid ||
1668      config1.coordSys.minX!=config2.coordSys.minX ||
1669      config1.coordSys.minY!=config2.coordSys.minY)
1670     return false ;
1671   if(config1.zoomLevels.length!=config1.zoomLevels.length)
1672     return false ;
1673   for(var i=0; i<config1.zoomLevels.length; i++)
1674   {
1675     if(config1.zoomLevels[i].tileWidth != config2.zoomLevels[i].tileWidth ||
1676        config1.zoomLevels[i].tileHeight != config2.zoomLevels[i].tileHeight ||
1677        config1.zoomLevels[i].tileImageWidth != config2.zoomLevels[i].tileImageWidth ||
1678        config1.zoomLevels[i].tileImageHeight != config2.zoomLevels[i].tileImageHeight)
1679       return false ;
1680   }
1681   return true ;
1682 }
