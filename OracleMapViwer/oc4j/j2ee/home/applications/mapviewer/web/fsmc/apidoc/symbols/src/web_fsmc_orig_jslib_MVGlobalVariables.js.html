202 MVGlobalVariables.getMapCacheConfig = function(basemap, callBack, errorHandler)
203 {
204   if (basemap.mcsURL == null || basemap.mapSource == null || MVUtil.strTrim(basemap.mapSource) == "")
205   {
206     basemap.mapSource = "null"; 
207     MVGlobal_CacheConfig[basemap.mapSource] = null ; //MVConfig.defaultJson;
208     return;
209   }
210   basemap.mapSource = basemap.mapSource.toUpperCase();
211   if(MVGlobal_CacheConfig[basemap.mapSource])
212   {
213     if(MVGlobal_CacheConfig[basemap.mapSource].transparent)
214       basemap.isTransparent = true ;
215     if(callBack)
216       callBack(MVGlobal_CacheConfig[basemap.mapSource]) ;
217     return MVGlobal_CacheConfig[basemap.mapSource];
218   }
219   
220   var formatStr ;
221                   
222   var configURL = basemap.mcscfgURL ;
223   var requestStr = "xml_request=" + encodeURIComponent("<?xml version=\"1.0\" standalone=\"yes\"?>"
224                + "<map_cache_admin_request><get_client_config map_cache_names=\"" + basemap.mapSource 
225                + "\" format=\"JSON\"/></map_cache_admin_request>");   
226  
227   var request =null;
228   
229   var localDomain = MVUtil.isLocalDomain(basemap.mcsURL) ;
230   var xmlHttp = localDomain || MVMapView.XMLHTTPEnabled ;
231   var configLoaded = function()
232   {
233     try
234     {
235       var resp = xmlHttp ? request.responseText : this.responseText ;
236       eval("var result="+resp);
237       if (result.length == 0)
238       {
239         MVi18n.reportError("MVGlobalVariables.getMapCacheConfig", "MAPVIEWER-05501", basemap.mapSource, errorHandler);
240         MVGlobal_CacheConfig[basemap.mapSource] = null ; //MVConfig.defaultJson;
241         result = null;
242         return;
243       } 
244       if(result[0].transparent)
245         basemap.isTransparent = true ;
246       MVGlobalVariables.addMapCacheConfig(basemap.mapSource, result[0]);
247       result = null;    
248     }
249     catch(e)
250     {
251       MVi18n.reportError("MVGlobalVariables.getMapCacheConfig", "MAPVIEWER-05523",
252                          basemap.mapSource+":"+request.responseText, errorHandler);
253       return ;
254     }
255     if(callBack)
256       callBack(MVGlobal_CacheConfig[basemap.mapSource]) ;
257   }
258   
259   try 
260   {
261     request = MVUtil.getXMLHttpRequest(xmlHttp);
262     if(!xmlHttp)
263       request.onreadystatechange = configLoaded ;
264     request.open("POST", MVUtil.getXMLHttpUrl(configURL), !xmlHttp);
265     request.setRequestHeader("Content-Type","application/x-www-form-urlencoded"); 
266     request.send(MVUtil.getXMLHttpReq(configURL, requestStr)) ;
267   } catch (e) 
268   {
269     MVi18n.reportError("MVGlobalVariables.getMapCacheConfig", "MAPVIEWER-05511", e, errorHandler); 
270     return ;
271   } 
272   
273   if (xmlHttp)
274   {
275     if(request.status == 200)
276     { 
277       configLoaded() ;
278     }
279     else
280     {
281       MVGlobal_CacheConfig[basemap.mapSource] = null; //MVConfig.defaultJson;
282       MVi18n.reportError("MVGlobalVariables.getMapCacheConfig", "MAPVIEWER-05511", 
283                          basemap.mapSource+":"+request.responseText, errorHandler);
284     }
285   }
286   request = null;
287 }
