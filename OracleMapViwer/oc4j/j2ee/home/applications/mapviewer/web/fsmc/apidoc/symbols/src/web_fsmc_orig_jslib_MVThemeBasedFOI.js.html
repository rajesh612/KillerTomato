489  /**
490    This method highlights the FOIs which are specified by an array of FOI IDs.  
491 
492    @param {Array} foiIdArray an array of FOI IDs, and those FOIs will be highlighted. 
493                             All FOIs, which are returned from the server 
494                             currently, will be highlighted if it is omitted. 
495    @returns  None
496    @type     void
497  */ 
498 MVThemeBasedFOI.prototype.highlight=function(foiIdArray)
499 {
500   if(!this.enableHighlightMode)
501   {
502     MVi18n.reportError("MVThemeBasedFOI.highlight", "MAPVIEWER-05524", null,
503                        this.parent?this.parent.errorHandler:null);
504     return;
505   }
506   var highlightall = false;
507   if(foiIdArray==null)
508   {
509     highlightall = true;
510   }
511   else if(typeof(foiIdArray)!="object" || (len=foiIdArray.length)==null || foiIdArray.length<1)
512   {
513     MVi18n.reportError("MVThemeBasedFOI.highlight", "MAPVIEWER-05519", "foiIdArray",
514                        this.parent?this.parent.errorHandler:null);
515     return;
516   }
517   
518   var highlightFoiList=this.highlightFoiList;
519   var foiArray=this.queryResult.foiarray;
520   var len=foiArray.length;
521   for(var i=0; i<len; ++i)
522   {
523     var foi=foiArray[i];
524     var flag = highlightall;
525     if (foi.mouseDown)
526       continue;
527     else {
528       for(var j=0;j<highlightFoiList.length;j++)
529         if (foi == highlightFoiList[j])
530         {
531           foi.mouseDown = true;
532           foi = null;
533           break;
534         }
535     }
536     if(foi!=null && foi.id!=null)
537     {
538       if (!flag)
539       {
540         for(var j=0;j<foiIdArray.length;j++)
541           if (foi.id == foiIdArray[j])
542           {
543             flag = true;
544             break;
545           }
546       }
547       
548       if(foi.nodeId!=null && flag)
549       {
550         //the foi should be highlighted
551         highlightFoiList.push(foi);
552         foi.mouseDown = true;
553         this.themeLayerControl.highlightFOI(this.themeLayerControl,foi);
554       }
555     }
556   }
557 }

559 /**
560   This method changes the FOIs, which are specified by an array of FOI IDs, from highlighted status to non-highlighted status.   
561 
562   @param {Array} foiIdArray an array of FOI IDs, and those FOIs will be changed from 
563                             highlighted status to non-highlighted status. All highlighted 
564                             FOIs will be changed to non-highlighted status if it is omitted. 
565   @returns  None
566   @type     void
567 */ 
568 MVThemeBasedFOI.prototype.deHighlight=function(foiIdArray)
569 {
570   if(!this.enableHighlightMode)
571   {
572     MVi18n.reportError("MVThemeBasedFOI.deHighlight", "MAPVIEWER-05524", null, 
573                        this.parent?this.parent.errorHandler:null);
574     return;
575   }
576   var highlightFoiList=this.highlightFoiList;
577   var len=highlightFoiList.length;
578   if(len<1)  
579     return;
580   var len1=null;
581   if(foiIdArray!=null && (typeof(foiIdArray)!="object" || (len1=foiIdArray.length)==null))
582   {
583     MVi18n.reportError("MVThemeBasedFOI.deHighlight", "MAPVIEWER-05519", "foiIdArray",
584                        this.parent?this.parent.errorHandler:null);
585     return;
586   }
587   if(foiIdArray!=null && len1<1)
588     return;  
589   
590   var dehighlightall = false;
591   if(foiIdArray==null)   //restore all
592   {
593     dehighlightall = true;
594   }
595   
596   for(var i=highlightFoiList.length; i>=0; i--)
597   {
598     var foi = highlightFoiList[i];
599     var flag = dehighlightall;
600     if(foi!=null && foi.id!=null)
601     {
602       if (!flag)
603       {
604         for(var j=0;j<foiIdArray.length;j++)
605           if (foi.id == foiIdArray[j])
606           {
607             flag = true;
608             break;
609           }
610       }
611     
612       if (flag)
613       {
614         //the foi should be highlighted
615         highlightFoiList[i]=null;
616         highlightFoiList.pop();
617         foi.mouseDown = false;
618         if (foi!=null && foi.node != null && foi.imgurl != null)
619           this.themeLayerControl.deHighlightFOI(this.themeLayerControl,foi);
620       }
621     }
622   }
623 }
