 46 /**
 47  @class MVNavigationPanel defines a map navigation panel that is to be displayed 
 48  inside a map decoration(MVMapDecoration). The visibility and positioning of the 
 49  scale bar is determined by the map decoration that holds the scale bar. An existing 
 50  navigation panel can be removed by removing the correspondind map decoration 
 51  object from the map.<br> 
 52 
 53  The default navigation panel has a zoom level bar that can be dragged to change 
 54  map zoom level, a zoom in button, a zoom out button and 8-direction panning buttons. 
 55  
 56  Typical usage of this class is illustrated below.
 57  <pre>
 58     var md=new MVMapDecoration(new MVNavigationPanel(),0 ,0 , null, null, 4, 4);
 59     mapview.addMapDecoration(md);
 60  </pre>
 61 
 62  @param  {int} panningButtonNumber an integer that specifies the number of directional panning 
 63                    buttons, which can be 0, 4 or 8. 
 64  @param  {boolean} zoomingBar a boolean this parameter specifies whether to display the zooming bar 
 65                    on the navigation panel.
 66  @param  {boolean} zoomingButtons a boolean this parameter specifies whether to display the zooming 
 67                    buttons(+/-) on the navigation panel.
 68  @param  {boolean} zoomLevelBars a boolean this parameter specifies whether to display the zoom level
 69                    bars on the navigation panel.
 70  @constructor
 71 */

 73 {

 86     zoomingBar = false ;

122         return;

126       }

136       for (div in nvpanel.infoTipDivs)

139       }

191 /**@private*/
192 MVNavigationPanel.prototype.init = function(containerDiv, mv)
193 {
194   if(containerDiv)
195   {
196     this.containerDiv = containerDiv;
197     if (containerDiv.isMapDecoration)
198     {

200     }
201   }

213     MVUtil.stopEvent(MVUtil.getEvent(evt));
214   });

216   {
217     MVUtil.stopEvent(MVUtil.getEvent(evt));
218   });
219     
220   this.containerDiv.appendChild(this.control_panel_div);
221   if(mv)
222   {
223     this.mvHandle = mv ;
224     mv.navigationPanel = this ;
225   }
226   
227   if (this.navPosition)
228   {
229     if (this.navPosition.toLowerCase() == "east")  
230     {
231       if (this.panButtonFlag==8)
232         this.control_panel_div.style.left = MVUtil.pixel(this.mvHandle.getPaneWidth() - 80);
233       else if(this.panButtonFlag==4)
234         this.control_panel_div.style.left = MVUtil.pixel(this.mvHandle.getPaneWidth() - 80);

237     }
238     else  
239     {
240       if (this.panButtonFlag == 8)

242       else if (this.panButtonFlag == 8)
243         this.control_panel_div.style.left = MVUtil.pixel(10);
244       else if (this.panButtonFlag == 0)
245         this.control_panel_div.style.left = MVUtil.pixel(30);
246     }
247     this.control_panel_div.style.top = MVUtil.pixel(10);      
248   }
249   else 
250   {
251     this.control_panel_div.style.left = MVUtil.pixel(0);
252     this.control_panel_div.style.top = MVUtil.pixel(0);      
253   }
254   
255   this.control_panel_div.style.zIndex = 2000;
256   MVUtil.setCursor(this.control_panel_div,"pointer"); 
257   
258   this.panImgW = 0;
259   this.panImgH = 0;
260   var mv = this.mvHandle ;
261   
262   var navMouseUp = function(evt)
263   {
264     if(mv.marqueeZoomOn && mv.clearMarqueeRect)
265     {
266       mv.marqueeZoomRectangleTool.clear() ;
267       mv.marqueeZoom() ;
268     }
269     evt = (evt) ? evt : ((window.event) ? event : null);
270     MVUtil.stopEvent(evt);
271   }
272   if (this.panButtonFlag != false)
273   {
274     this.addPanButtons() ;
275   }
276   this.sliderBarW = 0;
277   this.sliderBarH = 0;  
278   if (this.sliderBarFlag != false)
279   {
280     this.sliderBarW = MVGlobalVariables.sliderBarW;
281     this.sliderBarH = MVGlobalVariables.sliderBarH;     
282   }
283   
284   this.zoomInOutW = 0;
285   this.zoomInOutH = 0;      
286   if (this.zoomButtonFlag != false)
287   {
288     this.zoomInOutW = MVGlobalVariables.zoomInOutW;
289     this.zoomInOutH = MVGlobalVariables.zoomInOutH;    
290     var pZoomIn = new createDIVMarkerNode(this.control_panel_div, src=MVGlobalVariables.imgFilePath + "navicons/zoomin.gif",
291                   this.zoomInOutW, this.zoomInOutH, 2000, MVi18n.getInfoMessage("zoomInInfoTip"));
292     var l = this.panButtonFlag >0 ? (this.panImgW * 3 - this.zoomInOutW)/2 + 1 : 0 ;
293     var t = this.panButtonFlag >0 ? this.panImgH * 3 + 3 : 0;
294     MVUtil.setLeftTop(pZoomIn, l, t);
295     MVUtil.attachEvent(pZoomIn, "mousedown", MVUtil.closureWrapper(pZoomIn, function(evt)
296     {
297       evt = (evt) ? evt : ((window.event) ? event : null);
298       MVUtil.stopEvent(evt);
299       if(mv.marqueeZoomOn && mv.marqueeRectDone)
300         mv.clearMarqueeRect = true ;
301       mv.zoomIn();
302     }), true);
303     MVUtil.attachEvent(pZoomIn, "mouseup", MVUtil.closureWrapper(pZoomIn, function(evt)
304     {
305       navMouseUp(evt) ;
306     }), true);
307     
308     pZoomIn.onmouseover = this.showInfoTip;
309     pZoomIn.onmouseout = this.hideInfoTip;
310     
311     var pZoomOut = new createDIVMarkerNode(this.control_panel_div,MVGlobalVariables.imgFilePath + "navicons/zoomout.gif", 
312                   this.zoomInOutW, this.zoomInOutH, 2000, MVi18n.getInfoMessage("zoomOutInfoTip"));
313     MVUtil.setLeftTop(pZoomOut, l, t + this.zoomInOutH + this.sliderBarH+1);
314     MVUtil.attachEvent(pZoomOut, "mousedown", MVUtil.closureWrapper(pZoomOut, function(evt)
315     {
316       evt = (evt) ? evt : ((window.event) ? event : null);
317       MVUtil.stopEvent(evt);
318       if(mv.marqueeZoomOn && mv.marqueeRectDone)
319         mv.clearMarqueeRect = true ;
320       mv.zoomOut();
321     }), true);  
322     MVUtil.attachEvent(pZoomOut, "mouseup", MVUtil.closureWrapper(pZoomOut, function(evt)
323     {
324       navMouseUp(evt) ;
325     }), true);
326     
327     pZoomOut.onmouseover = this.showInfoTip;
328     pZoomOut.onmouseout = this.hideInfoTip;
329   }
330   if (this.sliderBarFlag != false)
331   {
332     this.zoombarandpin = new MVZoomBarAndPin(this);
333   }
334   this.initialized = true ;
335 }

539 {
540   var parent = par; 
541   this.div = document.createElement("div");
542   
543   var l = parent.panButtonFlag > 0 ? (parent.panImgW * 3 - parent.sliderBarW)/2 + 1 : 0 ;
