 21 /**
 22   @class MVBucketStyle defines an ADVANCED type style that is defined through a series of buckets with their
 23   associated values and styles.
 24   @constructor
 25   @param {String} name name of this Bucket style
 26   @param {Object} buckets an instance of BucketSeries which contains all buckets
 27 */  
 28 function MVBucketStyle(name, buckets)
 29 {
 30   /**@private*/
 31   this.name = name;
 32   /**@private*/
 33   this.buckets = buckets;
 34 }
 35 
 36 /**
 37  *@private
 38  */
 39 MVBucketStyle.prototype.getXMLString = function()
 40 {
 41   var s = '<style name="'+this.name+'"><AdvancedStyle><BucketStyle>';
 42   s += this.buckets.getXMLString() + "</BucketStyle></AdvancedStyle></style>";
 43   
 44   return s;
 45 }
