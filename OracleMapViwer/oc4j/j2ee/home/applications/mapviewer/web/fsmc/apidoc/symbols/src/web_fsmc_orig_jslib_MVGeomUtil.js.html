 22 /** @private */

 24 /** @private */
 25 MVGeomUtil.getPointAtBearing = function(point, dist, bearing)
 26 {
 27   var point = MVGeomUtil.getRadiansFromLatLon(point);
 28   var lon1 = point.x ;
 29   var lat1 = point.y ;
 30   var d = dist/MVGeomUtil.earthRadius ;
 31   var lat2 = Math.asin(Math.sin(lat1)*Math.cos(d) + Math.cos(lat1)*Math.sin(d)*Math.cos(bearing)) ;
 32   var lon2 =  lon1 + Math.atan2(Math.sin(bearing)*Math.sin(d)*Math.cos(lat1), Math.cos(d)-Math.sin(lat1)*Math.sin(lat2)) ;
 33   return MVGeomUtil.getLatLonFromRadians(new MVSdoPointType(lon2, lat2, 0));
 34 }
 35 /** @private */
 36 MVGeomUtil.getRadiansFromLatLon = function(point)
 37 {
 38   var lon = point.x ;
 39   var lat = point.y ;
 40   while(lon<-180) lon+=360;
 41   while(lon> 180) lon-=360;
 42   lon = lon* Math.PI/180;
 43   lat = lat*Math.PI/180;
 44   return new MVSdoPointType(lon, lat, 0) ;
 45 }
 46 /** @private */
 47 MVGeomUtil.getLatLonFromRadians = function(point)
 48 {
 49   var lon = point.x ;
 50   var lat = point.y ;
 51   lat = lat*180/Math.PI ;
 52   lon = lon*180/Math.PI ;
 53   return new MVSdoPointType(lon, lat, 0) ;
 54 }
 55 /** @private */
 56 MVGeomUtil.getGreatCircleDistance = function(point1, point2)
 57 {
 58   point1 = MVGeomUtil.getRadiansFromLatLon(point1) ;
 59   point2 = MVGeomUtil.getRadiansFromLatLon(point2) ;
 60   var lonDif = point2.x - point1.x ;
 61   var latDif = point2.y - point1.y ;
 62   var slat = Math.sin(latDif/2) ;
 63   var slon = Math.sin(lonDif/2) ;
 64   var a = slat*slat + Math.cos(point1.y)*Math.cos(point2.y)*slon*slon;
 65   var c = 2*Math.atan2(Math.sqrt(a), Math.sqrt(1-a)) ;
 66   return c*MVGeomUtil.earthRadius ;
 67 }
 68 /** @private */
 69 MVGeomUtil.getGreatCircleLength = function(ords)
 70 {
 71   var length = 0 ;
 72   for(i=0; i<ords.length-2; i+=2)
 73   {
 74     length += MVGeomUtil.getGreatCircleDistance(new MVSdoPointType(ords[i], ords[i+1]), new MVSdoPointType(ords[i+2], ords[i+3])) ;
 75   }
 76   return length ;
 77 }

 87        over (n-2)*Pi where n is the number of sides. The polygon may be concave
 88        or convex. Result is in the square of the same unit as the semi-major axis in

 90 */

106     i %= num;

108     var pj = MVGeoUtil.get3DVector(q[2*j], q[2*j+1]) ;
109     var pi = MVGeoUtil.get3DVector(q[2*i], q[2*i+1]) ;
110     var pk = MVGeoUtil.get3DVector(q[2*k], q[2*k+1]) ;
111     // Accumulate sum of sines of latitude (z)
112     //zAvg += *dP;
113     zAvg += pj.z ;
114     // Compute interior angle at current vertex
115     cross1 = MVGeoUtil.mdsphcp( pj, pj);
116     mag2 = MVGeoUtil.MAGNITUDE(cross1);
117     cross2 = MVGeoUtil.mdsphcp( pj, pk);
118     mag2 *= MVGeoUtil.MAGNITUDE(cross2);
119     argacos = MVGeoUtil.DOTPRODUCT(cross1,cross2)/mag2;
120     if(argacos >  1.) argacos =  1.;
121     if(argacos < -1.) argacos = -1.;

123     // If the cross product of the cross products points out from sphere,
124     // we have a non-convex vertex and the angle must be supplemented      
125     crossCross = MVGeoUtil.mdsphcp(cross1, cross2);
126     if(Math.abs(crossCross.x) < 1.e-9*mag2 && Math.abs(crossCross.y) < 1.e-9*mag2 &&

133     {

137         angleSum += Math.PI - angle;
138     }
139   }

142   // Done with vertices. The accumulated angleSum will either be
143   //    -2PI+excess  or  2PI-excess , the latter if the polygon is traced clockwise.

173 /*******************************************************************
174 Function: mdsphcp        CrossProduct
175 Purpose: 3-D vector cross product computation.
176 Author: B. Blackwell
177 Revisions: 09 Nov 99 - Original Version
178 Notes: 
179 ********************************************************************/
180 MVGeomUtil.mdsphcp = function(a, b)
181 {
182   var r = {"x":0, "y":0, "z":0} ;
183   if ( (a.x == b.x) && (a.y == b.y) && (a.z == b.z) )
184   {
185     r.x = r.y = r.z = 0.0;
186   }
187   else
188   {
189     r.x = a.y * b.z - a.z * b.y;
190     r.y = a.z * b.x - a.x * b.z;
191     r.z = a.x * b.y - a.y * b.x;
192   }
193   return r ;
194 }
195 /** @private*/
196 MVGeomUtil.get3DVector = function(x, y)
197 {
198   var p = {"x":x, "y":y} ;
199   var radians = MVGeomUtil.getRadiansFromLatLon(p);
200   radians.z = Math.sin(radians.y) ;
201   return radians ;
202 }
203 MVGeomUtil.MAGNITUDE = function(a)
204 {
205   return Math.sqrt(a.x*a.x+a.y*a.y+a.z*a.z) ;
206 }
207 MVGeomUtil.DOTPRODUCT = function(a,b)
208 {
209   return a.x*b.x+a.y*b.y+a.z*b.z ;
210 }
