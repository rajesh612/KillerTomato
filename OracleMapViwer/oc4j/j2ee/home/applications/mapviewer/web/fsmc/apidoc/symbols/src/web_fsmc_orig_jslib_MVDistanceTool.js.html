 34 /**
 35  *@class MVDistance defines a distance tool instance with which the user can 
 36  * calculate the straight-line distance between the points generated by clicking 
 37  * and moving the mouse on the map repeatedly. The total distance info is displayed 
 38  * in a tooltip around the last shape point in readable way.
 39  *
 40  *<p>When get all the distance info,  the user can double click the map to finish the
 41  * distance tool; When finished the distance tool, click on the map will not create 
 42  * any shape points and lines</p>
 43  *
 44  *<p>Once created, the line and distance info tooltip stays on the map
 45  *until the application invokes function MVDistanceTool.clear to remove these
 46  *feature from the map.</p>
 47  *
 48  *@constructor
 49  *@param {Integer} unitStandard A Integer that specifies the unit standard
 50  *                          uesd to display the distance info, the value should be
 51  *                          MVDistanceTool.METRIC ; or  MVDistanceTool.IMPERIAL ;
 52  *                          When use MVDistanceTool.METRIC , the distance will be displayed
 53  *                          in KM,M and CM; when use MVDistanceTool.METRIC, the distance
 54  *                          will be displayed in mi,yd,ft.
 55  *@param {String} foiServerURL An optional String parameter that 
 56  *                          specifies the URL of the FOI server that renders
 57  *                          the distance features(line). The FOI server 
 58  *                          URL should be in the form http://host:port/mapviewer/foi. If omitted, 
 59  *                          the URL of the FOI server is derived 
 60  *                          from the MapViewer base URL.
 61  *@return The newly created MVDistanceTool object.
 62 */
 63 function MVDistanceTool(unitStandard, foiServerURL)
 64 {
 65     var distance = this;
 66     
 67     /**@private*/
 68     this.displayDistance = true;
 69     /**@private*/
 70     this.doubleClickFinish = true;
 71     
 72     
 73     /**@private*/
 74     this.unitStandard = unitStandard;
 75 
 76     /**@private*/
 77     this.serverURL = foiServerURL;
 78     /**@private*/
 79     this.onFinishListener = null ;
 80     /**@private*/
 81     this.onNewShapePointListener = null ;
 82     /**@private*/
 83     this.onEditFinishListener = null;
 84     
 85     /**@private*/
 86     this.EventListeners = new Object();
 87     
 88     /**@private*/
 89     this.srid = null;
 90     /**@private*/
 91     this.tmpDouClick = null;
 92     /**@private*/
 93     this.redLineTool = new MVRedlineTool(null, null, foiServerURL);
 94     this.redLineTool.setGeneratePolygonTop(false);
 95     /**
 96      *This method initializes the distance tool instance. It should be the last method to call
 97      *before drawing on the map. After this function is invoked, the user can start getting 
 98      *distance info on the map by clicking and moving the mouse repeatedly. 
 99      *@return none
100      *@type void
101     */
102     this.init=function()
103     {
104         distance.parent.addRedLineTool(distance.redLineTool);
105         distance.srid = distance.parent.srid;
106     
107         if(distance.redLineTool.getStatus() != 0)
108         {
109             distance.clear();
110         }
111         
112         distance.redLineTool.finishLinkClick = distance.finish;
113         distance.redLineTool.clearLinkClick = distance.clear;
114         
115         distance.redLineTool.init(2);
116         distance.redLineTool.moveFoiFlag = -1;
117         
118         distance.redLineTool.setEventListener(MVEvent.NEW_SHAPE_POINT ,distance.calTotalDistance);
119         distance.redLineTool.setEventListener(MVEvent.MODIFY,distance.afterEditDistanceTool);
120         
121         if(distance.doubleClickFinish)
122         {
123             distance.tmpDouClick = distance.parent.dblclickListener;
124             distance.parent.setEventListener(MVEvent.MOUSE_DOUBLE_CLICK,distance.finish);
125         }
126 		
127     		MVUtil.runListeners(this.EventListeners,MVEvent.INIT,[MVToolBar.BUILTIN_DISTANCE]);
128     }
129     
130     /**
131      *@private
132      *This method is deprecated. Please use {@link #sattachEventListener} instead.
133      *@deprecated 
134     */
135     this.setEventListener = function(event, listener)
136     {
137         if(event== MVEvent.NEW_SHAPE_POINT)
138             distance.onNewShapePointListener = listener;
139         else if(event== MVEvent.FINISH)
140             distance.onFinishListener = listener ;
141         else if(event==MVEvent.MODIFY)
142             distance.onEditFinishListener = listener ;
143     }
144    /**
145      *This method attaches an event listener to handle one of the following distance tool events: 
146      MVEvent.NEW_SHAPE_POINT , MVEvent.FINISH and MVEvent.MODIFY
147      *<ul>
148      *<li>
149      *A MVEvent.NEW_SHAPE_POINT event is fired when the user creates a new distance shape point
150      *by clicking the mouse on the map. No parameter is passed to this function when 
151      *it is invoked.
152      *</li>
153      *<li>
154      *A MVEvent.FINISH event is fired when the user deactives/finishes the distance tool. 
155      *</li>
156      *<li>
157      *A MVEvent.MODIFY event is fired when the user moves any of the existing shape points or lines
158      *of the distance tool. 
159      *</li>
160      *</ul>
161      *@param {String} event a String that specifies the type of event to be
162      * handled by the listener. It can only be MVEvent.NEW_SHAPE_POINT or MVEvent.FINISH or MVEvent.MODIFY
163      *@param {Function} listener a javascript listener function. There is no
164      *parameter passed to the listener when it is called.
165      *@return none
166      *@type void
167     */    
168     this.attachEventListener = function(type, listener)
169     {
170         MVUtil.attachEventListener(distance.EventListeners,type,listener)
171     }
172    /**
173      *This method detaches an event listener that has previously been attached to the distance tool 
174      *by {@link #attachEventListener}. 
175      *@param {String} event a String that specifies the type of event to be
176      * handled by the listener. It can only be MVEvent.NEW_SHAPE_POINT or MVEvent.FINISH or MVEvent.MODIFY
177      *@param {Function} listener a javascript listener function. There is no
178      *parameter passed to the listener when it is called.
179      *@return none
180      *@type void
181     */
182     this.detachEventListener = function(type, listener)
183     {
184         MVUtil.detachEventListener(distance.EventListeners,type, listener);
185     }

187     /**
188     * This method sets the style for the distance tool shape points, lines or polygon area.  
189     *@param {String} objType It specifies the object for which a style is applied to. 
190     *                        Its value can only be MVRedlineTool.STYLE_POINT, MVRedlineTool.STYLE_LINE or MVRedlineTool.STYLE_AREA. 
191     *@param {Object} objStyle A string or style object which specifies the rendering style. 
192                              If it is a string, the string should be in the form of <data-source>.<style-name>, 
193                              where <style-name> is the name of the style which is 
194                              defined in the USER_SDO_STYLES view, 
195                              and <data-source> is the name of the 
196                              data source in which the style is defined.
197                              If it is a style object, it should be of type MVStyleMarker, MVStyleColor and MVXMLStyle etc. 
198     *@return none
199     *@type void
200     */
201     this.setRenderingStyle = function(objType, objStyle)
202     {
203       this.redLineTool.setRenderingStyle(objType, objStyle) ;
204     }

207     *This method set the distance tool's distance info box div style. It should
208     * be in CSS style string. eg:"font-family:Tahoma;"
209     *@param {String} style CCS style format
210     *@return none
211     *@type void
212     */

216     }

219          This method returns the array of ordinates of the distance shape points.
220          *@return the array of ordinates for the distance shape points.

223     this.getOrdinates=function()

226     }

228     /**
229      *This method gets the status of the distance tool. At any give time, the distance 
230      * tool can be in one of the following statuses:
231      *<ul>
232      *<li>0: Not ready</li>
233      *<li>1: Ready and initialized</li>
234      *<li>2: Distance tool finished distance info got</li>
235      *</ul>
236      @return the status of the distance tool.
237      @type int
238     */
239     this.getStatus=function()
240     {
241         return distance.redLineTool.getStatus();
242     }

244     /**
245      *This method finish the current MVDistance instance. After this method 
246      * is called, mouse clicks no longer draw any shape points or lines on the map.
247      *@return none
248      *@type void
249     */
250     this.finish=function(evt)
251     {
252       if (distance.redLineTool!=null && distance.getStatus() != 2)
253       {
254         if(distance.redLineTool.getOrdinates().length<=2)
255         {
256           distance.clear();
257         }
258         else
259         {
260           distance.redLineTool.generateArea();
261           if(distance.onFinishListener)
262             distance.onFinishListener();
263           MVUtil.runListeners(distance.EventListeners,MVEvent.FINISH);
264         }
265       }
266       //MVUtil.stopEvent(evt);
267       distance.parent.dblclickListener = distance.tmpDouClick;
268     }
269     
270     /**
271     *This method clears any lines or foi rendered by the current MVDistance instance. 
272     *@return none
273     *@type void
274     */
275     this.clear=function()
276     {
277       if (distance.redLineTool!=null)
278       {
279         //if(distance.redLineTool.getOrdinates().length>=2 &&
280         //   distance.redLineTool.getStatus() == 1)
281         //{
282         //    if(distance.onFinishListener)
283         //        distance.onFinishListener();
284         //    MVUtil.runListeners(distance.EventListeners,MVEvent.FINISH);
285         //}
286                   
287         distance.redLineTool.clear();
288         
289         var infoDecorationDiv = document.getElementById("distance_info_div_"+distance.redLineTool.id) ;
290         if(infoDecorationDiv)
291         	infoDecorationDiv.innerHTML = "";
292         	
293         MVUtil.runListeners(distance.EventListeners,MVEvent.CLEAR);
294         distance.parent.dblclickListener = distance.tmpDouClick;
295       }
296     }

298     /**@private*/
299     this.calTotalDistance=function()
300     {
301         if(distance.displayDistance)
302         {
303             if (distance.redLineTool == null)
304                 return;
305             var ords = distance.redLineTool.getOrdinates();
306             if (ords == null || ords.length<4)
307             {
308                 if(distance.onNewShapePointListener)
309                     distance.onNewShapePointListener();
310                 MVUtil.runListeners(distance.EventListeners,MVEvent.NEW_SHAPE_POINT);
311                 return;
312             }
313 
314             var line = MVSdoGeometry.createLineString(ords,distance.srid);
315 			
316 			var setDistanceInfo = function(totalDistance)
317 			{			
318 				var text = MVi18n.getInfoMessage("totalDistance") + distance.getReadableResult(totalDistance) ;
319 				var infoDecorationDiv = document.getElementById("distance_info_div_"+distance.redLineTool.id) ;
320 				if(!infoDecorationDiv)
321 				{
322 				  infoDecorationDiv = document.createElement("div") ;
323 				  infoDecorationDiv.id = "distance_info_div_"+distance.redLineTool.id ;
324 				  distance.redLineTool.getInfoDecorationDiv().appendChild(infoDecorationDiv);
325 				}
326 				infoDecorationDiv.innerHTML = text;
327 			}
328 			
329             if(distance.unitStandard == MVDistanceTool.METRIC )
330                 parseInt( line.getLength("meter",true,distance.serverURL,setDistanceInfo) );
331             else
332                 parseInt( line.getLength("Yard",true,distance.serverURL,setDistanceInfo) );
333         }
334         if(distance.onNewShapePointListener)
335             distance.onNewShapePointListener();
336         MVUtil.runListeners(distance.EventListeners, MVEvent.NEW_SHAPE_POINT);
337     }
338     
339     /**@private*/
340     this.getDistance = function(unit, transformToGeodetic, foiURL)
341     {
342         if (distance.redLineTool == null)
343             return 0;
344         var ords = distance.redLineTool.getOrdinates();
345         if (ords == null || ords.length<4)
346             return 0;
347         var length = ords.length;
348         var lineOrds = new Array(4);
349         for (var i=0;i<4;i++)
350             lineOrds[i] = ords[length+i-4];
351         var line = MVSdoGeometry.createLineString(lineOrds,distance.srid);
352 		
353 		var distanceInfo;
354 		var setDistanceInfo = function(totalDistance)
355 		{
356 			distanceInfo = totalDistance;
357 		}
358         
359 		parseInt( line.getLength(unit,transformToGeodetic,foiURL,setDistanceInfo) );
360 		
361 		return distanceInfo;
362     }
363 
364     /**@private*/
365     this.getTotalDistance = function(unit, transformToGeodetic, foiURL)
366     {
367         if (distance.redLineTool == null)
368             return 0;
369         var ords = distance.redLineTool.getOrdinates();
370         if (ords == null || ords.length<4)
371             return 0;
372         var line = MVSdoGeometry.createLineString(ords,distance.srid);
373 		
374 		var totalDistanceInfo;
375 		var setDistanceInfo = function(totalDistance)
376 		{
377 			totalDistanceInfo = totalDistance;
378 		}
379         parseInt( line.getLength(unit,transformToGeodetic,foiURL,setDistanceInfo) );
380 		
381 		return totalDistanceInfo;
382     }
383     
384     /**@private*/
385     this.getReadableResult=function(distanceValue)
386     {
387         //1ft=12in;1yd=3ft;1mi=1760yd=5280ft
388         if(distance.unitStandard == MVDistanceTool.METRIC)
389         {
390             if(distanceValue>1000)
391                 return MVUtil.formatNum(distanceValue/1000) + MVi18n.getInfoMessage("km");
392             else if(distanceValue > 1)
393                 return MVUtil.formatNum(distanceValue) + MVi18n.getInfoMessage("m");
394             else
395                 return MVUtil.formatNum(distanceValue*10) + MVi18n.getInfoMessage("cm");
396         }
397         else
398         {
399             if(distanceValue>1760)
400                 return MVUtil.formatNum(distanceValue/1760) + MVi18n.getInfoMessage("mi");
401             else if(distanceValue*3 > 1000)
402                 return MVUtil.formatNum(distanceValue) + MVi18n.getInfoMessage("yd");
403             else if(distanceValue*3 > 1) 
404                 return MVUtil.formatNum(distanceValue*3) + MVi18n.getInfoMessage("ft");
405             else 
406                 return MVUtil.formatNum(distanceValue*3*12) + MVi18n.getInfoMessage("in");
407         }
408     }
409     
410     /**@private*/
411     this.afterEditDistanceTool=function()
412     {
413         if(distance.displayDistance)
414             distance.calTotalDistance();
415         else
416         {
417             if(distance.onEditFinishListener)
418                 distance.onEditFinishListener();
419             MVUtil.runListeners(distance.EventListeners,MVEvent.MODIFY);
420         }
421     }
