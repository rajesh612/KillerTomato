 33 /**
 34  *@class MVToolBar defines a draggable toolbar that acts as a placeholder of 
 35  *       multiple control buttons. Each control button controls an Oracle Maps
 36  *       built-in tool or a custom tool and responses to mouse clicks by invoking
 37  *       event handlers associated with the tool. The toolbar can be displayed
 38  *       in two orientations, vertical or horizontal.
 39  *
 40  * <p>The following built-in tools are supported: Redline tool, Distance tool, 
 41  * Circle tool, Rectangle tool and Marquee Zoom tool. The application can choose
 42  * to display control buttons for all or some of these built-in tools in the toolbar.</p>
 43  *
 44  * <p> The tool bar can also hold custom tool buttons that controls application 
 45  *     defined operations.</p>
 46  *
 47  * A tool button can be defined in one of the following two types: 
 48  * <UL>
 49  *   <LI> COMMAND:   A command button has only one state, UP, and always reacts to mouse 
 50  *                   clicks by invoking the same BUTTON_DOWN event listener.
 51  *   </LI>
 52  *   <LI> TOGGLE:    A toggle button has two states, DOWN and UP. The initial state of
 53  *                   the button is UP. Its state changes to DOWN after it is clicked. 
 54  *                   It stays DOWN until it is clicked again, after which its state
 55  *                   changes to UP. The BUTTON_DOWN event listener is invoked when the 
 56  *                   button state changes from UP to DOWN. The BUTTON_UP event listener
 57  *                   is invoked when the button state changes from DOWN to UP. 
 58  *   </LI>
 59  *   <LI> SEPERATOR: A seperator simply displays a static seperator icon that does not
 60  *                   react to any user actions. 
 61  * </UL>
 62  *
 63  * <P>Multiple buttons can organized as a button group, 
 64  * where at most one button can be in the pushed DOWN state at any time. Pushing down a 
 65  * button will automatically pop up the other pushed DOWN button if there is any. All built-in
 66  * tool buttons, except the Marquee Zoom button, are TOGGLE buttons organized in the 
 67  * same button group. The Marquee Zoom button is a TOGGLE button, but does not belong to any
 68  * button group.</P>
 69  *
 70  *@constructor
 71  *@param {String} toolBarID A string that specifies the id of the toolbar.
 72  *@param {array}  builtInButtons An array that specifies the built-in tool buttons to be 
 73  *                included in the tool bar. Each array element specifies one 
 74  *                built-in tool bar to be included. It can be one of the following values, 
 75  *                MVToolBar.BUILTIN_CIRCLE, MVToolBar.BUILTIN_RECTANGLE, MVToolBar.BUILTIN_REDLINE,
 76  *                MVToolBar.BUILTIN_DISTANCE, MVToolBar.BUILTIN_MARQUEE_ZOOM, MVToolBar.BUILTIN_ALL. 
 77  *                To include all built-in tool buttons, you can use an array with a single 
 78  *                MVToolBar.BUILTIN_ALL element.
 79  *@param {int}    orientation This parameter specifies the orientation of the tool bar. It can be
 80                   MVToolBar.VERTICAL or MVToolBar.HORIZONTAL.
 81  *@return The newly created MVDistanceTool object.
 82  */

280     /**@private*/
281     this.addButtonToDOM = function(btObject)
282     {
283       var oTR = null;
284       var oTD = null;
285       
286       if(toolBar.Direction == MVToolBar.HORIZONTAL)
287       {    
288           if(toolBarTB.childNodes.length == 0)
289           {
290               oTR= toolBarTB.insertRow(0);
291           }
292           else
293           {
294               oTR = toolBarTB.firstChild.firstChild;
295           }
296           oTD= oTR.insertCell(oTR.childNodes.length); 
297           oTD.id = "_td_" + btObject.btID
298       }
299       else
300       {
301           if(toolBarTB.childNodes.length == 0)
302               oTR= toolBarTB.insertRow(0);
303           else
304               oTR= toolBarTB.insertRow(toolBarTB.firstChild.childNodes.length);
305           
306           oTR.id = "_tr_" + btObject.btID;    
307           oTD= oTR.insertCell(0);          
308       }
309       
310       btObject.parentToolBar = toolBar;
311       oTD.appendChild(btObject.btDiv);
312 
313       toolBar.allToolBts[btObject.btID] = btObject;
314       btObject.parentToolBar = toolBar;
315     }

386     /**
387      * This method deletes a button from the tool bar.
388      * @param {MVToolButton} btObject btObject specifies the button object 
389      *        to be deleted.
390      * @returns None
391      * @type void
392      */
393     this.removeButton = function(btObject) 
394     {
395         if(!btObject)
396           return ;
397         delete toolBar.allToolBts[btObject.btID];
398         if(btObject.group)
399         {
400           delete btObject.group.buttonTools[btObject.btID] ;
401           btObject.group = null ;
402         }
403         if(toolBar.Direction == MVToolBar.HORIZONTAL)
404         {    
405             if(toolBarTB.childNodes.length == 0)
406             {
407                 return;
408             }
409             else
410             {
411                 oTR = toolBarTB.firstChild.firstChild;
412                 for(var i=0; i<oTR.childNodes.length;i++)
413                 {
414                     if( oTR.childNodes[i].id == "_td_" + btObject.btID)
415                     {
416                         oTR.deleteCell(i);
417                     }
418                 }
419             }     
420         }
421         else
422         {
423             if(toolBarTB.childNodes.length == 0)
424             {
425                 return;
426             }
427             else
428             {
429                 for(var i=0; i<toolBarTB.firstChild.childNodes.length;i++)
430                 {
431                     if( toolBarTB.firstChild.childNodes[i].id == "_tr_" + btObject.btID)
432                     {
433                         toolBarTB.firstChild.deleteRow(i);
434                     }
435                 }
436             }
437         }
438     }

473      * @type void
474      */
475     this.removeButtonGroup = function(group)
476     {
477       if(group)
478       {
479         for(var i=0; i<this.buttonGroups.length; i++)
480         {
481           if(this.buttonGroups[i] == group)
482           {
483             this.buttonGroups.splice(i, 1) ;
484             for(var name in group.buttonTools)
485             {
486               this.removeButton(group.buttonTools[name]); 
487             }
488             return ;
489           }
490         }
491       }
492     }

594      * @returns None
595      * @type void
596      */
597     this.addBuiltInButton = function(btObject)
598     {
599 		var enableBuiltInBts = [false,false,false,false,false,false];
600 		
601         if(btObject==null && toolBar.builtInBts)
602         {
603             //it need to clean the built-in tool collection
604             //the user may add MVToolBar.BUILTIN_ALL and some other built-in tools
605             //Or just add one built-in tool twice
606             for(var i = 0; i < toolBar.builtInBts.length; i++)
607             {
608               enableBuiltInBts [ toolBar.builtInBts[i] ] = true;
609             }
610             if(enableBuiltInBts[MVToolBar.BUILTIN_ALL] == true)
611             {
612               enableBuiltInBts = [true, true, true, true, true,true] ;
613             }
614             
615             toolBar.builtInBts = enableBuiltInBts;
616             
617             //if anyone of the built-in tool is added to toolbar, the clear tool should be add
618             for(var i = 0; i < enableBuiltInBts.length; i++)
619             {
620                 if(enableBuiltInBts[i])
621                 {
622                     toolBar.addButton(toolBar.DrawingClearButton);
623                     toolBar.clearButtonAdded = true ;
624                     break;
625                 }
626             }
627 		}		
628 		else if(btObject != null)
629 		{
630 			if(toolBar.builtInBts == null)
631 			{
632 				toolBar.builtInBts = [false,false,false,false,false,false];
633 				toolBar.builtInBts[btObject.btName]=true;
634 				enableBuiltInBts = toolBar.builtInBts;
635 				
636 				//if anyone of the built-in tool is added to toolbar, the clear tool should be add
637 				for(var i = 0; i < enableBuiltInBts.length; i++)
638 				{
639 					if(enableBuiltInBts[i])
640 					{
641 						toolBar.addButton(toolBar.DrawingClearButton);
642 						toolBar.clearButtonAdded = true ;
643 						break;
644 					}
645 				}
646 				
647 			}
648 			else if(toolBar.builtInBts[btObject.btName]==true)
649 				return;
650 			else if(toolBar.builtInBts[btObject.btName]==false)
651 			{
652 				toolBar.builtInBts[btObject.btName]=true;
653 				enableBuiltInBts = [false,false,false,false,false,false];
654 				enableBuiltInBts[btObject.btName]=true;
655 			}
656 		}			
657 			
658 		var toolFinished = function(toolName)
659 		{
660 			for(var toolBt in toolBar.allToolBts)
661 			{
662 				if(toolBar.allToolBts[toolBt].tool)
663 				{    
664 					var tmpToolID = toolBar.allToolBts[toolBt].btID;
665 					if( tmpToolID.indexOf(toolName) != -1 )
666 					{
667 						toolBar.allToolBts[toolBt].forceBtUp();
668 					}
669 				}
670 			}
671 		}
672 		
673 		var redlineFinished = function()
674 		{
675 			toolFinished("redline");
676 		}
677 		var distanceFinished = function()
678 		{
679 			toolFinished("distance");
680 		}
681 		var circleFinished = function()
682 		{
683 			toolFinished("circle");
684 		}
685 		var rectangleFinished = function()
686 		{
687 			toolFinished("rectangle");				
688 		}
689 		
690 		var redlineClear = function()
691 		{
692 			if(toolBar.RedlineToolButton.tool.getPointFOIs().length > 0)
693 				toolBar.RedlineToolButton.tool.clear();
694 			toolFinished("redline");
695 		}
696 		
697 		var distanceClear = function()
698 		{
699 			if(toolBar.DistanceToolButton.tool.redLineTool.getPointFOIs().length > 0)
700 				toolBar.DistanceToolButton.tool.clear();
701 			toolFinished("distance");
702 		}
703 		
704 		var circleClear = function()
705 		{
706 			toolFinished("circle");
707 			if(toolBar.CircleToolButton.tool.getStatus() != 0)
708 				toolBar.CircleToolButton.tool.clear();
709 		}
710 		
711 		var rectangleClear = function()
712 		{
713 			toolFinished("rectangle");
714 			if(toolBar.RectangleToolButton.tool.getStatus() != 0)
715 				toolBar.RectangleToolButton.tool.clear();
716 		}		
717 		
718 		var marqueeZoomFinished = function()
719 		{
720 			toolFinished("zoom");				
721 		}
722 		
723 		var syncPressDown = function(tool)
724 		{
725 			var button = toolBar.getBuiltInButton(tool);
726 			if(button.btState == MVToolButton.UP)
727 			{
728 				button.group.reset();
729 				button.btImg.setAttribute("src", button.btSelectedIcon);
730 				button.btState = MVToolButton.DOWN;
731 				
732 				if(button.onButtonStateChangeListener)
733 					button.onButtonStateChangeListener(button.btID);
734 			}
735 		}
736 		
737 		for(var i = 1; i < enableBuiltInBts.length; i++)
738 		{
739 			if(enableBuiltInBts[i] == true)
740 			{
741 				toolBar.addSeparator("sp_builtIn_" + i);
742 				if(i == MVToolBar.BUILTIN_CIRCLE)
743 				{
744 					toolBar.addButton(toolBar.CircleToolButton);
745 					toolBar.mapviewer.addCircleTool(toolBar.CircleToolButton.tool);
746 					toolBar.CircleToolButton.tool.attachEventListener(MVEvent.FINISH ,circleFinished);
747 					toolBar.CircleToolButton.tool.attachEventListener(MVEvent.CLEAR ,circleClear);
748 					toolBar.CircleToolButton.tool.attachEventListener(MVEvent.INIT,syncPressDown);
749 				}
750 				else if(i == MVToolBar.BUILTIN_RECTANGLE)
751 				{
752 					toolBar.addButton(toolBar.RectangleToolButton);    
753 
754 					toolBar.mapviewer.addRectangleTool(toolBar.RectangleToolButton.tool);     
755 					toolBar.RectangleToolButton.tool.attachEventListener(MVEvent.FINISH ,rectangleFinished); 
756 					toolBar.RectangleToolButton.tool.attachEventListener(MVEvent.CLEAR ,rectangleClear); 
757 					toolBar.RectangleToolButton.tool.attachEventListener(MVEvent.INIT,syncPressDown);	
758 				}
759 				else if(i == MVToolBar.BUILTIN_REDLINE )
760 				{
761 					toolBar.addButton(toolBar.RedlineToolButton);
762 
763 					toolBar.mapviewer.addRedLineTool(toolBar.RedlineToolButton.tool);
764 					toolBar.RedlineToolButton.tool.attachEventListener(MVEvent.FINISH ,redlineFinished); 
765 					toolBar.RedlineToolButton.tool.attachEventListener(MVEvent.CLEAR ,redlineClear);
766 					toolBar.RedlineToolButton.tool.attachEventListener(MVEvent.INIT,syncPressDown);
767 				}
768 				else if(i == MVToolBar.BUILTIN_DISTANCE )
769 				{
770 					toolBar.addButton(toolBar.DistanceToolButton);                        
771 					toolBar.mapviewer.addDistanceTool(toolBar.DistanceToolButton.tool);
772 					toolBar.DistanceToolButton.tool.attachEventListener(MVEvent.FINISH ,distanceFinished);
773 					toolBar.DistanceToolButton.tool.attachEventListener(MVEvent.CLEAR ,distanceClear);		
774 					toolBar.DistanceToolButton.tool.attachEventListener(MVEvent.INIT,syncPressDown);					
775 				}
776 				else if(i == MVToolBar.BUILTIN_MARQUEE_ZOOM)
777 				{
778 					toolBar.addButton(toolBar.MarqueeZoomToolButton);
779 				}
780 			}
781 		}
782         
783     }

796                         tmpToolID == toolBarID + "_bt_circle" || tmpToolID == toolBarID + "_bt_rectangle") &&
797                         toolBar.allToolBts[toolBt].tool.parent!=null)

821             }
