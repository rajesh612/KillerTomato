 37  /**
 38     @class MVMapTileLayer defines a map tile layer displayed on the client side. 
 39     It assembles and displays pregenerated map image tiles that are rendered by 
 40     the map tile server and belong to a pre-defined server side map tile layer.
 41     
 42     <p>The map tile server is a servlet running as part of the MapViewer server. 
 43     The typical map tile server URL is http://hostname:port/mapviewer/mcserver.</p>
 44     
 45     @constructor
 46     @param {String} mapTileLayerName mapTileLayerName is a string that identifies the 
 47                     map tile layer. It is defined in the form of data-source-name.map-tile-layer-name, 
 48                     where map-tile-layer-name specifies the name of the map tile 
 49                     layer that has been pre-defined on the server side and 
 50                     data-source-name specifies the name of the MapViewer 
 51                     data source in which the map tile layer is defined. 
 52     @param {String} tileServerURL tileServerURL is a string that specifies the URL
 53                     of the map tile server, which should be in the form
 54                     http://host:port/mapviewer/mcserver. This is an optional parameter. When
 55                     omitted, the map tile server URL is derived from the
 56                     MapViewer base URL specified when creating the MVMapView 
 57                     object. 
 58     @returns        The newly created MVMapTileLayer object.
 59     @type           object
 60  */
 61  function MVMapTileLayer(mapTileLayerName, tileServerURL)
 62  {
 63    /**@private*/
 64    this.mapSource = mapTileLayerName;
 65    /***@private*/
 66    this.mcsURL = "";
 67    /**@private*/
 68    this.mcscfgURL = "";
 69    if (tileServerURL && tileServerURL!="")
 70    {
 71      this.mcsURL = tileServerURL;
 72      this.mcscfgURL = tileServerURL;
 73      if (MVUtil.strEndWith(tileServerURL, '/'))
 74      {
 75        this.mcsURL = tileServerURL.substring(0, tileServerURL.length-1);
 76        this.mcscfgURL = tileServerURL.substring(0,tileServerURL.length-1);
 77      }
 78    }
 79 
 80    /**@private*/
 81    this.parent = "";
 82    /**@private*/
 83    this.isTransparent = false;
 84    /**@private*/
 85    this.visible = true;
 86    /**@private*/
 87    this.tileImageFormat = "PNG";
 88    /**@private*/
 89    this.minVisibleLevel = null;
 90    /**@private*/
 91    this.maxVisibleLevel = null;
 92    /**@private*/
 93    this.reloadNumber = null ;
 94    /**@private*/
 95    this.layerControl = null ;
 96    /**@private*/
 97    this.parentLayer = null ;
 98    /**@private*/
 99    this.childrenLayers = null ;
100    /**@private*/
101    this.doNotCombine = true ;
102    /**@private*/
103    this.opacity = 1;
104    /**@private*/
105    this.mapType = 0;
106  }

114  /**
115     This method sets whether the tile layer is combinable. 
116     <p>A combinable tile layer can
117     be combined with other combinable tile layer(s) when a) they are adjacent to each other
118     in the tile layer stack, b) they are served by the same MapViewer server and c) they
119     have the same zoom level and coordinate system definition. </p>
120     
121     <p>When being combined, multiple tile layers will be combined and merged into one by 
122     MapViewer server and displayed in fact by the client as one tile layer. In other words, 
123     at any give location within the map, instead of fetching and overlaying multiple tile 
124     images belonging to multiple tile layers in the browser, the client asks MapViewer server
125     to merge these multiple tiles into one tile image and displays the merged tile image intead. 
126     
127     <p>When there are many tile layers, combining multiple tile layers into one can greatly reduce the number of tile images 
128     fetched and displayed by the client and hence improve client side map rendering performance.
129     A tile layer is not combinable by default. </p>
130     
131     @param {Boolean} combinable True if the map tile layer can be combined with other tile layer(s).
132     @returns        None
133     @type           void
134  */ 
135  MVMapTileLayer.prototype.setCombinable = function (combinable)
136  {
137    if(combinable)
138      this.doNotCombine = false ;
139    else
140      this.doNotCombine = true ;
141  }

158  /**
159    This method sets the visibility of the map tile layer.
160    @param {Boolean} visible visible is a Boolean true (make the map tile layer visible) or false 
161                     (make the map tile layer not visible).
162     @returns        None
163     @type           void
164  */ 
165  MVMapTileLayer.prototype.setVisible = function (visible)
166  {  
167    if (this.parent)
168    {
169      if(this.layerControl)
170      {
171        this.visible = visible;
172        this.layerControl.setVisible(visible);
173      }
174      else if(this.isExtAPITileLayer())
175      {
176        this.visible = visible;
177        if(this.layerDiv)
178        {
179         if(visible)
180         {
181           this.layerDiv.style.visibility = "visible" ;
182           var center = this.parent.getCenter();
183           this.parent.setExternalAPITileLayerCenter(this, center.getPointX(), center.getPointY(), this.parent.zoomLevel) ;
184           //this.display();
185         }
186         else
187           this.layerDiv.style.visibility = "hidden" ;
188        }
189      }
190      else if(this.parentLayer)
191      {
192        var visibilityBefore = this.parentLayer.isVisible() ;
193        this.visible = visible;
194        var visibilityAfter = this.parentLayer.isVisible() ;
195        if(visibilityBefore != visibilityAfter)
196          this.parentLayer.setVisible(visibilityAfter) ;
197        //set parentLayer transparent property
198        var transparent = this.parentLayer.getTransparent();
199        if (this.parentLayer.layerControl)
200          this.parentLayer.layerControl.isTransparent = transparent;
201        if(visibilityAfter)
202          this.parentLayer.refresh(true) ;
203      }
204      this.parent.showEmbeddedCopyright();
205    }
206    else
207      this.visible = visible;
208  }

210  /**
211    This method gets the visibility of the map tile layer. 
212     @returns        true if the map tile layer is currently visible, false if the 
213                     map tile layer is not currently visible.
214     @type           Boolean 
215  */
216  MVMapTileLayer.prototype.isVisible = function ()
217  {
218    if(!this.childrenLayers)
219      return this.visible;
220    else
221    {
222      for(var i=0; i<this.childrenLayers.length; i++)
223      {
224        if(this.childrenLayers[i].visible)
225          return true ;
226      }
227      return false ;
228    }
229  }

249  /**
250    This method reloads all map tiles currently displayed from the server.
251    @param {Boolean} reuseBrowserCache It specifies whether to reuse tiles cached in the local browser cache. 
252    @returns        None
253    @type           void
254  */
255  MVMapTileLayer.prototype.refresh = function(reuseBrowserCache)
256  {
257    if(!reuseBrowserCache)
258      this.reloadNumber = Math.round(Math.random()*10000000000) ;
259    if (this.parent)
260    {
261      if(this.layerControl && !this.parentLayer)
262      {
263        if(this.layerControl.zoomControl)
264           this.layerControl.zoomControl.removeCloneTiles() ;
265        this.layerControl.initTiles(this.parent.getPaneWidth(), this.parent.getPaneHeight(), 
266                                    this.parent.mapCenterX, this.parent.mapCenterY);
267      }
268    }
269  }

281  /**
282    @private
283  */
284  MVMapTileLayer.prototype.getMapSource = function ()
285  {
286    if(this.childrenLayers)
287    {
288      var ms = null ;
289      for(var i=0; i<this.childrenLayers.length; i++)
290      {
291        if(this.childrenLayers[i].visible  
292           && (this.childrenLayers[i].minVisibleLevel==null 
293           || this.childrenLayers[i].minVisibleLevel<=this.parent.zoomLevel) 
294           && (this.childrenLayers[i].maxVisibleLevel==null 
295           || this.childrenLayers[i].maxVisibleLevel>=this.parent.zoomLevel))
296        {
297         if(ms)
298           ms += ";"+this.childrenLayers[i].mapSource ;
299         else
300           ms = this.childrenLayers[i].mapSource ;
301        }
302      }
303      return ms ;
304    }
305    else
306      return this.mapSource ;
307  }

312  MVMapTileLayer.prototype.clone = function ()
313  {
314    var mt = new MVMapTileLayer(this.mapSource, this.mcsURL);
315    mt.isTransparent = this.isTransparent ;
316    mt.tileImageFormat = this.tileImageFormat ;
317    mt.minVisibleLevel = this.minVisibleLevel ;
318    mt.maxVisibleLevel = this.maxVisibleLevel ;
319    mt.doNotCombine = this.doNotCombine ;
320    mt.zIndex = this.zIndex;
321    if(this.config)
322      mt.config = this.config ;
323    return mt ;
324  }

362  */
363  MVMapTileLayer.prototype.getTransparent = function ()
364  {
365    if (this.childrenLayers)
366    {
367      for(var t=0;t<this.childrenLayers.length;t++)
368      {
369        if (!this.childrenLayers[t].isTransparent && this.childrenLayers[t].isVisble)
370        {
371          this.isTransparent = false;
372          return false;
373        }
374      }
375      this.isTransparent = true;
376    }
377    return this.isTransparent;
378  }

380  /**
381    This method sets opacity of tile layer image.
382    @param {int} opacity. The parameter opacity should be a number between 0 and 1. 
383    @returns        None
384    @type           void
385  */
386  MVMapTileLayer.prototype.setOpacity = function(opacity)
387  {
388    if (isNaN(opacity))
389    {
390      return;
391    }
392    if (opacity>1 || opacity<0)
393      return;
394    if (this.opacity == opacity)
395      return;
396    this.opacity = opacity;
397    if(this.layerControl)
398    {
399      this.layerControl.reload();
400      return;
401    }
402    if (this.parentLayer && this.parentLayer.childrenLayers.length>1)
403    {
404      var mv = this.parentLayer.layerControl.parent;
405      mv.removeMapTileLayer(this);
406      mv.addMapTileLayer(this);
407    }
408    else if (this.parentLayer && this.parentLayer.layerControl)
409    {
410      this.parentLayer.opacity = this.opacity;
411      this.parentLayer.layerControl.reload();
412    }
413  }
