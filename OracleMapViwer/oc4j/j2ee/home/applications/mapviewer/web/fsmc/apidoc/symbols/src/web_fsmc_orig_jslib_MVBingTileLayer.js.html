148 /**
149   This method sets the type of the map, which can be one of the following predefined
150   Bing map styles, VEMapStyle.Road, VEMapStyle.Shaded, VEMapStyle.Aerial and VEMapStyle.Hybrid.
151   @param {VEMapStyle} mapType the VEMapStyle constant that specifies the map style.

153   @type void
154 */
155 MVBingTileLayer.prototype.setMapType = function (mapType)
156 {
157   
158   var bing = this;
159   var do_setMapType = function()
160   {
161     bing.mapType = mapType; 
162     if(bing.map)
163     {
164       bing.checkMapType() ;
165       type = MVBingTileLayer.TYPES[bing.mapType] ;
166       if(bing.map)
167         bing.map.SetMapStyle(type) ;
168       
169       if(bing.clonedMap && bing.clonedMap.map)
170         bing.clonedMap.map.SetMapStyle(type) ;      
171     }
172     else
173       setTimeout(do_setMapType,1000);
174   }
175   do_setMapType();  
176 }

203 /**@private*/
204 MVBingTileLayer.prototype.init = function (div)
205 {
206   var bing = this;
207   var init_bing = function(div)
208   {
209     if(MVBingTileLayer.TYPES == null)
210     {
211       MVBingTileLayer.TYPES = new Array() ;
212       MVBingTileLayer.TYPES[MVBingTileLayer.TYPE_ROAD] = VEMapStyle.Road ;
213       MVBingTileLayer.TYPES[MVBingTileLayer.TYPE_SHADED] = VEMapStyle.Shaded;
214       MVBingTileLayer.TYPES[MVBingTileLayer.TYPE_SATELLITE] = VEMapStyle.Aerial;
215       MVBingTileLayer.TYPES[MVBingTileLayer.TYPE_HYBRID] = VEMapStyle.Hybrid;
216     }
217     bing.map = new VEMap(div.id);
218     bing.validateKey() ;
219     var center = null ;
220     if(bing.centerX != null && bing.centerY != null)
221       center = new VELatLong(bing.centerY, bing.centerX);
222     bing.checkMapType() ;
223     bing.map.LoadMap(center, bing.zoomLevel+1, MVBingTileLayer.TYPES[bing.mapType], true);
224     if(bing.initializeListener)
225       bing.initializeListener() ;
226   }  
227     
228   var myInterval = setInterval( 
229     function() {
230     
231       if (( eval("typeof VEMap") != "undefined") && (document.getElementById(div.id).attachEvent != undefined))   
232       {
233         clearInterval(myInterval);  
234         init_bing(div);   
235       }},2000);  
236 }

238 MVBingTileLayer.prototype.setCenterAndZoomLevel = function (x, y, zoomLevel, init)
239 {    
240   this.centerX = x ;
241   this.centerY = y ;
242   this.zoomLevel = zoomLevel ;
243   var bing = this;
244   var do_map = function()
245   {
246     if(bing.map)
247     {
248       bing.map.SetCenterAndZoom(new VELatLong(y, x), zoomLevel+1);
249     }
250     else if(!init)
251       setTimeout(do_map,1000);
252   };
253   do_map();
254 }

326 /**@private*/ 
327 MVBingTileLayer.prototype.validateKey = function ()
328 {
329   var warning = "You have entered an invalid Bing Maps Key! Please replace it with a valid key." ;
330   if(!this.map)
331     return ;
332   this.map.AttachEvent("oncredentialserror", function(){alert(warning)});
333   this.map.SetCredentials(this.key) ;
334 }
335 /**@private*/
336 MVBingTileLayer.prototype.getStaticMapURL=function()
337 {
338   var staticMapURL = "http://dev.virtualearth.net/REST/V1/Imagery/Map/";
339   var c = this.getCenter() ;
340   var center = c.getPointY() + "," + c.getPointX();
341   var mapType;
342   switch(this.mapType){
343     case MVBingTileLayer.TYPE_SATELLITE:
344       mapType = "Aerial" ;
345       break
346     case MVBingTileLayer.TYPE_HYBRID:
347       mapType = "AerialWithLabels" ;
348       break
349     default:
350       mapType = "Road" ;
351   }
352   var zoom = this.zoomLevel + 1;
353   var size = "mapSize=" + this.map.GetWidth() + "," + this.map.GetHeight();
354   var key = "key=" + this.key;
355   staticMapURL = staticMapURL +  mapType +"/" + center + "/"+ zoom + "?" + size + "&" + key;
356   return staticMapURL;
357 }
358 /**@private*/
359 MVBingTileLayer.prototype.checkMapType = function ()
360 {
361   var type = MVBingTileLayer.TYPES[this.mapType] ;
362   if(!type)
363   {
364     for (t in MVBingTileLayer.TYPES)
365     {
366       if(MVBingTileLayer.TYPES[t] == this.mapType)
367       {
368         this.mapType = t ;
369         break ;
370       }
371     }
372   }
373 }
