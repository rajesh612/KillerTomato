 17 /**
 18   @private
 19 */
 20 function MVInfoWindowStyle2()
 21 {
 22 }
 23 /**
 24   @private
 25 */
 26 MVInfoWindowStyle2.init=function()
 27 {
 28   this.window=null;
 29   this.windowDiv=null;
 30   this.iw = MVGlobalVariables.imgFilePath + "/infoicons/";
 31   this.iw_arrow =this.iw + "arrow_circle.gif";
 32   this.iw_h=this.iw+"h.gif";
 33   this.iw_v=this.iw+"v.gif";
 34   this.iw_c=this.iw+"iw_c.gif";
 35   this.iw_c1=this.iw+"iw_n.gif";
 36   this.iw_c2=this.iw+"iw_s.gif";
 37   this.iw_nw=this.iw+"tl.gif";
 38   this.iw_sw=this.iw+"bl.gif";
 39   this.iw_ne=this.iw+"tr.gif";
 40   this.iw_se=this.iw+"br.gif";
 41   this.closeImg=this.iw + "close_circle.gif";
 42   
 43   this.cornerwidth=25;
 44   this.cornerheight=25;
 45   this.arrowwidth=41;
 46   this.arrowheight=40;
 47   this.arrowendheight=20;
 48 }
 49 /**
 50   @private
 51 */
 52 MVInfoWindowStyle2.createWindow=function(x, y,width,height)
 53 {
 54   this.window=new Object();
 55   this.window.arrow=FigLayer(this.iw_arrow,this.arrowwidth+2,this.arrowheight,1,7,0);
 56   this.window.e=FigLayer(this.iw_v,1,height+1-this.cornerheight*2,width+ this.arrowwidth,this.cornerheight,0);
 57   this.window.w=FigLayer(this.iw_v,1,height-this.cornerheight*2-20,this.arrowwidth,this.arrowendheight+this.cornerheight,0);
 58   this.window.wu=FigLayer(this.iw_nw,this.cornerwidth,this.cornerheight,this.arrowwidth,0,0);
 59   this.window.wd=FigLayer(this.iw_sw,this.cornerwidth,25,this.arrowwidth,height+1-this.cornerheight,0);
 60 
 61   this.window.eu=FigLayer(this.iw_ne,this.cornerwidth,this.cornerheight,width+ this.arrowwidth + 1-this.cornerwidth,0,0);
 62   this.window.ed=FigLayer(this.iw_se,this.cornerwidth,25,width+ this.arrowwidth + 1-this.cornerwidth,height+1-this.cornerheight,0); 
 63   this.window.c=FigLayer(this.iw_c,width-1,height+1-this.cornerheight*2,this.arrowwidth+1,this.cornerheight,0);
 64   this.window.cu=FigLayer(this.iw_c1,width+1-this.cornerwidth*2,this.cornerheight,this.arrowwidth+this.cornerwidth,0,0);
 65   this.window.cd=FigLayer(this.iw_c2,width+1-this.cornerwidth*2,25,this.arrowwidth+this.cornerwidth,height-this.cornerheight+1,0);
 66   this.windowDiv = window.document.createElement("div");
 67   this.windowDiv.style.position="relative";
 68   this.windowDiv.style.left=MVUtil.pixel(x -1);
 69   this.windowDiv.style.top=MVUtil.pixel(y + 2);
 70   this.windowDiv.appendChild(this.window.e);
 71   this.windowDiv.appendChild(this.window.wu);
 72   this.windowDiv.appendChild(this.window.w);
 73   this.windowDiv.appendChild(this.window.cu);
 74   this.windowDiv.appendChild(this.window.c);
 75   
 76   this.windowDiv.appendChild(this.window.cd);
 77   this.windowDiv.appendChild(this.window.wd);
 78   this.windowDiv.appendChild(this.window.ed);
 79   this.windowDiv.appendChild(this.window.eu);
 80   this.windowDiv.appendChild(this.window.arrow);
 81   this.width=width;
 82   return this.windowDiv;
 83 }
 84 /**
 85   @private
 86 */
 87 MVInfoWindowStyle2.addCloseButton=function(closefunction)
 88 {
 89   this.closeButton=FigLayer(this.closeImg,16,16,this.width+10,10,4);
 90   this.closeButton.style.position="absolute";
 91   this.closeButton.onclick = MVUtil.closureWrapper(this.closeButton, 
 92                              function(){closefunction();});
 93   MVUtil.setCursor(this.closeButton,"pointer");
 94   this.windowDiv.appendChild(this.closeButton);
 95 }
 96 /**
 97   @private
 98 */
 99 MVInfoWindowStyle2.addContent=function(cdiv)
100 {
101   cdiv.style.left=MVUtil.pixel(55);
102   cdiv.style.top=MVUtil.pixel(17);
103   this.windowDiv.appendChild(cdiv);
104 }
105 /**
106   @private
107 */
108 MVInfoWindowStyle2.getWindow = function()
109 {
110   return this.windowDiv;
111 }
