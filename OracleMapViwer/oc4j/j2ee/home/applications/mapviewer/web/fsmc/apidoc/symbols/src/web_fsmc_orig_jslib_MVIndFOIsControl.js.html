343    This is a static function that creates a marker FOI that displays the
344    user-supplied marker image. The image file must be available online.
345    
346    @param {String} id id is a string that uniquely identifies the marker FOI.
347    @param {MVSdoGeometry} location location is a point MVSdoGeometry object that specifies the 
348                    location of the marker FOI.
349    @param {String} imageURL imageURL is a string that specifies the URL of the
350                    user-supplied marker image.
351    @param {int}    width width is an integer that specifies the native width of the marker FOI image.
352    @param {int}    height height is an integer that specifies the native height of the marker FOI image.
353    @returns        The newly created FOI. 
354    @type           object
355  */
356  MVFOI.createMarkerFOI = function(id, location, imageURL, width, height)
357  {
358    var foi = new MVFOI( id, location, "", imageURL, width, height);
359    foi.createMarker = true;
360    if ((!width || isNaN(width)) || (!height || isNaN(height))){
361      foi.markerImageLoading = true ;
362      var fun = function(w,h){
363        if (!width || isNaN(width))
364          width = w;
365        if (!height || isNaN(height))
366          height = h;
367        foi.width = width;
368        foi.height = height;
369        foi.markerImageLoading = false ;
370        foi = null;
371      }
372      MVUtil.getImageSize(imageURL,fun);
373    }
374    return foi;
375  }
