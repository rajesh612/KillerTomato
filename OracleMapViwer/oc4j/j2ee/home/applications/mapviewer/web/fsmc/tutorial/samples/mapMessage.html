Oracle Maps - error message display options
1. Choose how to display Oracle Maps error messages:

Option 1: display error messages in an alert dialog (default).

Option 2: display error messages in a new pop up window

Option 3: display error messages in the same HTML page

Option 4: do not display any error message.

2. click here to display a sample error message

Instructions

This demo shows how to choose a different reporting mechanism for the built-in error messages of Oracle Maps.

  function showMap()
  {	
    var baseURL  = "http://"+document.location.host+"/mapviewer";
    var mapCenterLon = -122.45;
    var mapCenterLat =  37.6706;
    var mapZoom      =  4;  
    var mpoint = MVSdoGeometry.createPoint(mapCenterLon,mapCenterLat,8307);
    
    var mapview = new MVMapView(document.getElementById("map"), baseURL);     			
    mapview.addMapTileLayer(new MVMapTileLayer("mvdemo.demo_map"));   
    mapview.setCenter(mpoint);   
    mapview.setZoomLevel(mapZoom);

    mapview.display(); 				
  }

  function style1()
  {
  	MVi18n.addBehavior("func", "alert", "text");
  }

  function style2()
  {
  	MVi18n.addBehavior("func", "window.open", "url");
  }

  function style3()
  {
  	MVi18n.addBehavior("func", "messageArea", "text");  	
  }

  function style4()
  {
  	MVi18n.addBehavior("func", "keepSilent", "url");  	
  }

  function messageArea(content)
  {
  	var area=document.getElementById("message");
  	area.innerHTML=area.innerHTML+"
<font color='#FF0000'>"+content+"    "+
  	               (new Date()).toLocaleString()+"</font>";
  }

  function keepSilent(url)
