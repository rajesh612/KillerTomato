 62 /**
 63  *@class MVRedlineTool defines a redline tool instance with which the user can
 64  *draw a polyline by clicking and moving the mouse on the map repeatedly. 
 65  *After the redline tool instance is activated, a new redline shape point is 
 66  *created and the redline polyline is extended whenever the user clicks the 
 67  *mouse on the map. 
 68  *
 69  *<p>After all shape points have been created, the application can invoke function
 70  *MVRedlineTool.MVRedlineTool to render a polygon feature constructed with all 
 71  *redline shape points on the map. </p>
 72  *<p>Once created, the redline feature (polyline or polygon) stays on the map
 73  *until the application invokes function MVRedlineTool.clear to remove the redline
 74  *feature from the map.</p>
 75  *
 76  *@constructor
 77  *@param {String} lineStyle A string that specifies the style used to render the
 78  *                          polyline that connects the redline shape points. The line style is 
 79  *                          defined in the USER_SDO_STYLES view. The style
 80  *                          is specified in the form "data-source-name"."style-name", 
 81  *                          where "style-name" is the name of the style 
 82  *                          and "data-source-name" is the name of the 
 83  *                          data source in which the style is defined.
 84  *@param {String} areaStyle A string that specifies the style used to render the
 85  *                          redline polygon. The area style is 
 86  *                          defined in the USER_SDO_STYLES view. The style
 87  *                          is specified in the form "data-source-name"."style-name", 
 88  *                          where "style-name" is the name of the style 
 89  *                          and "data-source-name" is the name of the 
 90  *                          data source in which the style is defined.
 91  *@param {String} foiServerURL An optional String parameter that 
 92  *                          specifies the URL of the FOI server that renders
 93  *                          the redline features(line and polygon). The FOI server 
 94  *                          URL should be in the form http://host:port/mapviewer/foi. If omitted, 
 95  *                          the URL of the FOI server is derived 
 96  *                          from the MapViewer base URL.
 97  *@return The newly created MVRedlineTool object.
 98 */

388 
389       var lastAreaCode;
390       if(lastPointFOI)
391         lastAreaCode = lastPointFOI.areacode;
392       else
393         lastAreaCode = redline.parent.indFOIsControl.centerArea;
394       //var newAreaCode = newPointFOI.areacode;
395       
396       var newXOrdinate = MVUtil.keepInCSBoundary(redline.parent, newPointFOI.getGeometry().getPointX() );
397       var lastXOrdinate;
398       if(lastPointFOI)
399         lastXOrdinate = MVUtil.keepInCSBoundary(redline.parent, lastPointFOI.getGeometry().getPointX() );
400       else
401         lastXOrdinate = MVUtil.keepInCSBoundary(redline.parent, redline.parent.getCenterLon() );;
402       
403       redline.lineFOI[vIndex *2] = newXOrdinate;
404     
405       newPointFOI.areacode = lastAreaCode;
406       var distance = newXOrdinate - lastXOrdinate;
407       while(Math.abs(distance) > width /2)
408       {
409         if(distance < 0)
410         {
411           newXOrdinate = newXOrdinate + width;
412           newPointFOI.areacode = lastAreaCode + 1;
413         }
414         else
415         {
416           newXOrdinate = newXOrdinate - width;
417           newPointFOI.areacode = lastAreaCode - 1;
418         } 
419         distance = newXOrdinate - lastXOrdinate;
420       }
421 
422       //newPointFOI.updateGeometry(MVSdoGeometry.createPoint(newXOrdinate, newPointFOI.getGeometry().getPointY(),(redline.parent.srid)));  
423     }

426    /**@private*/
427   this.refreshRedline = function(moveTag)
428   {
429     var calNearestDis = function(x1,x2)
430     {
431       x1 = MVUtil.keepInCSBoundary(redline.parent,x1);
432       x2 = MVUtil.keepInCSBoundary(redline.parent,x2);
433       var tmpdistance = x2 - x1;
434       if(tmpdistance > 0)
435         x1 = x1 + (redline.parent.msi.coverMaxX - redline.parent.msi.coverMinX);
436       else
437         x2 = x2 + (redline.parent.msi.coverMaxX - redline.parent.msi.coverMinX);
438         
439       if(Math.abs(x2 - x1) >  Math.abs(tmpdistance) )
440         return Math.abs(tmpdistance);
441       else
442         return Math.abs(x2 - x1);
443     }    
444     
445     if(redline.parent.wraparound)
446     {
447       var shapePoints = redline.pointFOIs;    
448       if(redline.getStatus() == 1)
449       {    
450         
451         if(shapePoints.length !=0)
452         {
453           var nearCenterIndex = 0;
454           var nearestDis = Number.MAX_VALUE;
455           var centerPos = MVUtil.keepInCSBoundary(redline.parent, redline.parent.getCenterLon() );
456           for(var i=0; i<shapePoints.length;i++)
457           {
458               var dis = Math.abs( MVUtil.keepInCSBoundary(redline.parent, shapePoints[i].getGeometry().getPointX() ) - centerPos );
459               if(nearestDis>dis)
460               {
461                 nearestDis = dis;
462                 nearCenterIndex = i;
463               }
464           }
465           
466           redline.keepTwoPointsInHalfEarth(nearCenterIndex,shapePoints[nearCenterIndex],null);
467           if(nearCenterIndex>0)
468           {
469             for(var i = nearCenterIndex-1; i >= 0; i--)
470             {
471               redline.keepTwoPointsInHalfEarth(i,shapePoints[i],shapePoints[i+1]);
472             }
473           }
474           if(nearCenterIndex<shapePoints.length)
475           {
476             for(var i=nearCenterIndex+1; i<shapePoints.length; i++)
477             {
478               redline.keepTwoPointsInHalfEarth(i,shapePoints[i],shapePoints[i-1]);
479             }
480           }
481         }
482 		
483         redline.refreshLines(moveTag);
484       }
485       else if(redline.getStatus() == 2)
486       {      
487         if(redline.parent.getFOI("-RL2RL"+redline.id))
488         {
489            var redlineObj = redline.parent.getFOI("-RL2RL"+redline.id);
490            //redlineObj.areacode = redline.parent.indFOIsControl.centerArea;
491            //return;
492            
493            var nearestCenter =  Number.MAX_VALUE;
494            var nearestNode = 0;
495            for(var i=0;i<shapePoints.length;i++)
496            {
497               var dis = calNearestDis( shapePoints[i].getGeometry().getPointX(),redline.parent.getCenterLon() );
498               if( nearestCenter>dis )
499               {
500                 nearestCenter = dis;
501                 nearestNode = i;
502               }
503            }
504            
505            redline.keepTwoPointsInHalfEarth(nearestNode,shapePoints[nearestNode],null);
506            
507            if(nearestNode>0)
508            {
509              for(var i = nearestNode-1; i >= 0; i--)
510              {
511                redline.keepTwoPointsInHalfEarth(i,shapePoints[i],shapePoints[i+1]);
512              }
513            }
514            if(nearestNode<shapePoints.length)
515            {
516              for(var i=nearestNode+1; i<shapePoints.length; i++)
517              {
518                redline.keepTwoPointsInHalfEarth(i,shapePoints[i],shapePoints[i-1]);
519              }
520            }
521            
522            var minxArray = new Array();
523            var minxAreaCode =  Number.MAX_VALUE;
524            for(var i=0;i<shapePoints.length;i++)
525            {
526               if(shapePoints[i].areacode < minxAreaCode)
527                 minxAreaCode = shapePoints[i].areacode;
528            }
529            for(var i=0;i<shapePoints.length;i++)
530            {
531               if(shapePoints[i].areacode == minxAreaCode)
532                 minxArray.push([i,shapePoints[i]]);
533            }
534            var minxNode = 0;
535            var mixOrd = Number.MAX_VALUE;
536            for(var i=0;i<minxArray.length;i++)
537            {
538               var tmpPoint = (minxArray[i])[1];
539               if( MVUtil.keepInCSBoundary(redline.parent, tmpPoint.getGeometry().getPointX() ) < mixOrd)
540               {
541                 mixOrd = MVUtil.keepInCSBoundary(redline.parent, tmpPoint.getGeometry().getPointX() );
542                 minxNode = (minxArray[i])[0];
543               }
544            }    
545            
546            redlineObj.areacode = shapePoints[minxNode].areacode;
547 
548         }       
549         
550       }
551     }
552   }
