 18 /**
 19   @private
 20 */
 21 function MVInfoWindowStyle3()
 22 {
 23 }
 24 /**
 25   @private
 26 */
 27 MVInfoWindowStyle3.init=function()
 28 {
 29   this.window=null;
 30   this.windowDiv=null;
 31   this.iw = MVGlobalVariables.imgFilePath + "/infoicons/";
 32   this.iw_arrow =this.iw + "arrow.gif";
 33   this.iw_h=this.iw+"hline.gif";
 34   this.iw_v=this.iw+"vline.gif";
 35   this.iw_c=this.iw+"iw_c.gif";
 36 
 37   this.closeImg=this.iw + "close.gif";
 38 
 39   this.iwcWidth = 600;
 40   this.iwcHeight = 600;
 41   this.iwarrowWidth = 30;
 42   this.iwarrowHeight = 16;
 43   this.iwhlWidth = 600;
 44   this.iwhlHeight = 1;
 45   this.iwvlWidth = 1;
 46   this.iwvlHeight = 600;
 47   this.iwwuWidth = 1;
 48   this.iwwuHeight = 10;
 49 }
 50 /**
 51   @private
 52 */
 53 MVInfoWindowStyle3.createWindow=function(x,y,width,height)
 54 {
 55   this.window=new Object();
 56   this.window.arrow=FigLayer(this.iw_arrow,this.iwarrowWidth,this.iwarrowHeight,1,this.iwwuHeight,0);
 57   this.window.n=FigLayer(this.iw_h,width+1,1,this.iwarrowWidth+1,0,0);
 58   this.window.s=FigLayer(this.iw_h,width+1,1,this.iwarrowWidth+1,height+1,0);
 59   this.window.e=FigLayer(this.iw_v,1,height+1,width + this.iwarrowWidth + 1,1,0);
 60   this.window.wu=FigLayer(this.iw_v,1,this.iwwuHeight,this.iwarrowWidth,0,0);
 61   this.window.wd=FigLayer(this.iw_v,1,height-this.iwarrowHeight-this.iwwuHeight+2,this.iwarrowWidth+1,this.iwarrowHeight+this.iwwuHeight,0);
 62   this.window.c=FigLayer(this.iw_c,width,height,this.iwarrowWidth+1,1,0);
 63   this.windowDiv=window.document.createElement("div");
 64   this.windowDiv=window.document.createElement("div");
 65   this.windowDiv.style.position="relative";
 66   this.windowDiv.style.left=MVUtil.pixel(x);
 67   this.windowDiv.style.top=MVUtil.pixel(y);
 68   this.windowDiv.appendChild(this.window.n);
 69   this.windowDiv.appendChild(this.window.s);
 70   this.windowDiv.appendChild(this.window.e);
 71   this.windowDiv.appendChild(this.window.wu);
 72   this.windowDiv.appendChild(this.window.c);
 73   this.windowDiv.appendChild(this.window.wd);
 74   this.windowDiv.appendChild(this.window.arrow);
 75   this.width=width;
 76   return this.windowDiv;
 77 }
 78 /**
 79   @private
 80 */
 81 MVInfoWindowStyle3.addCloseButton=function(closefunction)
 82 {
 83   this.closeButton=FigLayer(this.closeImg,15,15,this.width-1,10,4);
 84   this.closeButton.style.position="absolute";
 85   MVUtil.setCursor(this.closeButton,"pointer");
 86   this.windowDiv.appendChild(this.closeButton);
 87 
 88   this.closeButton.onclick = MVUtil.closureWrapper(this.closeButton, 
 89      function(){closefunction();});
 90 }
 91 /**
 92   @private
 93 */
 94 MVInfoWindowStyle3.addContent=function(cdiv)
 95 {
 96   cdiv.style.left=MVUtil.pixel(50);
 97   cdiv.style.top=MVUtil.pixel(25);
 98   this.windowDiv.appendChild(cdiv);
 99 }
