 30 /**
 31   @class MVNSDP provides info about a Non Spatial Data Provider for a theme based FOI layer. 
 32   Such information is used to invoke a pre-registered Non-Spatial Data Provider
 33   by the FOI server, which will join the result attribute data with a theme's
 34   spatial data to produce a thematic rendering of the theme.
 35   @constructor
 36   @param nsdpId the ID of the Non Spatial Data Provider that has been registered with MapViewer
 37  */
 38 function MVNSDP(nsdpId)
 39 {
 40   /**@private*/
 41   this.nsdp_id        = nsdpId;   //id of the registered provider
 42   /**@private*/
 43   this.nsdp_theme     = null;     //for which theme
 44   /**@private*/
 45   this.nsdp_keycol    = null;     //key column in spatial table
 46   /**@private*/
 47   this.nsdp_rendersty = null;     //overriding style
 48   /**@private*/
 49   this.nsdp_labelsty = null;     //overriding label style
 50   /**@private*/
 51   this.nsdp_params    = null;     //per-request parameters
 52   /**@private*/
 53   this.nsdp_timeout   = 20;       //timeout enforced by mapviewer
 54   /**@private*/
 55   this.nsdp_smfo      = true;     //show matched feature only
 56   
 57 }
 58 
 59 /**
 60   Sets the name of the theme for this NS Data Provider. It should
 61   be the real name of the theme as known to MapViewer server, not
 62   the layerName (the first parameter) specified when creating a 
 63   MVThemeBasedFOI.  In other words, if the theme is pre-defined 
 64   (definition is stored in the database) then you should use the
 65   theme's name as stored in the database. For a dynamic/JDBC theme, 
 66   the name should be the one specified in the XML definition's <theme> element.
 67   @param {String} themeName  the name of the MapViewer theme
 68  */
 69 MVNSDP.prototype.setTheme = function(themeName)
 70 {
 71   this.nsdp_theme = themeName;
 72 }

106 /**
107   Sets whether MapViewer should display only those theme features
108   that have a match with the NS data set. Default is true. If
109   set to false, then every feature produced by the theme for the
110   current map request will be displayed. Note that in this case
111   theme features may be displayed in different styles 
112   depending on whether they have the attribue data from the matching
113   NS data rows.
114   @param {Boolean} flag  a boolean value (true | false).
115 */
116 MVNSDP.prototype.setShowMatchedFeatureOnly = function(flag)
117 {
118   this.nsdp_smfo = flag;
119 }

121 /**
122   Sets the properties to be consumed by the NS Data Provider implementation.
123   These properties are key-value pairs that are understood only by the
124   NS Data Provider. <p>
125   
126   You must collect all the properties into one associative array (Object), then
127   pass this object as the argument when calling this method. Both the key and 
128   value of each property must be of String type.
129   <p>
130   Sample usage: <br>
131   <pre>
132     var  params = new Object();
133     params['url'] = 'http://www.my_corp.com:3456/bi';
134     params['layers'] = 'sales_2005';
135     params['regions'] = 'North_America';
136     
137     nsdp.setParameters(params);
138   </pre>
139   
140   @param {Array} associativeArrayObj an associative array containing all the properties
141  */
142 MVNSDP.prototype.setParameters = function(associativeArrayObj)
143 {
144   this.nsdp_params = associativeArrayObj;
145 }

159 /**
160   Gets the JSON object string to be sent to the FOI server.
161   @private
162  */
163 MVNSDP.prototype.getFlatString = function() 
164 {
165   var s = "{";
166   s += "\"nsdp_id\":"+"\""+this.nsdp_id+"\",";
167   s += "\"nsdp_theme\":"+"\""+this.nsdp_theme+"\",";
168   s += "\"nsdp_keycol\":"+"\""+this.nsdp_keycol+"\",";
169   if(this.nsdp_smfo!=null && this.nsdp_smfo==false)
170     s += "\"nsdp_smfo\":"+"\""+this.nsdp_smfo+"\",";
171   if(this.nsdp_rendersty != null)
172     s += "\"nsdp_rendersty\":"+"\""+this.nsdp_rendersty+"\",";
173   if(this.nsdp_labelsty != null)
174     s += "\"nsdp_labelsty\":"+"\""+this.nsdp_labelsty+"\",";
175   s += "\"nsdp_timeout\":"+"\""+this.nsdp_timeout+"\"";
176   
177   if(this.nsdp_params==null)
178   {
179     return s += "}";
180   }
181   
182   s += ",\"nsdp_params\":[";
183   var cnt = 0;
184   for (p in this.nsdp_params)
185   {
186     var contents = this.nsdp_params[p];
187     //escape any double quotes in the contents
188     var ec = MVUtil.strReplaceAll(contents, "\"", "\\\"") ; 
189 
190     s += "{"+"\""+p + "\":"+"\""+ec+"\"},";
191     cnt ++;
192   }
193   
194   //removing extra ','
195   if(cnt>0)
196     s = s.substr(0, s.length-1);
197     
198   s += "]}";
199   return s;
200 }

202 /**
203   Gets the XML string to be sent to mapviewer as part of the map request.
204   @private
205  */
206 MVNSDP.prototype.getXML = function(themebasedfoi) 
207 {
208    /*     <ns_data_provider provider_id="mydp1" time_out="20000">
209    *        <parameters>
210    *          <parameter name="report_type" value="sales_by_quarter" />
211    *          <parameter name="report_year" value="2005" />
212    *          <parameter name="foo" value="bar" />
213    *          ...
214    *        </parameters>
215    *        <for_theme name="theme_us_states" />  <!-- must reference to an existing theme -->
216    *        <custom_rendering_style name="myChartStyle1" />
217    *        
218    *        <join spatial_key_column="state_abrv" />
219    *        <display_unmatched_features>true</display_unmatched_features>
220    *      </ns_data_provider>
221    */
222   var s = "<ns_data_provider provider_id=\""+this.nsdp_id+"\" ";
223   s += "time_out=\""+this.nsdp_timeout+"\">" ;
224   s += "<for_theme name=\""+ this.nsdp_theme+"\"/>" ;
225   var style = null ;
226   if(this.nsdp_rendersty != null)
227     style = this.nsdp_rendersty ;
228   else if(themebasedfoi && themebasedfoi.renderStyle!=null)
229     style = themebasedfoi.renderStyle ;
230   if(style)
231     s += "<custom_rendering_style name=\""+style+"\"/>" ;
232   s += "<join spatial_key_column=\""+this.nsdp_keycol+"\"/>" ;
233   
234   if(this.nsdp_smfo!=null && this.nsdp_smfo==false)
235     s+= "<display_unmatched_features>true</display_unmatched_features>" ;
236     
237   if(this.nsdp_params!=null)
238   {
239     s += "<parameters>" ;
240     for (p in this.nsdp_params)
241     {
242       var val = null ;
243       if(p=="xml")
244         val = this.nsdp_params[p].replace(/</g, "<") ;
245       else
246         val = this.nsdp_params[p] ;
247       s += "<parameter name=\""+ p + "\" value=\"" + val+"\"/>" ;
248     }
249     s += "</parameters>" ;
250   }
251   s += "</ns_data_provider>";
252   return s;
253 }
