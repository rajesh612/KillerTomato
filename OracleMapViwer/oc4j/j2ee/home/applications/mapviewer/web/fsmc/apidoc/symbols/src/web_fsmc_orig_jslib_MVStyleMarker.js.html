 22 /**
 23  * @class MVStyleMarker lets you create a MapViewer MARKER style instance at the
 24    client side.  
 25    You can specify the essential information when creating a new Marker style, 
 26    such as the style name and marker type (image or vector based). You can set 
 27    other aspects through various setter methods,  such as the vector shape and fill 
 28    and stroke colors as well as their opacity for a vector type Marker, or the 
 29    image URL for an image based Marker style.
 30    <br>
 31    Note that for vector type, only basic shapes such as circles, rectangles and simple 
 32    polygons/polylines are supported. These vector shapes must be able to be drawn in
 33    one continuous pen stroke. In other words, you cannot have a polygon Marker with a
 34    hole inside it (to draw the hole you must lift your pen and start on a different
 35    stroke).
 36 
 37    <br> Typical useage of this class is illustrated below.
 38    <pre>
 39       var style = new MVStyleMarker("my_vector_icon", "vector");
 40       style.setSize(19, 19);
 41       style.setStrokeColor("ff0000");
 42       style.setStrokeWidth("1.5");
 43       style.setFillColor("00ff00");
 44 
 45       //this sets the shape to be a circle with a dummy radius value (50).
 46       //the actual size of the circle when displayed is specified in the
 47       //style.setSize() call above.
 48       style.setVectorShape("c:50");
 49 
 50       //this is how you set up a simple polygon instead of circle:
 51       //style.setVectorShape("50,199,0,100,50,1,100,100,50,199");
 52       
 53       themebasedfoi.addStyle(style);
 54       themebasedfoi.setRenderingStyle("my_vector_icon");
 55 
 56    </pre>
 57    
 58    <br>
 59    Note that once a style object is added to the theme based FOI instance
 60    and set as its rendering style, the MapViewer server side will typically
 61    cache and reuse a renderered version of the marker style for all subsequent
 62    theme-based FOI requests that uses a style with the same name and size.  
 63    As such, when you modify a style client-side, you should 
 64    also modify its name (and/or size) in order to display the modified version
 65    on the map. If you do modify the style's name, don't forget to change
 66    it in the themebasedfoi.setRenderingStyle() call too.
 67    
 68    @constructor   
 69    @param {string} name  name of the MARKER style
 70    @param {string} type  specifies the type of the MARKER: must be "image" or 
 71         "vector". 
 72  */
 73 function MVStyleMarker(name, type)
 74 {
 75   /**@private*/
 76   this.name = name;
 77   /**@private*/
 78   if(type==null)
 79    this.type = "image";
 80   else
 81   /**@private*/
 82     this.type = type.toLowerCase();
 83     
 84   /**@private*/
 85   this.description = null;
 86 
 87   /**@private*/
 88   this.fill=null;    
 89   /**@private*/
 90   this.fill_opacity = 255;
 91 
 92   /**@private*/
 93   this.stroke = null;    
 94   /**@private*/
 95   this.stroke_opacity = 255;
 96   /**@private*/
 97   this.stroke_width = 1;
 98   
 99   /**@private*/
100   this.imageUrl = null;
101   
102   /**@private*/
103   this.vector = null;
104   
105   /**@private*/
106   this.width = 16;
107   
108   /**@private*/
109   this.height= 16;
110 }

122 /**
123   Sets the preferred width and height (in pixel) of this marker style as displayed on
124   screen. MapViewer server will automatically scale the image or vector marker
125   acoordingly based on these values. Default values (when not specified) are
126   16 by 16 pixels.

130   if(width!=null)
131     this.width = width;
132   
133   if(height!=null)
134     this.height = height;

137 /**
138   Sets the URL of the image to be used for this marker style.
139   @param {string} imgUrl the url to an image file.
140  */
141 MVStyleMarker.prototype.setImageUrl = function(imgUrl)
142 {
143   if(imgUrl==null)
144     this.imageUrl = null;
145   else
146   {
147     if(imgUrl.indexOf("://")<0)
148     {
149       if(imgUrl.indexOf("/")==0) 
150         imgUrl = document.location.protocol + "//"+document.location.host + imgUrl ;
151       else
152       {
153         var url = document.location+"" ;
154         var idx = url.lastIndexOf("/") ;
155         if(idx>0)
156           url = url.substring(0, idx+1) ;
157         imgUrl = url + imgUrl ;
158       }
159     }
160     this.imageUrl = imgUrl;
161   }
162 }

164 /**
165   Sets the stroke color and (optionally) its opacity. Only used by vector type 
166   markers. Opacity value ranges from 0 (completely transparent) to 255 
167   (completely opaque). Default value is 255.
168   @param {String} stroke the stroke color (for drawing the boundaries of vector
169       shape) specified in the format of "RRGGBB";
170   @param {int} op  the opacity value for the stroke color ([0 - 255]).
171  */
172 MVStyleMarker.prototype.setStrokeColor = function(stroke, opacity)
173 {
174   if(stroke==null)
175     this.stroke = null;
176   else if(stroke.charAt(0)=='#')
177     this.stroke = stroke.substring(1, stroke.length);
178   else
179     this.stroke = stroke;
180     
181   this.stroke_opacity = opacity;
182 }
183 
184 /**
185   Sets the fill color and (optionally) its opacity. Fill color is used to fill 
186   the interior of the shape of a vector type marker style. 
187   @param {String} fill the fill color specified in the format of "RRGGBB".
188   @param {int} op  the opacity value for the fill color ([0 - 255]).
189  */
190 MVStyleMarker.prototype.setFillColor = function(fill, opacity)
191 {
192   if(fill==null)
193     this.fill=null;
194   else if(fill.charAt(0)=='#')
195     this.fill = fill.substring(1, fill.length);
196   else
197     this.fill = fill;
198   
199   if(opacity!=null)
200     this.fill_opacity = opacity;
201 }

222 /**
223   Sets the shape coordinates for the vector type marker.  It basically supports
224   two types of shapes: polygon/polyline and circle. 
225   
226   For polygon/polyline (which can be used to represent rectangle, triangle and 
227   other simple shapes), the parameter must be a series of comma separated 
228   ordinates (x1,y1,x2,y2,...). 
229 
230   Note that for polygon/polyline shapes, the cooridnates can be in any coordinate
231   system with arbitrary origin. MapViewer will do all the necessary 
232   transformation at the server side. For instance, the following cooridnates 
233   represent a diamond shape:
234   
235     "50,199,0,100,50,1,100,100,50,199"
236   
237   For circle type marker, the parameter must start with "c:" and followed by a
238   single numeric value representing the radius of the cirlce, such as "c:50". 
239   
240   @param {String} vector a string representing the vector shape
241 */
242 MVStyleMarker.prototype.setVectorShape = function(vector)
243 {
244   if(vector==null)
245     this.vector = null;
246   else
247     this.vector = vector;
248 }

250 /**
251   @private
252   Gets the MapViewer XML representation of this COLOR style.
253  */
254 MVStyleMarker.prototype.getXMLString = function()
255 {
256 /*
257 image:
258 
259 <svg width="1in" height="1in">
260     <desc/>
261     <g class="marker" style="width:12;height:12;font-family:Dialog;font-size:12;font-fill:#FFFFFF">
262         <image x="0" y="0" width="1" height="1" markerType="gif" href="dummy.gif"/>
263   </g>
264 </svg>
265 
266 vector:
267 
268 <svg width="1in" height="1in">
269     <desc/>
270     <g class="marker" style="stroke:#FF0066;fill:#FF9900;width:10;height:10;font-family:Dialog;font-size:12;font-fill:#FF0000">
271         <polyline points="50.0,199.0,0.0,100.0,50.0,1.0,100.0,100.0,50.0,199.0"/>
272   </g>
273 </svg>
274 
275 */
276 
277   var size = "width:"+this.width+";height:"+this.height+";";
278   var s = (this.stroke==null)? "" : "stroke:0x"+this.stroke+";";
279   var f = (this.fill==null)?"" : "fill:0x"+this.fill+";";
280   var sop = this.stroke_opacity<255? "stroke-opacity:"+this.stroke_opacity+";" : "";
281   var fop = this.fill_opacity<255? "fill-opacity:"+this.fill_opacity+";" : "";
282   var sw = this.stroke_width!=1? "stroke-width:"+this.stroke_width : "";
283   
284   var marker = "";
285   if(this.type=="image")
286   {
287     marker = '  <image_marker url="'+this.imageUrl+'" />';
288   }
289   else if(this.type=="vector")
290   {
291     if(this.vector.charAt(0)=='c' || this.vector.charAt(0)=='C')
292     {
293       //circle type
294       marker = '  <circle cx="0" cy="0" r="1"/>';
295     }
296     else
297     {
298       marker = '  <polyline points="'+this.vector+'"/>';
299     }
300   }
301   
302   var xml = '<style name="' + this.name + '">' +
303             '<svg width="1in" height="1in">' +
304             '<g class="marker" style="'+size+s+sop+f+fop+sw+'">'+
305             marker +
306             '</g></svg></style>';
307   return xml;
308   
309 }
