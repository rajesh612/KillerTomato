 25 /**
 26  @class MVBarInfo this class represents the definition of a single bar in
 27         a Bar chart.
 28  @constructor
 29  @param {string} name the name of the bar; used only for labeling in a map
 30         legend.
 31  @param {string} color the RRGGBB value of the color of the bar, such
 32                    as "FFFFFF" for white.
 33   */
 34 function MVBarInfo(name, color)
 35 {
 36   /**@private*/
 37   this.name = name;
 38   /**@private*/
 39   this.color = null;
 40   
 41   if(color)
 42   {
 43     if(color.charAt(0)=='#')
 44       this.color = color.substring(1, color.length);
 45     else
 46       this.color = color;
 47   }
 48   /**@private*/
 49   this.color_opacity = 255;
 50 }

 52 /**
 53  @class MVBarChartStyle represents the client side Bar Chart style.  
 54  @constructor 
 55  @param {string} name the name of this style
 56  @param {integer} width the overall width of the bar chart
 57  @param {integer} height the overall height of the bar chart
 58  @param {Array} bars the array containing all the bar definitions
 59  */
 60 function MVBarChartStyle(name, width, height, bars)
 61 {
 62   /**@private*/
 63   this.name = name;
 64   /**@private*/
 65   this.width = width;
 66   /**@private*/
 67   this.height = height;
 68   /**@private*/
 69   this.bars = bars;
 70   /**@private*/
 71   this.minValue = null;
 72   /**@private*/
 73   this.maxValue = null;  
 74   /**@private*/
 75   this.shareScale = true;  //all bar charts in the same map will share same scale
 76   /**@private*/
 77   this.showXAxis = false;
 78 }
 79 /**
 80  * Sets the bar minimum value.
 81  * @param min the bar minimum value
 82  * @return
 83  */
 84 MVBarChartStyle.prototype.setMinValue = function(min)
 85 {
 86   this.minValue = min;
 87 }
 88 
 89 /**
 90  * Sets the bar maximum value.
 91  * @param min the bar maximum value
 92  * @return
 93  */
 94 MVBarChartStyle.prototype.setMaxValue = function(max)
 95 {
 96   this.maxValue = max;
 97 }
 98 /**
 99  * Sets if the size of bars is to be based in an individual feature, or if it is to based on values 
100  * of all map features of the theme.
101  * @param ss true means size are related with value of all features, false means the each bar size is independent.
102  * @return
103  */
104 MVBarChartStyle.prototype.setShareScale = function(ss)
105 {
106   this.shareScale = ss;
107 }
108 /**
109  * Sets if X axis should be rendered.
110  * @param showx true to render X axis, false to not render
111  * @return
112  */
113 MVBarChartStyle.prototype.setShowXAxis = function(showx)
114 {
115   this.showXAxis = showx;
116 }
117 /**@private*/
118 MVBarChartStyle.prototype.getXMLString = function()
119 {
120   var xmlDef = 
121   '<style name="' + this.name + '">' +
122   '<AdvancedStyle><BarChartStyle' +
123   ' width="' + this.width + '" height="' + this.height + '"';
124   
125   if(this.shareScale==true)
126   {
127     xmlDef = xmlDef + ' share_scale="true"';   
128   }
129   
130   if(this.maxValue!=null && this.minValue!=null)
131   {
132     xmlDef = xmlDef + ' min_value="'+this.minValue+'" max_value="'+this.maxValue+'"';
133   }
134   
135   if(this.showXAxis)
136   {
137     xmlDef = xmlDef + ' show_x_axis="true"';
138   }
139   
140   xmlDef =xmlDef + '>';
141   
142   if(this.bars != null)
143   {
144     /*       <PieSlice name="jan" color="#ff00ff" /> */
145     for(var i=0;i<this.bars.length;i++)
146     {
147       var bi = this.bars[i];
148       xmlDef = xmlDef + ' <Bar name="' + bi.name + '"';
149       if(bi.color != null)
150         xmlDef = xmlDef + ' color="#' + bi.color + '"';
151       xmlDef = xmlDef + ' />';
152     }
153   }
154 
155   xmlDef += ' </BarChartStyle></AdvancedStyle></style>';    
156   return xmlDef;  
157 }
