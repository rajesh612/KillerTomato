 17 /**
 18  * @class MVInfoWindowTab creates an information window tab.
 19     An array of instances of this class can be passed as the tabs argument 
 20     to MVMapview.displayTabbedInfoWindow
 21 
 22    @constructor   
 23    @param {string} title  the title of tab.
 24    @param {string} content  an HTML content string to be displayed inside the tab.
 25  */
 26  function MVInfoWindowTab(title, content)
 27  {
 28    this.title=title;
 29    this.content=content;
 30  }
 31  /**
 32   @private
 33  */
 34  MVInfoWindowTab.prototype.getTitle = function()
 35  {
 36    return this.title;
 37  }
 38  /**
 39   @private
 40  */ 
 41  MVInfoWindowTab.prototype.getContent = function()
 42  {
 43    return this.content;
 44  }
