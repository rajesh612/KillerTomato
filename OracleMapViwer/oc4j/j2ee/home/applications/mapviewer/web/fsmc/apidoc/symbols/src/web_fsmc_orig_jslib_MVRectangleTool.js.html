 42 /**
 43  *@class MVRectangleTool defines a rectangle tool that can be used to
 44  * draw a "rubber-band" style rectangle on the map, by holding down
 45  * and dragging the mouse. <p>
 46  * As soon as the mouse button is released, a rectangle FOI will
 47  * be displayed on the map. To remove this FOI you must call the tool's
 48  * clear() method.
 49  *
 50  *@constructor
 51  *@param {String} rectStyle a string parameter that specifies the MapViewer area style to
 52  *                          be used to render the rectangle area FOI after the
 53  *                          user finishes rubber banding and releases the mouse button.
 54  *                          The style is specified in the form "data-source-name"."style-name",
 55  *                          where "style-name" is the name of the
 56  *                          MapViewer area style and "data-source-name" is the
 57  *                          name of data source in which the style is defined.
 58  *                          If this parameter is omitted or set to null, no rectangle
 59  *                          FOI will be generated or displayed after the user
 60  *                          releases the mouse button.
 61  *@param {String} foiServerUrl an optional String parameter that specifies the
 62  *                          URL of the FOI server, which should be in the form
 63                             http://host:port/mapviewer/foi. If this parameter is omitted,
 64  *                          the URL of the FOI server is derived
 65  *                          from MapViewer base URL.
 66  *@return The newly created MVRectangleTool object.
 67 */function MVRectangleTool(rectStyle, foiServerUrl)
 68 {
 69   /**@private*/
 70   this.parent = null;

 73   /**@private*/
 74   this.innerRectangleDiv = null ;
 75   this.rectangleDiv.style.position = "absolute";
 76   this.rectangleDiv.style.zIndex = 5000;
 77   this.rectangleDiv.style.border = "#FF0000 2px dashed";
 78   this.rectangleDiv.style.visibility = "hidden";
 79   /**@private */
 80   this.divWidth = 2;
 81   
 82   /**@private*/
 83   this.foiServerUrl = foiServerUrl;
 84   
 85   /**@private*/
 86   this.areaStyle = rectStyle;
 87   /**@private*/
 88   this.fromLon = 0;
 89   /**@private*/
 90   this.fromLat = 0;
 91   /**@private*/
 92   this.toLon = 0;
 93   /**@private*/
 94   this.toLat = 0;
 95   /**@private*/
 96   this.scrollMapFromLon = 0;
 97   /**@private*/
 98   this.scrollMapFromLat = 0;
 99    /**@private*/
100   this.EventListeners = new Object();
101   /**@private*/
102   this.onStartListener=null;
103   /**@private*/
104   this.onFinishListener=null;
105   /**@private*/
106   this.onDragListener=null;
107   /**@private*/
108   this.mouseupHandler = null ;
109   
110   /**@private*/
111   this.sdoGeometry=null;
112   
113   var rectangleToolId=eval(MVGlobalVariables.rectangleToolId);
114   ++rectangleToolId;

117   /**@private*/

119   /**@private*/

121   /**@private*/

123   /*Only for Netscape*/
124   var rt = this ;
125   /**@private*/
126   this.keyPressFunc = function(evt)
127   {
128     evt = (evt) ? evt : ((window.event) ? event : null);
129     if(rt.parent.enableMapEventListner)
130       MVUtil.stopEvent(evt);
131   };
132   /*Only for Netscape*/
133   /**@private*/
134   this.disableKeyPress = function()

138       if(MVGlobalVariables.checkBrowser("netscape"))
139         this.netscape = true;
140       else
141         this.netscape = false;
142     }

144     {
145       window.addEventListener("keypress", this.keyPressFunc, false);
146     }
147   }
148   /*Only for Netscape*/
149   /**@private*/
150   this.enableKeyPress = function()
151   {
152     if(this.netscape == null)
153     {
154       if(MVGlobalVariables.checkBrowser("netscape"))

168     {
169       if(rt.getStatus() == 3)
170       {      
171 	    var rectObj = rt.getRectangleFOI();
172         if(rectObj)
173         {
174 		  rectObj.areacode = rt.parent.indFOIsControl.centerArea;
175         }
176       }
177     }
178   }

216 /**
217  *This method customizes the filling color of the rectangle 
218  *DIV displayed while the user drags the mouse with the left mouse button being pressed. 
219  *@param {String} color a String that specifies the filling color, e.g. "#FF0000".
220  *@param {Number} opacity a valid number ranging from 0 to 1, e.g. 0.2
221  *@return none
222  *@type void
223 */ 
224 MVRectangleTool.prototype.setDivFillColor = function(color, opacity)
225 {
226   if(!this.innerRectangleDiv)
227   {
228     this.innerRectangleDiv = document.createElement("div") ;
229     this.innerRectangleDiv.style.width="100%";
230     this.innerRectangleDiv.style.height="100%";
231     this.rectangleDiv.appendChild(this.innerRectangleDiv) ;
232   }
233   
234   if(color && color !="")
235   {
236     if(this.innerRectangleDiv.style.setAttribute)
237       this.innerRectangleDiv.style.cssText += ";background-color:" + color;
238     else
239       this.innerRectangleDiv.style.backgroundColor = color;
240   }  
241   
242   if(isNaN(opacity) || opacity<0 || opacity>1)
243   {
244     MVi18n.reportError("MVRectangleTool.setDivFillColor", "MAPVIEWER-05519", "opacity",
245                        this.parent?this.parent.errorHandler:null);
246     return ;
247   }
248   else
249   {
250     if(window.ActiveXObject)
251     {
252       this.innerRectangleDiv.style.filter = "alpha(opacity=" + (opacity*100) + ")"; 
253     }
254     else
255     {
256       this.innerRectangleDiv.style.opacity = opacity; 
257     }
258   }
259 }

296 	if(tmpwidth > wholewidth)
297   {
298     if( this.fromLon < this.toLon )

301       this.toLon = this.fromLon - wholewidth ;

316 /**
317  *@private
318 */
319 MVRectangleTool.prototype.drawArea = function()
320 {
321   if(this.dragStatus!=3)
322   {
323     return;
324   }
325 
326   var fromX = (this.fromLon-this.parent.getCenterLon())*this.parent.pixPerX + 
327           (this.parent.getPaneWidth()/2);
328   var fromY = (this.fromLat-this.parent.getCenterLat())*(-this.parent.pixPerY) + 
329           (this.parent.getPaneHeight()/2);
330           
331   var toX = (this.toLon-this.parent.getCenterLon())*this.parent.pixPerX + 
332           (this.parent.getPaneWidth()/2);
333   var toY = (this.toLat-this.parent.getCenterLat())*(-this.parent.pixPerY) + 
334           (this.parent.getPaneHeight()/2);
335       
336   var width=Math.abs(fromX-toX);
337   var height=Math.abs(fromY-toY);
338   width=Math.round(width);
339   height=Math.round(height);
340   
341   var leftPos=(fromX<toX?fromX:toX);
342   var topPos=(fromY<toY?fromY:toY);
343   leftPos=Math.round(leftPos);
344   topPos=Math.round(topPos);
345   
346   this.rectangleDiv.style.left = leftPos-1-MVUtil.getLeft(this.parent.div) + 'px';
347   this.rectangleDiv.style.top = topPos-1-MVUtil.getTop(this.parent.div) + 'px';
348   
349   if(width>0)
350     width ++ ;
351   if(height>0)
352     height ++;
353   if(this.browser == "NS")
354   {
355     if(width==0)
356       width=(2*this.divWidth);
357     
358     if(height==0)
359       height=(2*this.divWidth);
360       
361     this.rectangleDiv.style.width = Math.abs(width-(2*this.divWidth)) + 'px';
362     this.rectangleDiv.style.height = Math.abs(height-(2*this.divWidth)) + 'px';
363     this.rectangleDiv.style.clip = "rect(0px, "+(width)+"px, "+
364                          (height)+"px, 0px)";
365   }
366   else
367   {
368     this.rectangleDiv.style.width = width + 'px';
369     this.rectangleDiv.style.height = height + 'px';
370     this.rectangleDiv.style.clip = "rect(0px, "+(width+(2*this.divWidth))+"px, "+
371                          (height+(2*this.divWidth))+"px, 0px)";
372   }
373   
374   this.rectangleDiv.style.visibility="visible" ;
375 }

441 /**
442  *This method activates and initializes the rectangle tool. 
443  *The application must call this method to activate the 
444  *rectangle tool before the user can draw rectangle on the map.
445  *@return none
446  *@type void
447 */  
448 MVRectangleTool.prototype.init = function()
449 { 
450   //this.clear() ;
451   this.hideRectangleDIV() ;
452   this.parent.removeFOI("MVRectangleTool"+this.id);
453   this.dragStatus=0;
454   this.parent.addEventHandlers();
455   
456   this.dragStatus=1;
457   MVUtil.clearListeners(this.rectangleDiv);
458   var tmpMapView=this.parent;
459   var tmpDiv=tmpMapView.base_div;
460   var tmpRectangleTool=this;
461   this.fromLon=this.fromLat=this.toLon=this.toLat=0;
462   var interVal;
463   
464   MVUtil.runListeners(tmpRectangleTool.EventListeners, MVEvent.INIT, [MVToolBar.BUILTIN_RECTANGLE]);  
465   
466   this.browser = this.parent.getBrowserType();
467   
468   // For the js obfuscator to properly replace local function names,

471   {
472     tmpMapView.enableMapEventListner=true;
473     
474     var centerLon=tmpMapView.getCenterLon();
475     var centerLat=tmpMapView.getCenterLat();
476      
477     if((tmpRectangleTool.scrollMapFromLon)!=centerLon || (tmpRectangleTool.scrollMapFromLat)!=centerLat)
478     {
479        if(tmpRectangleTool.dragStatus==2)
480        {   
481          tmpRectangleTool.dragStatus=3;          
482        }
483     
484        tmpRectangleTool.toLon+=(centerLon-tmpRectangleTool.scrollMapFromLon);
485        tmpRectangleTool.toLat+=(centerLat-tmpRectangleTool.scrollMapFromLat);
486        
487        tmpRectangleTool.scrollMapFromLon=centerLon;
488        tmpRectangleTool.scrollMapFromLat=centerLat;
489 
490        tmpRectangleTool.drawArea();
491     }
492   }
493     
494   MVUtil.setListener(tmpDiv, "mousedown", MVUtil.closureWrapper(tmpDiv, function(evt)

496     tmpMapView.removeFOI("MVRectangleTool"+tmpRectangleTool.id);
497     
498     tmpRectangleTool.dragStatus=2;
499     
500     evt = (evt) ? evt : ((window.event) ? event : null);
501     var Coords = tmpMapView.getEventCoords(evt);
502     tmpMapView.mouseDownLoc = Coords ;   
503     tmpMapView.getMPos(evt);
504     
505     tmpRectangleTool.startDrag();
506     tmpRectangleTool.endDrag();
507     tmpRectangleTool.rectangleDiv.style.visibility = "hidden";
508     if(tmpRectangleTool.onStartListener!=null)
509       (tmpRectangleTool.onStartListener)();
510     MVUtil.runListeners(tmpRectangleTool.EventListeners,MVEvent.START);

512     tmpRectangleTool.scrollMapFromLon = tmpMapView.getCenterLon();
513     tmpRectangleTool.scrollMapFromLat = tmpMapView.getCenterLat();  
514         
515     var waitTime=5;

517     
518     tmpRectangleTool.disableKeyPress();
519     return false;
520   }));
521   
522   MVUtil.setListener(tmpDiv, "mousemove", MVUtil.closureWrapper(tmpDiv, function(evt)
523   {    
524     if(tmpRectangleTool.dragStatus==2)
525       tmpRectangleTool.dragStatus=3;          
526     
527     if(tmpRectangleTool.dragStatus==3)
528     {      
529        evt = (evt) ? evt : ((window.event) ? event : null);
530        tmpMapView.getMPos(evt); 
531        tmpRectangleTool.endDrag(); 
532        tmpRectangleTool.drawArea();
533        if(tmpRectangleTool.onDragListener)

544     //add by zhengwan  
545     //avoid following detachEvent and removeEventListener bug in some condition

547         return false;
548         
549 		
550     evt = (evt) ? evt : ((window.event) ? event : null);
551     var Coords = tmpMapView.getEventCoords(evt);
552     tmpMapView.mouseUpLoc = Coords ;   
553     tmpMapView.getMPos(evt);
554     
555     tmpRectangleTool.endDrag();
556     tmpRectangleTool.generateArea();
557     
558     clearInterval(interVal);
559 
560     MVUtil.detachEvent(tmpDiv, "mouseup", mouseupHandler, false) ;
561     //this detachEvent removeEventListener have bug in some condition 
562     //can not remove the mouseupHandler
563     //but this bug is avoid by the first three line of this function
564     /*if(MVGlobalVariables.browser == "IF" )
565     {
566       tmpDiv.detachEvent("onmouseup", mouseupHandler);
567     } 
568     else
569     {
570       tmpDiv.removeEventListener("mouseup", mouseupHandler, true) ;
571     }*/
572 
573     tmpMapView.addEventHandlers();
574     
575     tmpRectangleTool.enableKeyPress();
576     
577     if(tmpRectangleTool.onFinishListener) {
578       (tmpRectangleTool.onFinishListener)();
579     }
580     MVUtil.runListeners(tmpRectangleTool.EventListeners,MVEvent.FINISH);
581     
582     
583     return false;
584   });
585 
586   if(this.mouseupHandler)
587     MVUtil.detachEvent(tmpDiv, "mouseup", this.mouseupHandler, false);
588   MVUtil.attachEvent(tmpDiv, "mouseup", mouseupHandler, false);
589 
590   //this detachEvent removeEventListener have bug in some condition 
591   /*if(MVGlobalVariables.browser == "IF" )
592   {
593     if(this.mouseupHandler)
594       tmpDiv.detachEvent("onmouseup", this.mouseupHandler);
595     tmpDiv.attachEvent("onmouseup", mouseupHandler);
596   } 
597   else
598   {
599     if(this.mouseupHandler)
600       tmpDiv.removeEventListener("mouseup", this.mouseupHandler, true) ;

603   this.mouseupHandler = mouseupHandler ;
604   
605   tmpDiv.onmouseout = MVUtil.closureWrapper(tmpDiv, function(evt)
606   {
607    if(tmpRectangleTool.dragStatus!=3)
608      return false;
609    
610    if(tmpRectangleTool.browser == "NS" || tmpRectangleTool.browser == "OS")
611    {
612      evt = (evt) ? evt : ((window.event) ? event : null);
613                
614      var clientx=evt.clientX;
615      var clienty=evt.clientY;
616 
617      var width = tmpMapView.getPaneWidth();
618      var height = tmpMapView.getPaneHeight();
619      var origXY = MVUtil.offsetPositionAbsolute(this);
620 
621      var OffsetDIVX = evt.pageX - origXY.x;
622      var OffsetDIVY = evt.pageY - origXY.y;
623      
624      if(OffsetDIVX<=0 || OffsetDIVX>=width || 
625      OffsetDIVY<=0 || OffsetDIVY>=height ||
626      clientx<=0 || clientx>=document.body.clientWidth ||
627      clienty<=0 || clienty>=document.body.clientHeight)
628      {     
629        return this.onmouseup(evt);
630      }
631    }
632   });
633   
634    //refresh the circle tool when refresh the ind foi
635    this.parent.indFOIsControl.attachEventListener("refreshFOI",this.refreshRetangle);
636    
637    MVUtil.runListeners(tmpRectangleTool.EventListeners,MVEvent.START,[MVToolBar.BUILTIN_DISTANCE]);
638    
639 }

651   this.parent.addEventHandlers();
652   
653 
654   MVUtil.runListeners(this.EventListeners,MVEvent.CLEAR);
655 }

657 /**
658  *This method returns the MVSdoGeometry object that specifies the rectangle on the map.
659  *@return the MVSdoGeometry object that specifies the rectangle on the map.
660  *@type MVSdoGeometry
661  *@see MVSdoGeometry
662  *@requires MVSdoGeometry
663 */
664 MVRectangleTool.prototype.getRectangle = function()
665 { 
666   if(this.fromLon!=this.toLon && this.fromLat!=this.toLat)
667   {
668   	var wholewidth = this.parent.msi.coverMaxX - this.parent.msi.coverMinX;
669 	if( (this.toLon - this.fromLon) < wholewidth/2)
670 	{
671 		this.sdoGeometry=MVSdoGeometry.createRectangle(
672 				  this.fromLon, this.fromLat, this.toLon, this.toLat, this.parent.srid);
673 		return this.sdoGeometry;
674 	}
675 	else
676 	{
677 		var minX = this.fromLon;
678 		var minY = this.fromLat;
679 		var maxX = this.toLon;
680 		var maxY = this.toLat;
681     var a = (maxX-minX)/3;
682     var b = (maxX-minX)/3*2;
683 		var rectanglePolygonArray = [minX, minY, minX, maxY, minX+a, maxY, minX+b, maxY, maxX, maxY, maxX, minY, minX+b, minY, minX+a, minY];
684 		this.sdoGeometry=MVSdoGeometry.createPolygon(rectanglePolygonArray, this.parent.srid);
685 		return this.sdoGeometry;
686 	}
687   }
688   else
689     return null;
690 }

808   }
809   else

811     //this.toLon = MVUtil.transLongitudeOnWarpAroungMap(this.parent, this.fromLon, this.toLon);

815   }

817 /**
818  *This method returns the proximate height of the rectangle.
819  *@param unit a string that specifies the name of the distance unit. If omitted, the 
820  * height is calculated in the default distance unit of the map coordinate system.
821  *@return the rectangle height.
822  *@type double
823 */
824 MVRectangleTool.prototype.getHeight = function(unit)
825 {
826 
827   var tmpFromY = this.fromLat;
828   var tmpToY = this.toLat
829 
830   if(tmpFromY > this.parent.msi.coverMaxY)
831     tmpFromY = this.parent.msi.coverMaxY;
832   else if(tmpFromY < this.parent.msi.coverMinY)
833     tmpFromY = this.parent.msi.coverMinY;
834 
835   if(tmpToY > this.parent.msi.coverMaxY)
836     tmpToY = this.parent.msi.coverMaxY
837   else if(tmpToY < this.parent.msi.coverMinY)
838     tmpToY = this.parent.msi.coverMinY
839   
840   if(!unit)
841     return Math.abs(tmpFromY-tmpToY);
842   else
843     return MVSdoGeometry.getDistance(this.fromLon, tmpFromY, this.fromLon, tmpToY, this.parent.srid, unit) ;
844 }
