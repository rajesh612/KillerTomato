1 /** @private
  2    The class defines an HTML table to hold the context menu 
  3  */
  4 function MVMenu(mv, loc,areaCode)
  5 {
  6   this.mapview = mv ;
  7   this.menuTableId = "mvmenu_table_0" ;
  8   this.menuTableBodyId = "mvmenu_table_body_0";
  9   this.menuFOIId = "mvmenu_0" ;
 10   mv.removeFOI(this.menuFOIId);
 11   var html = "<table id='" + this.menuTableId+"' cellpadding=1 cellspacing=0 style='width:1px'><tr>"+
 12              "<td style='border-width:1px;border-style:solid;border-color:black;background-color:white'>"+
 13              "<table><tbody id='"+this.menuTableBodyId+"'></tbody></table>"+
 14              "</td>"+
 15              "</tr></table>"
 16   var menu = MVFOI.createHTMLFOI(this.menuFOIId, loc, html, 0, 0) ;
 17   if(areaCode)
 18     menu.areacode = areaCode;
 19   menu.setInfoTip(null);
 20   menu.setInfoWindow(null, 0, 0);
 21   menu.setTopFlag(true);
 22   mv.addFOI(menu);
 23   var menuFOIId = this.menuFOIId;
 24   var removeRightClickMenu = function ()
 25   {
 26     mv.removeFOI(menuFOIId) ;
 27   }
 28   mv.addEventListener(MVEvent.MOUSE_CLICK, removeRightClickMenu) ;
 29 }
 30 
 31 MVMenu.count = 0 ;
 32 
 33 /** Add a menu item to the menu */
 34 MVMenu.prototype.addMenuItem = function(label, listener)
 35 {
 36   var menu = document.getElementById(this.menuTableBodyId) ;
 37   var wStr = document.getElementById(this.menuTableId).style.width ; //= label.length+"em";
 38   var currentWidth = parseInt(wStr.substr(0, wStr.length-2)) ;
 39   if(currentWidth < (label.length*12)) 
 40     document.getElementById(this.menuTableId).style.width = (label.length*12)+"px" ;
 41   var item = document.createElement('tr') ;
 42   menu.appendChild(item);
 43   item.style.fontFamily = "Arial, Helvetica, Geneva, sans-serif" ;
 44   item.style.fontSize = "12px" ;
 45   item.style.cursor = "pointer";
 46   item.align = 'left' ;
 47   var td = document.createElement("td") ;
 48   td.innerHTML = label ;
 49   item.appendChild(td);
 50   item.onmouseover = function()
 51   {
 52     this.style.backgroundColor = "#336699" ;
 53     this.style.color = "#ffffff" ;   
 54     
 55   }
 56   item.onmouseout = function()
 57   {
 58     this.style.backgroundColor = "transparent" ;
 59     this.style.color = "#000000" ;
 60   }
 61   if(listener)
 62   {
 63     var menu = this ;
 64     item.onmousedown = function(event)
 65     {
 66       MVUtil.stopEvent(event);
 67       menu.removeRightClickMenu();
 68       listener() ;
 69     }
 70   }
 71 }
 72 
 73 MVMenu.prototype.removeRightClickMenu = function ()
 74 {
 75   this.mapview.removeFOI(this.menuFOIId) ;
 76 }
