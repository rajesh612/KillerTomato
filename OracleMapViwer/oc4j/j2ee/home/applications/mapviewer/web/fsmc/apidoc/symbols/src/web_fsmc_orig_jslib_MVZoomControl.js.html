 69  /**
 70  private
 71  Calculate offset value of current layer div. point x/y on screen should adjust 
 72  by this value before call getScreenCor function, otherwise, the result will 
 73  not be correct.
 74  This method called in showTiles function.
 75  */
 76  MVZoomControl.prototype.computeTilesOffset = function() 
 77  {
 78     var div = this.mapLayer.tileArray[0][0];
 79     if (div && div.parentNode)
 80     {
 81       this.ldiff = -div.parentNode.offsetLeft
 82       this.tdiff = -div.parentNode.offsetTop;
 83       this.movedLeft = -div.parentNode.parentNode.offsetLeft;
 84       this.movedTop = -div.parentNode.parentNode.offsetTop;
 85     }
 86  }

 88  /**
 89  private
 90  This function get point cordinate by specify x, y pixel in screen and zoom level
 91  This method called in initZoomInfo to calculate a point cordinate.
 92  @param {int} x the x value of a screen point
 93  @param {int} y the y value of a screen point
 94  @param {int} zoom level of current map
 95  */
 96  MVZoomControl.prototype.getScreenCor = function(x, y, level) 
 97  {    
 98    var ppx = this.getPPXByZoomLevel(level);
 99    var ppy = this.getPPYByZoomLevel(level);
100    var tx = x-this.mapview.getPaneWidth()/2 ;
101    var ty = y-this.mapview.getPaneHeight()/2 ;
102    return {x: tx/ppx + this.mapview.mapCenterX, 
103            y: this.mapview.mapCenterY-ty/ppy};
104  }

106  /**
107  private
108  This function get point screen pixel x, y by specify cordinate and zoom level
109  This method called in initZoomInfo to calculate a point location in 
110  screen at specified zoom level.
111  @param {int} x the x value of cordinate
112  @param {int} y the y value of cordinate
113  @param {int} level target zoom level
114  */
115  MVZoomControl.prototype.getScreenXY = function(x, y, level) 
116  {
117    var ppx = this.getPPXByZoomLevel(level);
118    var ppy = this.getPPYByZoomLevel(level);
119    
120    return {x: (x-this.mapview.mapCenterX)*ppx + this.mapview.getPaneWidth()/2, 
121            y: (this.mapview.mapCenterY-y)*ppy + this.mapview.getPaneHeight()/2};
122  }

124  /**
125  private
126  Calculate pixel per x in specified zoom level
127  This method called in getScreenCor and getScreenXY.
128  @param {int} level the zoom level
129  */
130  MVZoomControl.prototype.getPPXByZoomLevel = function(level)
131  {
132    var tileWidth = this.mapview.msi.mapConfig.zoomLevels[level].tileImageWidth;
133    var tw = this.mapview.msi.mapConfig.zoomLevels[level].tileWidth;
134    var pixPerX = tileWidth / tw ;  
135    return pixPerX;
136  }

138  /**
139  private
140  Calculate pixel per y in specified zoom level
141  This method called in getScreenCor and getScreenXY.
142  @param {int} level the zoom level
143  */
144  MVZoomControl.prototype.getPPYByZoomLevel = function(level) //get pixPerY by zoom level
145  {
146    var tileHeight = this.mapview.msi.mapConfig.zoomLevels[level].tileImageHeight;
147    var th = this.mapview.msi.mapConfig.zoomLevels[level].tileHeight; 
148    var pixPerY = tileHeight / th ; 
149    return pixPerY;
150  }

152  /**
153  private
154  Calculate size and position of a tile, both of start value (before zoom) and 
155  end value (after zoom). Then showTiles function will change the size and 
156  postion little by little from start value to end value
157  This method called in showTiles function.
158  @param {int} fromLevel the zoom level before zoom
159  @param {int} toLevel the zoom level after zoom
160  @param {DIV} tile the target tile div
161  */
162  MVZoomControl.prototype.initZoomInfo = function(fromLevel, toLevel, tile) 
163  {
164    var fromPpx = this.getPPXByZoomLevel(fromLevel);
165    var fromPpy = this.getPPYByZoomLevel(fromLevel);
166 
167    var toPpx = this.getPPXByZoomLevel(toLevel);
168    var toPpy = this.getPPYByZoomLevel(toLevel);
169     
170    var fw = this.mapview.msi.mapConfig.zoomLevels[fromLevel].tileImageWidth;
171    var fh = this.mapview.msi.mapConfig.zoomLevels[fromLevel].tileImageWidth;
172    var tiw = this.mapview.msi.mapConfig.zoomLevels[fromLevel].tileImageWidth * toPpx / fromPpx;
173    var tih = this.mapview.msi.mapConfig.zoomLevels[fromLevel].tileImageHeight * toPpy / fromPpy;
174     
175    var lstart = MVUtil.getLeft(tile) - this.movedLeft;
176    var tstart = MVUtil.getTop(tile) - this.movedTop;
177    
178    var ltCor = this.getScreenCor(lstart - this.ldiff + this.movedLeft , tstart - this.tdiff + this.movedTop, fromLevel);
179    var ltXY = this.getScreenXY(ltCor.x, ltCor.y, toLevel);
180    
181    var lend = ltXY.x + this.ldiff - this.movedLeft*toPpx/fromPpx;
182    var tend = ltXY.y + this.tdiff - this.movedTop*toPpx/fromPpx;
183     
184    tile.oldZoomInfo = new Object();
185    
186    tile.oldZoomInfo.lstart = lstart;
187    tile.oldZoomInfo.tstart = tstart;
188    tile.oldZoomInfo.lend = lend;
189    tile.oldZoomInfo.tend = tend;
190    
191    tile.oldZoomInfo.fw = fw;
192    tile.oldZoomInfo.fh = fh;
193    tile.oldZoomInfo.tiw = tiw;
194    tile.oldZoomInfo.tih = tih;
195  }

197  /**
198  private
199  ShowTiles function will change the size and postion little by little from 
200  start value to end value, this function is actually doing the little steps.
201  This method called in runZoom function.
202  @param {DIV} div the tile div
203  @param {float} factor current step of zooming, a float value from 0 to 1
204  */
205  MVZoomControl.prototype.zoomToFactor = function(div, factor) 
206  {   
207    if(factor>1) 
208    {
209      return;
210    }
211     
212    var oz = div.oldZoomInfo;
213     
214    var l = (oz.lend - oz.lstart) * factor + oz.lstart;
215    var t = (oz.tend - oz.tstart) * factor + oz.tstart;
216    var w = (oz.tiw - oz.fw) * factor + oz.fw;
217    var h = (oz.tih - oz.fh) * factor + oz.fh;
218     
219    MVUtil.setLeftTop(div, l, t);
220    
221    div.style.width = (w+1) + 'px';
222    div.style.height = (h+1) + 'px'; 
223  }

225  /**
226  private
227  Copy layer div and tiles array from MVMapTilesControl to MVZoomControl. This 
228  is a pre-process of zooming animation called in showTiles function. After do
229  this, tile div will zoomed one by one. If the tile DIV/IMG has not finished 
230  loading, it will stop loading by set its src to null.
231  */
232  MVZoomControl.prototype.copyTiles = function() 
233  {
234    if(this.tileLayerDiv && this.tileLayerDiv.parentNode)
235    {
236      this.tileLayerDiv.parentNode.removeChild(this.tileLayerDiv);
237      MVUtil.deleteNode(this.tileLayerDiv);
238    }
239 
240    this.tileLayerDiv = this.mapLayer.mapLayer;
241    this.tileLayerDiv.zooming = true;
242    this.tileArray = this.mapLayer.tileArray;
243    this.tileLayerDiv.style.zIndex = 200 + this.mapLayer.zIndex;
244    return;
245  }

247  /**
248  private 
249  If the map has move to new center by draging, the offset value should be 
250  calculated, and adjust div left/top value accordingly
251  @param {MVGeometry} oldCenter the old center
252  @param {MVGeometry} newCenter the new center
253  @param {MVGeometry} oldLevel the zoom level before zoom 
254  */
255  MVZoomControl.prototype.zoomToNewCenter = function(oldCenter, newCenter, oldLevel) 
256  {
257    if((oldCenter) && (newCenter) && (oldCenter.getPointX() != newCenter.getPointX() || oldCenter.getPointY() != newCenter.getPointY()))
258    {
259      var oldCXY = this.getScreenXY(oldCenter.getPointX(), oldCenter.getPointY(), oldLevel);
260      var newCXY = this.getScreenXY(newCenter.getPointX(), newCenter.getPointY(), oldLevel);
261 
262      var xdiff = newCXY.x - oldCXY.x;
263      var ydiff = newCXY.y - oldCXY.y;
264 
265      var p = this.mapview.div;
266      if(MVGlobalVariables.browserClass==2)
267      {
268        p.style.pixelLeft = p.style.pixelLeft - xdiff;
269        p.style.pixelTop = p.style.pixelTop - ydiff;
270      }
271      else
272      {
273        p.style.left = (MVUtil.getLeft(p) - xdiff)+'px';
274        p.style.top = (MVUtil.getTop(p) - ydiff)+'px';
275        
276      }
277    }
278  }

280  /**
281  private 
282  If last zooming didn't finished, clear the temporary data and timeouts
283  */
284  MVZoomControl.prototype.clearLastZoom = function() {
285    for(var i=0; i<this.mapLayer.tileArray.length; i++) 
286    {
287      for(var j=0; j<this.mapLayer.tileArray[i].length; j++) 
288      {
289        if(!this.hasTileLoaded(this.mapLayer.tileArray[i][j]))
290        {
291          var src = MVGlobalVariables.imgFilePath + "transparent."+(MVGlobalVariables.browser=="IF"?"gif":"png") ;
292          this.mapLayer.tileArray[i][j].src = src;
293          this.mapLayer.tileArray[i][j].imgLoadFinished = true;
294        }
295      }
296    }
297 
298    //clear all timeout
299    for(var i=0; i<this.timeouts.length; i++) 
300    {
301      clearTimeout(this.timeouts[i]);
302    }
303    this.timeouts = new Array();
304  }

306  /**
307  private 
308  Entrance function of zooming animation. It will execute pretreatments and 
309  then run zooming animation on individual tile one by one
310  @param {MVGeometry} fromLevel the zoom level before zoom
311  @param {MVGeometry} toLevel the zoom level after zoom
312  @param {MVGeometry} oldCenter the old center
313  @param {MVGeometry} newCenter the new center
314  */
315  MVZoomControl.prototype.showTiles = function(fromLevel, toLevel, oldCenter, newCenter) 
316  { 
317    //if(this.mapLayer.tileArray == '' || Math.abs(fromLevel - toLevel) > 1 || this.zooming || this.mapLayer.isTransparent ) 
318    if(!this.mapLayer.isVisible() || this.mapLayer.tileArray == '' || Math.abs(fromLevel - toLevel) > 1 || this.mapLayer.isTransparent) 
319    {
320      this.reset();
321      return;
322    }
323 
324    // hide the tile layer until zooming animation is done.
325    this.mapLayer.setVisible(false);
326    // set the map tile div container to visible because tiles inside it will be animated
327    this.mapLayer.mapLayer.style.visibility = "visible";
328    
329    this.zooming = true;
330    this.zoomToNewCenter(oldCenter, newCenter, fromLevel);
331    this.computeTilesOffset()
332    this.copyTiles();
333    this.clearLastZoom();
334     
335    for(var i=0; i< this.tileArray.length; i++) 
336    {
337      for(var j=0; j<this.tileArray[i].length; j++) 
338      {
339        this.initZoomInfo(fromLevel, toLevel, this.tileArray[i][j]);   
340        this.runZoom(this, i, j);  
341      }
342    }
343  }

346   private
347   Clear current MVZoomControl object to default value, clear temporary data, timeouts.
348   */
349  MVZoomControl.prototype.reset = function() 
350  {
351     this.ldiff = 0;
352     this.tdiff = 0;
353     this.movedLeft = 0;
354     this.movedTop = 0;
355     
356     if(this.tileLayerDiv && this.tileLayerDiv.parentNode)

360     }
361 
362     this.tileArray = null;
363     this.tileLayerDiv = null;
364     this.zoomStep = 0.2
365     this.interval = 100;
366     this.zooming = false;
367     this.oldTilesLoaded = false;
368 
369     //clear all timeout
370     for(var i=0; i<this.timeouts.length; i++) 
371     {
372       clearTimeout(this.timeouts[i]);
373     }
374     this.timeouts = new Array();
375     if(this.mapLayer.basemap.isVisible())
376       this.mapLayer.setVisible(true);
377  }

379  /**
380  private
381  Run zooming animation on individual tile.
382  @param {MVZoomControl} zc target MVZoomControl object
383  @param {int} ti i-index of current tile in tileArray[i][j]
384  @param {int} ti j-index of current tile in tileArray[i][j]
385  @param {float} factor current step of zooming, a float value from 0 to 1
386  */
387  MVZoomControl.prototype.runZoom = function(zc, ti, tj, factor) 
388  {
389    factor = (factor) ? factor : zc.zoomStep;
390    var tile = zc.tileArray[ti][tj];
391    zc.zoomToFactor(tile, factor);
392    if(factor>=1) 
393    {
394      tile.oldZoomInfo = null;
395      tile.zoomFinished = true;
396      if(this.hasZoomFinished()) 
397      {
398        this.startClear();
399      }
400      return;
401    } 
402    else 
403    {
404      var st = setTimeout(function(){zc.runZoom.call(zc, zc, ti, tj, factor + zc.zoomStep);}, zc.interval);
405      this.timeouts.push(st);
406    }    
407  }

409  /**
410  private
411  Whether all tiles have zoomed from old size and position to new size and position.
412  */
413  MVZoomControl.prototype.hasZoomFinished = function() 
414  {
415    if(!(this.tileArray)) 
416    {
417      return true;
418    }
419    for(var i=0; i< this.tileArray.length; i++) 
420    {
421      for(var j=0; j < this.tileArray[i].length; j++) 
422      {
423        if(!this.tileArray[i][j].zoomFinished) 
424        {
425          return false ;
426        }
427      }
428    }
429    return true;
430  }

451  /*
452   private
453   tells whether current tile has already loaded
454   @parameter tile  Current tile
455   @return true if loaded, false if not
456   */
457  MVZoomControl.prototype.hasTileLoaded = function(tile) 
458  {
459    if(tile.src == this.mapview.transImage.src)
460      return true ;
461    if(MVGlobalVariables.browserClass==2 ) //IE
462    {
463      return tile.imgLoadFinished;
464    }
465    else //Firefox
466    {
467      return (tile.offsetWidth == this.mapLayer.tileWidth);
468    }

471  /**
472  private
473  Call this function to clear temporary tiles and set real tiles FOIs to be 
474  show, it will then call clearCloneTiles to actually clear temporary tiles 
475  when tiles loaded
476  */
477  MVZoomControl.prototype.startClear = function() 
478  {
479    // show the new tiles
480    this.mapLayer.setVisible(true);
481    //show all individual FOIs
482    for(var i=0; i<this.mapview.indFOIsControl.indFOIArray.length; i++) 
483    {
484      var node = this.mapview.indFOIsControl.indFOIArray[i].node;
485      if (node instanceof Array)
486      {
487        //to deal with FOI nodes on wrap-around map
488        for(var j=0;j<node.length;j++)
489          node[j].style.display = '';
490      }
491      else if (node!=null)
492        node.style.display = '';
493    }
494    //show all group FOIs
495    if (this.mapview.foiGroupArray != null)
496    {
497      for(var i=0; i<this.mapview.foiGroupArray.length; i++) 
498      {
499        var group = this.mapview.foiGroupArray[i];
500        for(var j=0; j<group.pointArray.length; j++)
501        {
502          var foi = group.pointArray[j];
503          if (foi.displayed && foi.node)
504          {
505            if (foi.node instanceof Array)
506            {
507              for(var m=0;m<foi.node.length;m++)
508                foi.node[m].style.display = '';
509            }
510            else
511              foi.node.style.display = '';
512          }
513        }
514        for(var j=0; j<group.polygonArray.length; j++)
515        {
516          var foi = group.polygonArray[j];
517          if (foi.displayed && foi.node)
518          {
519            if (foi.node instanceof Array)
520            {
521              for(var m=0;m<foi.node.length;m++)
522                foi.node[m].style.display = '';
523            }
524            else
525              foi.node.style.display = '';
526          }
527        }
528        if (group.clusterFOIArray!=null)
529        {
530          for(var j=0; j<group.clusterFOIArray.length; j++)
531          {
532            var foi = group.clusterFOIArray[j];
533            if (foi.node instanceof Array)
534            {
535              for(var m=0;m<foi.node.length;m++)
536                foi.node[m].style.display = '';
537            }
538            else
539              foi.node.style.display = '';
540            
541          }
542        }
543        if (group.overlayFOIArray!=null)
544        {
545          for(var j=0; j<group.overlayFOIArray.length; j++)
546          {
547            var foi = group.overlayFOIArray[j];
548            if (foi.node instanceof Array)
549            {
550              for(var m=0;m<foi.node.length;m++)
551                foi.node[m].style.display = '';
552            }
553            else
554              foi.node.style.display = '';
555          }
556        }
557      }
558    }
559    if(MVGlobalVariables.browserClass==2 ) //IE
560    {
561      //show all tiles
562      for(var i=0; i<this.mapLayer.tileArray.length; i++) 
563      {
564        for(var j=0; j<this.mapLayer.tileArray[i].length; j++) 
565        {
566          this.mapLayer.tileArray[i][j].style.display='';
567        }
568      }
569    }
570    this.tileLayerDiv.style.zIndex = this.mapLayer.zIndex;;
571    this.clearCloneTiles();
572  }

614  /**
615  private 
616    Scale the tile layer based on the current zoom level.
617    @param scale the scale relative to the current zoom level scale
618  */
619  MVZoomControl.prototype.scaleTiles = function(scale) 
620  { 
621    //if(this.mapLayer.tileArray == '' || Math.abs(fromLevel - toLevel) > 1 || this.zooming || this.mapLayer.isTransparent ) 
622    if(!this.scaling && (!this.mapLayer.isVisible() || this.mapLayer.tileArray == '')) 
623    {
624      reset() ;
625      return;
626    }
627 
628    // hide the tile layer until zooming animation is done.
629    this.mapLayer.setVisible(false);
630    // set the map tile div container to visible because tiles inside it will be animated
631    
632    this.mapLayer.mapLayer.style.visibility = "visible";
633    
634    if(!this.scaling)
635    {
636      this.computeTilesOffset()
637      this.copyTiles();
638      this.layerLeft = this.tileLayerDiv.offsetLeft;
639      this.layerTop = this.tileLayerDiv.offsetTop;
640      this.parentLeft = this.tileLayerDiv.parentNode.offsetLeft ;
641      this.parentTop = this.tileLayerDiv.parentNode.offsetTop ;
642      this.scaling = true ;
643      for(var i=0; i< this.tileArray.length; i++) 
644      {
645        for(var j=0; j<this.tileArray[i].length; j++) 
646        {
647          var tile = this.tileArray[i][j];
648          tile.origLeft = MVUtil.getLeft(tile) ;
649          tile.origTop = MVUtil.getTop(tile) ;
650        }
651      }
652    }
653    
654    var zoomLevel = this.mapview.getZoomLevel() ;
655    var tw = this.mapview.msi.mapConfig.zoomLevels[zoomLevel].tileImageWidth * scale;
656    var th = this.mapview.msi.mapConfig.zoomLevels[zoomLevel].tileImageWidth * scale;
657 
658    var xOffset = this.mapview.getPaneWidth()/2 + this.ldiff + this.movedLeft ;
659    var yOffset = this.mapview.getPaneHeight()/2 + this.tdiff + this.movedTop ;
660    
661    /*this.zooming = true;
662    this.clearLastZoom();*/
663    
664    for(var i=0; i< this.tileArray.length; i++) 
665    {
666      for(var j=0; j<this.tileArray[i].length; j++) 
667      {
668        var tile = this.tileArray[i][j];
669        this.scaleTile(tile, scale, tw, th, xOffset, yOffset);
670      }
671    }
672  }

674  /**
675   * SCale one tile
676   * @param div
677   * @param scale
678   * @param w
679   * @param h
680   * @param xOffset
681   * @param yOffset
682   * @return
683   */
684  MVZoomControl.prototype.scaleTile = function(div, scale, w, h, xOffset, yOffset) 
685  {   
686    div.zoomFinished = true;
687    if(scale==1) 
688    {
689      return;
690    }
691 
692    var l = xOffset - (xOffset - div.origLeft)*scale ;
693    var t = yOffset - (yOffset - div.origTop)*scale ;
694    
695    MVUtil.setLeftTop(div, l, t);
696    
697    div.style.width = (w+1) + 'px';
698    div.style.height = (h+1) + 'px'; 
699  }

701  MVZoomControl.prototype.endScaling = function()
702  {
703    this.scaling = false ;
704    this.mapLayer.setVisible(true);
705    for(var i=0; i<this.mapLayer.tileArray.length; i++) 
706    {
707      for(var j=0; j<this.mapLayer.tileArray[i].length; j++) 
708      {
709        this.mapLayer.tileArray[i][j].style.display='';
710      }
711    }
712    
713    var layerLeft = this.tileLayerDiv.offsetLeft;
714    var layerTop = this.tileLayerDiv.offsetTop;
715    var parentLeft = this.tileLayerDiv.parentNode.offsetLeft ;
716    var parentTop = this.tileLayerDiv.parentNode.offsetTop ;
717    
718    MVUtil.setLeftTop(this.tileLayerDiv, layerLeft + this.parentLeft, layerTop + this.parentTop);
719    this.startClear();
720    //this.removeCloneTiles() ;
721 }
