 24 MVCSTransformSphereMercator.geodeticToMercator = function(lon, lat, earthRadius)
 25 {
 26   var radians = MVGeomUtil.getRadiansFromLatLon({x:lon, y:lat}) ;
 27   lon = radians.x ;
 28   lat = radians.y ;
 29   var sin = Math.sin(lat) ;
 30   var log = Math.log((1 + sin) / (1 - sin)) ;
 31   var rx  = earthRadius * lon ;
 32   var ry  = log *  earthRadius /2 ;
 33   return {x:rx, y:ry};
 34 }
 35 
 36 MVCSTransformSphereMercator.mercatorToGeodetic = function(x, y, earthRadius)
 37 {
 38   var lon = x/earthRadius;
 39   var log = y*2/earthRadius ;
 40   var p = Math.pow(Math.E, log) ;
 41   var sin = (p-1)/(p+1) ;
 42   var lat = Math.asin(sin) ;
 43   return MVGeomUtil.getLatLonFromRadians({x:lon, y:lat});
 44 }
