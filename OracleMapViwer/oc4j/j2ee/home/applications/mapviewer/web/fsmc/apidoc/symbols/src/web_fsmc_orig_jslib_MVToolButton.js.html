 27 /**
 28  *@class MVToolButton defines a clickable tool button displayed in a MVToolBar object. 
 29  * It response to mouse clicks by invoking the corresponding event handlers 
 30  * defined by the application.
 31  *
 32  * A tool button can be defined in one of the following three types: 
 33  * <UL>
 34  *   <LI> COMMAND:   A command button has only one state, UP, and always reacts to mouse 
 35  *                   clicks by invoking the same BUTTON_DOWN event listener.
 36  *   <LI> TOGGLE:    A toggle button has two states, DOWN and UP. The initial state of
 37  *                   the button is UP. Its state changes to DOWN after it is clicked. 
 38  *                   It stays DOWN until it is clicked again, after which its state
 39  *                   changes to UP. The BUTTON_DOWN event listener is invoked when the 
 40  *                   button state changes from UP to DOWN. The BUTTON_UP event listener
 41  *                   is invoked when the button state changes from DOWN to UP. 
 42  *   <LI> SEPERATOR: A seperator simply displays a static seperator icon that does not
 43  *                   react to any user actions. 
 44  * </UL>
 45  *
 46  *@constructor
 47  *@param {String} buttonID A string that specifies the id of the button. 
 48  *@param {integer} buttonType ButtonType specifies the type of the new created button.
 49  *             It can be MVToolButton.COMMAND, MVToolButton.TOGGLE or MVToolButton.SEPARATOR.
 50  *             The default type is MVToolButton.COMMAND.
 51  *@param {string} buttonIcon A string specifies a icon to be displayed when the button is in 
 52  *             the UP state.
 53  *@param {string} buttonSelectedIcon A string specifies a icon to be displayed when the button 
 54  *             is in the DOWN state.
 55  *@return The newly created MVToolButton object.
 56 */    
 57 function MVToolButton(buttonID,buttonType,buttonIcon,buttonSelectedIcon, width, height)
 58 {
 59     /**@private*/
 60     this.btID = buttonID;
 61     /**@private*/
 62     this.btType = buttonType;
 63     /**@private*/
 64     this.btIcon = buttonIcon;
 65     /**@private*/
 66     this.btSelectedIcon = buttonSelectedIcon; 
 67     /**@private*/
 68     this.btState = MVToolButton.UP;
 69     /**@private*/
 70     this.parentToolBar = null;
 71 
 72     /**@private*/
 73     this.EventListeners = new Object();
 74     /**@private*/
 75     this.onButtonSelectedListener = null;
 76     /**@private*/
 77     this.onButtonUnselectedListener = null;
 78     /**@private*/
 79     this.onButtonStateChangeListener = null;
 80     
 81     /**@private*/
 82     this.btDiv = document.createElement("div");
 83     /**@private*/
 84     this.btImg = document.createElement("img");
 85     /**@private*/
 86     this.tipDiv = document.createElement("div");
 87 	
 88 	  /**@private*/
 89     this.btName = null;
 90 
 91     /**@private*/
 92     this.width = width;
 93     /**@private*/
 94     this.height = height;
 95     
 96     var bt = this;
 97     /**@private*/
 98     this.destroy = function()
 99     {
100       MVUtil.clearNode(bt.btDiv) ;
101       bt.btDiv = null;
102       MVUtil.clearNode(bt.btImg) ;
103       bt.btImg = null;
104       MVUtil.clearNode(bt.tipDiv) ;
105       bt.tipDiv = null;
106       bt = null ;
107     }
108 
109     /**
110      *This method attaches an event listener that handles one of the following tool button events: 
111      MVEvent.BUTTON_DOWN and MVEvent.BUTTON_UP 
112      *<ul>
113      *<li>
114      *A MVEvent.BUTTON_DOWN event is fired when the user clicks a button and its state changes from 
115      *UP to DOWN.
116      *</li>
117      *<li>
118      *A MVEvent.BUTTON_UP event is fired when the user clicks a button and its state changes from DOWN
119      *to UP. 
120      *</li>
121      *</ul>
122      *@param {String} event a String that specifies the type of event to be
123      * handled by the listener. It can only be MVEvent.BUTTON_DOWN or MVEvent.BUTTON_UP.
124      *@param {Function} listener a javascript listener function. No
125      *parameter is passed to the listener when it is called.
126      *@return none
127      *@type void
128      */    
129     this.attachEventListener = function(type, listener)
130     {
131         MVUtil.attachEventListener(bt.EventListeners,type,listener)
132     }
133     /**
134      *This method detaches an event listener that has previously been attached to the tool button
135      *by {@link #attachEventListener}.
136      *@param {String} event a String that specifies the type of event handled
137      * by the listener. It can only be MVEvent.BUTTON_DOWN or MVEvent.BUTTON_UP. 
138      *@param {Function} listener a javascript listener function. No 
139      *parameter is passed to the listener when it is called.
140      *@return none
141      *@type void
142     */
143     this.detachEventListener = function(type, listener)
144     {
145         MVUtil.detachEventListener(bt.EventListeners,type, listener);
146     }
147     
148     /**
149     * This method adds a tooltip to the button. The tooltip is displayed when 
150     * the mouse moves over the button.
151     * @param {string} tooltip a string that specifies the content of the tooltip.
152     * @returns None
153     *@type void
154      */
155     this.setToolTip = function(tooltip)
156     {
157         bt.tipDiv.innerHTML = tooltip;
158     }    
159     
160     /**
161      * This method resets and releases the button if it is pressed down. 
162      * @returns None
163      * @type void
164      */
165     this.reset = function()
166     {
167         this.forceBtUp() ;
168     }
169 
170     /**@private*/
171     this.initButton = function()
172     {
173         bt.btDiv.appendChild(bt.btImg);
174         bt.btDiv.appendChild(bt.tipDiv);
175 
176         bt.tipDiv.style.position = "absolute";
177         bt.tipDiv.style.display = "none";
178         bt.tipDiv.style.borderColor = "black";
179         bt.tipDiv.style.backgroundColor = "#ffffe1";
180         bt.tipDiv.style.borderWidth = "1px";
181         bt.tipDiv.style.borderStyle ="solid";
182         bt.tipDiv.style.fontSize ="14px";
183         bt.tipDiv.style.whiteSpace ="nowrap";
184         bt.tipDiv.style.padding ="2px";
185 
186         if(bt.btType != MVToolButton.SEPARATOR)
187         {
188             bt.btImg.setAttribute("src", bt.btIcon);
189             if(bt.width)
190               bt.btImg.setAttribute("width", bt.width) ;
191             if(bt.height)
192               bt.btImg.setAttribute("height", bt.height) ;
193         }
194     }
195 
196     /**@private*/
197     this.btImg.onmouseover = function()
198     {
199         if(bt.btType != MVToolButton.SEPARATOR)
200         {
201             MVUtil.setCursor(bt.parentToolBar.toolBarDecorationDiv,"pointer") ;
202             if(bt.tipDiv.innerHTML)
203             {
204                 bt.tipDiv.style.display = "";
205                 MVUtil.setLeftTop(bt.tipDiv, bt.btDiv.parentNode.offsetLeft+5, bt.btDiv.parentNode.offsetTop+30);
206             }
207         }
208         else
209         {
210             MVUtil.setCursor(bt.parentToolBar.toolBarDecorationDiv,"default");
211         }
212     }
213     
214     /**@private*/
215     this.btImg.onmouseout = function(evt)
216     {
217         if(bt.btState == MVToolButton.DOWN)
218             bt.btImg.onmouseup(evt);
219         bt.tipDiv.style.display = "none";
220     }
221     
222     /**@private*/
223     MVUtil.setListener(this.btImg, "mousedown", function(evt)
224     {
225         if(bt.onButtonStateChangeListener)
226             bt.onButtonStateChangeListener(bt.btID);
227         
228         if(bt.btType == MVToolButton.COMMAND)
229         {
230             bt.btImg.setAttribute("src", bt.btSelectedIcon);
231             bt.btState = MVToolButton.DOWN;
232             evt = (evt) ? evt : ((window.event) ? event : null);
233             MVUtil.stopEvent(evt);
234             
235             if(bt.onButtonSelectedListener)
236                 bt.onButtonSelectedListener(bt.btID);
237             MVUtil.runListeners(bt.EventListeners,MVEvent.BUTTON_DOWN,[bt.btID]);
238         }
239         else if(bt.btType == MVToolButton.TOGGLE)
240         {
241             if(bt.btState == MVToolButton.UP)
242             {
243                 bt.btImg.setAttribute("src", bt.btSelectedIcon);
244                 bt.btState = MVToolButton.DOWN;
245                 evt = (evt) ? evt : ((window.event) ? event : null);
246                 MVUtil.stopEvent(evt);                            
247                 
248                 if(bt.onButtonSelectedListener)
249                     bt.onButtonSelectedListener(bt.btID);
250                 MVUtil.runListeners(bt.EventListeners,MVEvent.BUTTON_DOWN,[bt.btID]);
251             }
252             else
253             {
254                 bt.btImg.setAttribute("src", bt.btIcon);        
255                 bt.btState = MVToolButton.UP;
256                 evt = (evt) ? evt : ((window.event) ? event : null);
257                 MVUtil.stopEvent(evt);    
258                 
259                 if(bt.onButtonUnselectedListener)
260                     bt.onButtonUnselectedListener(bt.btID);
261                 MVUtil.runListeners(bt.EventListeners,MVEvent.BUTTON_UP,[bt.btID]);
262             }
263         }
264     }, true);
265     
266     /**@private*/
267     MVUtil.setListener(this.btImg, "mouseup", function(evt)
268     {
269         if(bt.btType == MVToolButton.COMMAND && bt.btState == MVToolButton.DOWN)
270         {
271             bt.btImg.setAttribute("src", bt.btIcon);
272             
273             if(bt.onButtonUnselectedListener)
274                 bt.onButtonUnselectedListener(bt.btID);
275             MVUtil.runListeners(bt.EventListeners,MVEvent.BUTTON_UP,[bt.btID]);
276                 
277             bt.btState = MVToolButton.UP;
278         }
279     }, true) ;
280     
281     /**@private*/
282     //used by MVButtonGroup, force only on button in a group is selected
283     this.forceBtUp = function(toolId,clearTag)
284     {
285         if(bt.btType == MVToolButton.TOGGLE && (toolId == undefined || bt.btID != toolId))
286         {
287             if(bt.btState == MVToolButton.DOWN)
288             {
289                 bt.btImg.setAttribute("src", bt.btIcon);        
290                 bt.btState = MVToolButton.UP;
291 				
292 				if(clearTag==null || !clearTag)
293 					MVUtil.runListeners(bt.EventListeners,MVEvent.BUTTON_UP,[bt.btID]);
294                 
295                 /*if(bt.onButtonUnselectedListener)
296                     bt.onButtonUnselectedListener(bt.btID);
297                 MVUtil.runListeners(bt.EventListeners,MVEvent.BUTTON_UP,[bt.btID]);*/
298             }
299         }
300     }
301     
302     /**@private*/
303     bt.initButton();
304 
305 }
306 
307 
308 MVToolButton.COMMAND = 0;
309 MVToolButton.TOGGLE = 1;
310 MVToolButton.SEPARATOR = 2;
311 
312 MVToolButton.UP = 0;
313 MVToolButton.DOWN = 1;
