 29 /** 
 30  @class MVScaleBar defines a scale bar object that is to be displayed inside
 31  a map decoration(MVMapDecoration) object. 
 32  The visibility and positioning of the scale bar is determined by the 
 33  map decoration that holds the scale bar. 
 34  An existing scale bar can be removed by removing the correspondind map 
 35  decoration object from the map.
 36  <br> 
 37  
 38  Typical usage of this class is illustrated below.
 39  <pre>
 40     var md=new MVMapDecoration(new MVScaleBar(),0,1,null,null,4,-45);
 41     mapview.addMapDecoration(md);
 42  </pre>
 43    
 44     
 45  @param {object} options optional, an object that specifies the rendering options of the scale bar. 
 46             It can have the following three attributes.
 47               <UL>
 48                 <LI>transformToGeodetic: A boolean that tells the map client whether to convert projected 
 49                 coordinates to geodetic coordinates and calculate the scale bar using sphere surface 
 50                 distances. It should be used only for map images rendered in projected systems. </LI>
 51                 <LI>maxWidth: A number that specifies the maximum pixel width of the scale bar.</LI>
 52                 <LI>format: A string that specifies the scale bar format, which can be one of the following, 
 53                 "METRIC", "IMPERIAL" and "BOTH".</LI>
 54               </UL>
 55  @constructor
 56 */
 57 function MVScaleBar(options)
 58 {
 59   //Deal the argument
 60   if(options == null || options =="undefined")
 61   {
 62     /**@private*/
 63     this.format = "BOTH";
 64     /**@private*/
 65     this.maxLength = (arguments.length == 2 && typeof arguments[1] == "number") ? arguments[1] : 125;
 66     /**@private*/
 67     this.transformToGeodetic = false;
 68   }
 69   //transformToGeodetic, maxWidth and format are involved in the object.
 70   else if(typeof options == "object")
 71   {      
 72     /**@private*/
 73     this.format = options.format ? options.format : "BOTH";
 74     /**@private*/
 75     this.maxLength = (options.maxWidth && (typeof options.maxWidth == "number")) ? options.maxWidth : 125;
 76     /**@private*/
 77     this.transformToGeodetic = (options.transformToGeodetic && (typeof options.transformToGeodetic == "boolean")) ? options.transformToGeodetic : false;
 78   }
 79   //transformToGeodetic and maxWidth are specified directly by the arguments. This is for being compatible with customer's old codes.
 80   else
 81   {
 82      /**@private*/
 83     this.format = "BOTH";
 84     /**@private*/
 85     this.maxLength = (arguments.length == 2 && typeof arguments[1] == "number") ? arguments[1] : 125;
 86     /**@private*/
 87     this.transformToGeodetic = (typeof options == "boolean") ? options : false;
 88   }
 89 
 90   /**@private*/
 91   this.type = "MVScaleBar" ;
 92   /**@private*/
 93   this.isMapControl = true ;
 94 
 95   /**@private*/
 96   this.insideMapDecoration = false ;
 97   /**@private*/
 98   this.textSize = 11 ;
 99   /**@private*/
100   this.mvHandle = null ;
101   /**@private*/
102   this.initialized = false ;
103   /**@private*/
104   this.style = 
105     { barStyle:"border-top:1px solid #AAAAAA;border-bottom:1px solid #555555;border-left:1px solid #AAAAAA;border-right:1px solid #555555;background-color:#D0E0F0",
106       barHeight:3, textStyle: "font-family: Tahoma;font-size:11px;font-weight:bold"};
107   /**@private*/
108   this.barSpacing = 7 ;
109 }
110  /**
111    This methods sets the visual style of the scale bar object. 
112    <p>The style parameter is an object that specifies aspects such as scale bar style, height and 
113    text style. The object has the following attributes:
114    <UL>
115      <LI>barStyle : The CSS style string that specifies the color and border of the scale bar. 
116      <LI>barHeight: The height of the scale bar in screen pixels,
117      <LI>textStyle : The CSS style string that specifies the font style of the scale measurement text.

120    <pre>

141   }

161     par.scalectl = this;
162   this.mvHandle = par ;
163   if(!this.div)
164   {
165     this.div=document.createElement("div");
166     this.div.style.zIndex = 2000;

169     else

178   this.div.style.width=MVUtil.pixel(125);
179   this.div.style.fontSize=MVUtil.pixel(5);
180   var ms = this;
181   var tmpcallBack = function(mscale)

183     var fl = mscale.fpsLength;
184     var ml = mscale.metricLength;
185     var e = Math.max(fl,ml);
186     
187     var topTextY = 3 ;

212     }
213     else

248 /**@private*/
249 MVScaleBar.prototype.update=function(par)
250 {
251   if(!this.div || !this.initialized)
252     return ;
253   var ms = this;
254   var callBack = function(mscale)
255   {
256     var ml=Math.max(mscale.fpsLength,mscale.metricLength);
257     ms.setText(ms.fps,mscale.fpsText);
258     ms.setText(ms.metric,mscale.metricText);
259     ms.width=ml+4;
260     ms.topBar.style.width = MVUtil.pixel(mscale.fpsLength);
261     ms.bottomBar.style.width = MVUtil.pixel(mscale.metricLength);
262     ms.metric.style.width=MVUtil.pixel(mscale.metricLength +2);
263     ms.fps.style.width=MVUtil.pixel(mscale.fpsLength +2);
264   }
265   this.decideScale(par,this.maxLength, callBack);
266 }
267 /**@private*/
268 MVScaleBar.prototype.decideScale=function(par,_len,callBack)
269 {
270   //"GEODETIC", "PROJECTED", "LOCAL" and "UNDEFINED".
271   var _radLon;
272   var _radLat;
273   var total_metric_dist;
274   var metric_dist;
275   var metric_text;
276   var metric_length; 
277   var csdcf  = par.msi.getCSdistConvFactor();
278   var csType = par.msi.getCSType();
279   
280   var mv = this;
281   var doTheRest = function()
282   {
283     metric_dist       = mv.round125(total_metric_dist);
284     if( metric_dist >= 1000)
285     {
286       var major_metric_ki = mv.round125(total_metric_dist/1000);
287       var total_metric_ki = total_metric_dist/1000;
288       metric_length = Math.round(mv.maxLength*major_metric_ki/total_metric_ki);
289       metric_text = MVUtil.formatNum(major_metric_ki)+" "+MVi18n.getInfoMessage("kilometers");
290     }
291     else
292     {
293       metric_length = Math.round(mv.maxLength*metric_dist/total_metric_dist);
294       metric_text = MVUtil.formatNum(metric_dist)+" "+MVi18n.getInfoMessage("meters");
295     }
296     var mile_dist=total_metric_dist/1609.344;
297     var fps_dist=total_metric_dist*3.28084;
298     var fps_length;
299     var fps_text;
300     if(mile_dist >= 1)
301     {
302       var major_mile_dist =mv.round125(mile_dist);
303       fps_length = Math.round(mv.maxLength*major_mile_dist/mile_dist);
304       fps_text = MVUtil.formatNum(major_mile_dist)+" "+MVi18n.getInfoMessage("miles");
305     }
306     else
307     {
308       var major_fps_dist=mv.round125(fps_dist);
309       fps_length = Math.round(mv.maxLength*major_fps_dist/fps_dist);
310       fps_text = MVUtil.formatNum(major_fps_dist) +" "+MVi18n.getInfoMessage("feet");
311     }
312     return new MVScale(fps_length,metric_length,fps_text,metric_text);
313   }
314   
315   if (csType=='GEODETIC')
316   {
317     var callBack2 = function(newradLat)
318     {
319       _radLat = newradLat;
320       var smax         = 6378137.0;
321       var t            = 0.9966471893352525 ;        
322       var flat         = 0.0033528106647474805;
323       total_metric_dist = MVScaleBar.calcu_dist( _radLon,  _radLat, smax, t,flat);
324       callBack(doTheRest());
325     }
326     var callBack1 = function(newradLon)
327     {
328       _radLon = newradLon;
329       MVScaleBar.getRadiaLatLng(par.mapScreenCenterX + _len/2,par.mapScreenCenterY, par, callBack2);
330     }
331     MVScaleBar.getRadiaLatLng(par.mapScreenCenterX - _len/2, par.mapScreenCenterY, par, callBack1);
332     return ;
333   }
334   else if (csType=='PROJECTED')
335   {
336     if (this.transformToGeodetic == true)
337     {
338       var  halfLen = _len/2 ;
339       var  _lon1 =  par.mapCenterX - halfLen/par.pixPerX;
340       var  _lon2 =  halfLen/par.pixPerX + par.mapCenterX;
341       var  _lat  = par.mapCenterY; 
342       var mgeometry = MVSdoGeometry.createLineString([_lon1,_lat,_lon2,_lat],par.srid);
343       var callBack3 = function(newLen)
344       {
345         total_metric_dist = newLen ;
346         callBack(doTheRest());
347       }
348       total_metric_dist = mgeometry.getLength ("meter", true, par.baseURL+'/foi', callBack3);
349       return ;
350     }
351     else
352     {
353       total_metric_dist =  csdcf *_len/par.pixPerX; 
354     }
355   }
356   else if (csType=='LOCAL' || (csType=='UNDEFINED' && csdcf!=0))
357   {
358     total_metric_dist =  csdcf *_len/par.pixPerX; 
359   }
360   else if (csType=='UNDEFINED' && csdcf==0)
361   {
362     total_metric_dist =  _len/par.pixPerX; 
363     MVGlobalVariables.mKilometers = '';
364     MVGlobalVariables.mMiles = '';
365     MVGlobalVariables.mMeters = '';
366     MVGlobalVariables.mFeet = '';
367   }
368   callBack(doTheRest());
369 }
370 /**@private*/
371 MVScaleBar.prototype.round125 = function(_num)
372 {
373   var num = _num;
374   if( num > 1)
375   {
376     var i = 0;
377     while ( num >= 10 ){ num = num/10; i = i + 1;}
378     if( num >= 5 ){ num = 5;}
379     else if( num >= 2){ num = 2;}
380     else { num = 1;}
381     while(i > 0){num = num*10;i = i - 1;}
382   }
383   return num;
384 }

410 /**@private*/
411 MVScaleBar.getRadiaLatLng = function(_lon, _lat, mv, callBack)
412 {
413   var _point = MVSdoGeometry.createPoint (0,0, mv.srid); 
414   _point.sdo_point.x = (_lon - mv.mapScreenCenterX)/mv.pixPerX + mv.mapCenterX;
415   _point.sdo_point.y= mv.mapCenterY - (_lat-mv.mapScreenCenterY)/mv.pixPerY;
416   var tmpcallBack = function(newPoint)
417   {
418     if(newPoint)
419       _point = newPoint;
420     while(_point.sdo_point.x<-180) _point.sdo_point.x+=360;
421     while(_point.sdo_point.x>180) _point.sdo_point.x-=360;
422     _point.sdo_point.x =( _point.sdo_point.x)* Math.PI/180;
423     _point.sdo_point.y =(_point.sdo_point.y )*Math.PI/180;
424     callBack(_point.sdo_point);
425   }
426   if (_point.srid && _point.srid !=8307)
427     _point = mv.transformGeom(_point,8307,null,tmpcallBack);
428   else
429     tmpcallBack() ;
430 }
431 /**Author: B. Blackwell
432 Revisions: 21 Mar 01 - Fixed antipodal test BHB
433             30 May 00 - Original Version BHB
434 Notes: Input is geod. long, lat in radians of 2 points on Earth spheroid.
435         Based on Mathematica Geodesy package by John Novak
436         Will return 0. for arcs of the order of a meter or less. For very short
437         arcs use mdsphedis().
438         Will return -1. for points that are nearly precise antipodes (within
439         a meter or so).
440         Therefore can only be used for arcs less than 179.99999 degress (almost Pi)
441         of central angle.
442         Accuracy is of order 1e-9.
443         Here are the parameter values for a few commonly used SRIDs:
444         srid           smax       t                 flat
445         8265     6378137.0,  0.9966471893188177,   0.003352810681182319
446         8307     6378137.0,  0.9966471893352525,   0.0033528106647474805
447       @private
448 */
449 MVScaleBar.calcu_dist = function( _point1,  _point2, smax, t,flat)
450 {
451   var lamm; var phim; var dphim; var cl; var d; var ce; var cyp; var ctp; var cx; var cy;
452   var ct; var cd; var ca; var phi1; var phi2;
453   var sphim; var cphim; var sdphim; var cdphim; var slamm; var sd; /* Sines and Cosines */
454   var HalfPI = Math.PI/2.0;
455 
456   if( HalfPI-Math.abs(_point1.y) < 1.e-10) /* Trap points very near poles */
457     phi1 = _point1.y;
458   else
459     phi1 = Math.atan(t*Math.tan(_point1.y));
460   if( HalfPI-Math.abs(_point2.y) < 1.e-10)
461     phi2 = _point2.y;
462   else
463     phi2 = Math.atan(t*Math.tan(_point2.y));
464   phim  = (phi1+phi2)/2.0;
465   dphim = (phi1-phi2)/2.0;         /* corrected latitude half difference */
466   lamm = Math.abs(_point1.x - _point2.x);
467   if( lamm > Math.PI)
468       lamm = 2.0 *Math.PI - lamm;
469   lamm /= 2.0;          /* longitude half difference */
470   if( lamm < 5.e-8  &&  Math.abs(dphim) < 5.e-8 )
471       return 0.0;                   /* arc length < 1 meter */
472   else if((HalfPI-lamm < 5.e-8  && Math.abs(phim) < 5.e-8) ||
473             HalfPI-Math.abs(dphim) < 5.e-8 )
474       return -1.0 ;                                 /* too close to antipodes */
475   sphim  = Math.sin(phim);
476   cphim  = Math.cos(phim);
477   sdphim = Math.sin(dphim);
478   cdphim = Math.cos(dphim);
479   slamm  = Math.sin(lamm);
480   cl = sdphim*sdphim + slamm*slamm*(cdphim*cdphim - sphim*sphim);
481   d = Math.acos(1.0 - 2.0*cl);
482   sd = Math.sin(d);
483   ce = 2.0*(1.0 - 2.0*cl);
484   cyp = 2.0*sphim*sphim*cdphim*cdphim/(1.0 - cl);
485   ctp = 2.0*sdphim*sdphim*cphim*cphim/cl;
486   cx = cyp + ctp;
487   cy = cyp - ctp;
488   ct = d/sd;
489   cd = 4.0*ct*ct;
490   ca = cd*ce;
491   var t1=(ca+(ct-(ca-ce)/2.0)*cx);
492   return smax*sd*(ct-flat*(ct*cx-cy)/4.0 +flat*flat *
493           (cx*t1-cy*(2.0*cd+ce*cy)+cd*cx*cy)/64.0);
494 }
