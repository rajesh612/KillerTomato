 35 /**
 36  *@class MVCircleTool defines a circle tool that can be used to interactively
 37  * draw a circle on the map by holding down and dragging the mouse. 
 38  *<p>After the user finishes circle drawing and releases the mouse
 39  *button, a circular FOI can be generated and displayed
 40  *on the map until explicitly removed using its clear() method.</p>
 41  *
 42  *@constructor
 43  *@param {String} circleStyle a string parameter that specifies the MapViewer style to
 44  *                          be used to render the circle area FOI after the
 45  *                          user finishes drawing circle and releases the mouse button.
 46  *                          The style is specified in the form "data-source-name"."style-name",
 47  *                          where "style-name" is the name of the
 48  *                          MapViewer area style and "data-source-name" is the
 49  *                          name of data source in which the style is defined.
 50  *                          If this parameter is omitted or set to null, no circle
 51  *                          FOI is generated and displayed after the user
 52  *                          releases the mouse button.
 53  *@param {String} foiServerUrl an optional String parameter that specifies the
 54  *                          URL of the FOI server, which should be in the form
 55                             http://host:port/mapviewer/foi. If this parameter is omitted,
 56  *                          the URL of the FOI server is derived
 57  *                          from MapViewer base URL.
 58  *@return The newly created MVCircleTool object.
 59 */

 81   this.fromLon = 0;
 82   /**@private*/
 83   this.fromLat = 0;
 84   /**@private*/

 86   /**@private*/

 88   /**@private*/

 92   /**@private*/
 93   this.scrollMapFromLat = 0;
 94   /**@private*/
 95   this.EventListeners = new Object();
 96   /**@private*/
 97   this.onStartListener=null;
 98   /**@private*/
 99   this.onFinishListener=null;
100   /**@private*/
101   this.onDragListener=null;
102   /**@private*/
103   this.mRadius=null;
104   /**@private*/
105   this.sdoGeometry=null;

108   /**@private*/

112   /**@private*/
113   this.dragStatus=0;  //0: not init; 1: init; 2: begin dragging; 3: dragged
114   /**@private*/

116   /*Only for Netscape*/
117   /**@private*/
118   var ct = this ;

120   {
121     evt = (evt) ? evt : ((window.event) ? event : null);

123       MVUtil.stopEvent(evt);

132         this.netscape = true;
133       else
134         this.netscape = false;
135     }
136     if(this.netscape)
137     {
138       window.addEventListener("keypress", this.keyPressFunc, false);
139     }
140   }
141   /*Only for Netscape*/
142   /**@private*/
143   this.enableKeyPress = function()
144   {
145     if(this.netscape == null)
146     {
147       if(MVGlobalVariables.checkBrowser("netscape"))
148         this.netscape = true;
149       else
150         this.netscape = false;
151     }
152     if(this.netscape)
153       window.removeEventListener("keypress", this.keyPressFunc, false);
154   }
155   
156     /**@private*/
157 	this.refreshCircle = function()
158 	{
159 		var tmpMapView= ct.parent;
160 		if(tmpMapView.wraparound)
161 		{
162 			if(ct.getStatus() == 3)
163 			{			
164 				var circleObj = ct.getCircleFOI();
165 				if(circleObj)
166 				{
167 					circleObj.areacode = ct.parent.indFOIsControl.centerArea;
168 				}
169 			}
170 		}
171 	}

286 /**
287  *@private
288 */
289 MVCircleTool.prototype.drawCircle = function()
290 {
291   if(this.dragStatus!=3)
292   {
293     return;
294   }
295   var pixPX = this.parent.pixPerX;
296   var pixPY = this.parent.pixPerY;
297   var fromX = Math.round((this.fromLon - this.parent.getCenterLon())*pixPX + (this.parent.getPaneWidth()/2));
298   var fromY = Math.round((this.fromLat - this.parent.getCenterLat())*(-pixPY) + (this.parent.getPaneHeight()/2));                   
299   var difX = (this.toLon - this.fromLon)*pixPX;
300   var difY = (this.toLat - this.fromLat)*(-pixPY);
301   var radius = Math.round(Math.sqrt(difX*difX + difY*difY)); 
302   
303   var halfEarthLength = Math.floor( pixPX * (this.parent.msi.coverMaxX - this.parent.msi.coverMinX)/2 );
304   if( radius > halfEarthLength)
305   {
306 	radius = halfEarthLength;
307 	this.mRadius = Math.floor( (this.parent.msi.coverMaxX - this.parent.msi.coverMinX)/2 );
308   }
309   else
310 	 this.mRadius = radius / pixPX;
311   
312  
313   var width = 2*(radius + this.plotPara.divWidth);
314   var height = 2*(radius + this.plotPara.divHeight);
315   var leftPos = fromX - radius - Math.round((this.plotPara.divWidth)/2);
316   var topPos = fromY - radius - Math.round((this.plotPara.divHeight)/2);  
317   this.circleDiv.style.left = leftPos + 'px';
318   this.circleDiv.style.top = topPos + 'px';  
319   this.circleDiv.style.width = width + 'px';
320   this.circleDiv.style.height = height + 'px';
321   this.circleDiv.style.clip = "rect(0px, " + height + "px, " + width + "px, 0px)";
322   this.circleDiv.style.visibility = "visible";
323   var plotsDivId = (this.namePrefix + this.id + "PlotsDiv");
324   var plotsDiv = document.getElementById(plotsDivId);
325   if(plotsDiv != null)
326     this.circleDiv.removeChild(plotsDiv);
327   plotsDiv = document.createElement("div");
328   plotsDiv.id = plotsDivId;
329   plotsDiv.style.position = "absolute";
330   plotsDiv.style.left = 0 + 'px';
331   plotsDiv.style.top = 0 + 'px';
332   this.circleDiv.appendChild(plotsDiv);
333   /*
334   this.plotPos(radius, 0);
335   this.plotPos(radius, (2*radius));
336   this.plotPos((2*radius), radius);
337   this.plotPos(0, radius);
338   */
339   /*The First Implementation
340   for(var i = 1; i < radius; i += this.stepPx)
341   {
342     difY = Math.round(Math.sqrt(radius*radius-i*i));
343     this.plotPos((radius + i), (radius + difY));
344     this.plotPos((radius + i), (radius - difY));
345     this.plotPos((radius - i), (radius + difY));
346     this.plotPos((radius - i), (radius - difY));
347   }
348   */    
349   /*The Second Implementation
350   var slice = (Math.PI)/(2*stepRad);
351   for(var i = 1; i < stepRad; ++i)
352   {
353     var rad = i*slice;
354     difX = Math.round(radius*Math.cos(rad));
355     difY = Math.round(radius*Math.sin(rad));
356     this.plotPos((radius + difX), (radius + difY));
357     this.plotPos((radius + difX), (radius - difY));
358     this.plotPos((radius - difX), (radius + difY));
359     this.plotPos((radius - difX), (radius - difY));
360   }
361   */
362   
363   //The Third Implementation
364   var ratio = (this.stepPx)/radius;
365   ratio = ratio>1?1:ratio;
366   var stepRad1 = Math.acos(ratio);
367   var stepRad2 = Math.asin(ratio);
368   var stepRad = stepRad1<stepRad2?stepRad1:stepRad2;
369   var maxSpec = 50;
370   if(stepRad == 0)
371       stepRad = Math.PI/(2*maxSpec);
372   var iter = Math.ceil(Math.PI/(2*stepRad));
373   for(var i = 0; i < iter; ++i)
374   {
375     var rad = i*stepRad;
376     difX = Math.round(radius*Math.cos(rad));
377     difY = Math.round(radius*Math.sin(rad));
378     this.plotPos((radius + difX), (radius + difY));
379     this.plotPos((radius + difX), (radius - difY));
380     this.plotPos((radius - difX), (radius + difY));
381     this.plotPos((radius - difX), (radius - difY));
382   }
383 }

388 MVCircleTool.prototype.plotPos = function(posX, posY)
389 {
390   var plotsDivId = (this.namePrefix + this.id + "PlotsDiv");
391   var plotsDiv = document.getElementById(plotsDivId);
392   if(plotsDiv == null)
393   {
394     MVi18n.reportError("MVCircleTool.plotPos", "MAPVIEWER-05500", "null plodsDiv found",

396     return;
397   }
398   var plotDiv = document.createElement("DIV");
399   plotDiv.style.position = "absolute";
400   plotDiv.style.zIndex = this.plotPara.zIndex;
401   if(this.plotPara.backgroundColor != null)
402     plotDiv.style.backgroundColor = this.plotPara.backgroundColor;
403   if((this.plotPara.opacity) != null)
404   {
405       if(window.ActiveXObject)
406       {
407         plotDiv.style.filter = "alpha(opacity=" + ((this.plotPara.opacity)*100) + ")"; 
408       }
409       else
410       {
411         plotDiv.style.opacity = this.plotPara.opacity; 
412       }
413   }
414   plotDiv.style.left = posX + "px";
415   plotDiv.style.top = posY + "px";
416   plotDiv.style.width = (this.plotPara.divWidth) + "px";
417   plotDiv.style.height = (this.plotPara.divHeight) + "px";
418   plotDiv.style.clip = "rect(0px, " + (this.plotPara.divWidth) + "px, " + 
419                          (this.plotPara.divHeight) + "px, 0px)";
420   plotsDiv.appendChild(plotDiv);

423 /**
424  *@private
425 */
426 MVCircleTool.prototype.generateArea = function()
427 {
428   if(this.dragStatus!=3)
429   {
430     return;
431   }
432   var foiId = this.namePrefix + this.id;
433   this.parent.removeFOI(foiId);
434 
435   var radius = this.mRadius;
436   
437   this.sdoGeometry=MVSdoGeometry.createCirclePolygon(this.fromLon, this.fromLat, radius, this.parent.srid);
438   
439   if(!(this.areaStyle))
440     return;   
441   var foi=new MVFOI(foiId, this.sdoGeometry, this.areaStyle, this.foiServerUrl);          
442   var leftUpperPoint = this.fromLon - radius;
443   foi.areacode = MVUtil.calAreaCodeFromCenter(this.parent,leftUpperPoint);
444   //jslog.info("circle center:" +  this.fromLon + " | " +  this.fromLat +" radius: " + radius + " areacode: " + foi.areacode );
445   
446   foi.setClickable(false);
447   foi.enableInfoTip(false);
448   this.parent.addFOI(foi);
449 }

452 /**
453  *This method activates and initializes the circle tool. 
454  *The application must call this method to activate the 
455  *circle tool before the user can draw circle on the map.
456  *@return none
457  *@type void
458 */  
459 MVCircleTool.prototype.init = function()
460 { 
461   //this.clear() ;
462   this.parent.removeFOI(this.namePrefix + this.id);
463   this.dragStatus=0;
464   this.parent.addEventHandlers();
465   
466   this.dragStatus=1;  
467   var tmpMapView=this.parent;
468   var tmpDiv=tmpMapView.base_div;
469   var tmpCircleTool=this;  
470   var interVal;
471 
472   MVUtil.runListeners(tmpCircleTool.EventListeners, MVEvent.INIT, [MVToolBar.BUILTIN_CIRCLE]);  
473   
474   // For the js obfuscator to properly replace local function names, 
475   // a local function must be declared before it is invoked. -- Ji
476   var refreshRealTime=function()
477   {
478     tmpMapView.enableMapEventListner=true;
479     
480     var centerLon=tmpMapView.getCenterLon();
481     var centerLat=tmpMapView.getCenterLat();
482      
483     if((tmpCircleTool.scrollMapFromLon)!=centerLon || (tmpCircleTool.scrollMapFromLat)!=centerLat || (tmpCircleTool.refresh))
484     {
485        tmpCircleTool.refresh = false;
486        if(tmpCircleTool.dragStatus==2)
487        {   
488          tmpCircleTool.dragStatus=3;          
489        }    
490        tmpCircleTool.toLon+=(centerLon-tmpCircleTool.scrollMapFromLon);
491        tmpCircleTool.toLat+=(centerLat-tmpCircleTool.scrollMapFromLat);      
492        tmpCircleTool.scrollMapFromLon=centerLon;
493        tmpCircleTool.scrollMapFromLat=centerLat;
494        tmpCircleTool.drawCircle();
495     }
496   }
497       
498   MVUtil.setListener(tmpDiv, "mousedown", MVUtil.closureWrapper(tmpDiv, function(evt)
499   {    
500     tmpMapView.removeFOI(tmpCircleTool.namePrefix + tmpCircleTool.id);    
501     tmpCircleTool.dragStatus=2;
502     
503     evt = (evt) ? evt : ((window.event) ? event : null);
504     MVUtil.stopEvent(evt) ;
505     tmpMapView.getMPos(evt);
506     
507     tmpCircleTool.startDrag();
508     tmpCircleTool.endDrag();
509     tmpCircleTool.circleDiv.style.visibility = "hidden";
510     
511     if(tmpCircleTool.onStartListener!=null)
512       (tmpCircleTool.onStartListener)();
513     MVUtil.runListeners(tmpCircleTool.EventListeners,MVEvent.START);
514     
515     tmpCircleTool.scrollMapFromLon = tmpMapView.getCenterLon();
516     tmpCircleTool.scrollMapFromLat = tmpMapView.getCenterLat();  
517         
518     var waitTime=5; 
519     interVal=setInterval(refreshRealTime, waitTime);
520 
521     tmpCircleTool.disableKeyPress();
522     
523     return false;
524   }), true);
525   
526   MVUtil.setListener(tmpDiv, "mousemove", MVUtil.closureWrapper(tmpDiv, function(evt)
527   {    
528     if(tmpCircleTool.dragStatus==2)
529     {   
530       tmpCircleTool.dragStatus=3;          
531     }   
532     if(tmpCircleTool.dragStatus==3)
533     {      
534        evt = (evt) ? evt : ((window.event) ? event : null);
535        tmpMapView.getMPos(evt);        
536        tmpCircleTool.endDrag();
537        tmpCircleTool.refresh = true;
538        if(tmpCircleTool.onDragListener)
539          tmpCircleTool.onDragListener();
540        MVUtil.runListeners(tmpCircleTool.EventListeners,MVEvent.DRAG);
541     }         
542     return false;
543   }), true);
544   
545   MVUtil.setListener(tmpDiv, "mouseup", MVUtil.closureWrapper(tmpDiv, function(evt)
546   {
547     evt = (evt) ? evt : ((window.event) ? event : null);
548     MVUtil.stopEvent(evt) ;
549     tmpMapView.getMPos(evt);    
550     tmpCircleTool.endDrag(); 
551     tmpCircleTool.generateArea();
552     var plotsDiv = document.getElementById(tmpCircleTool.namePrefix + tmpCircleTool.id + "PlotsDiv");
553     if(plotsDiv != null)
554       tmpCircleTool.circleDiv.removeChild(plotsDiv);
555     tmpCircleTool.circleDiv.style.width="0px";
556     tmpCircleTool.circleDiv.style.height="0px";
557     tmpCircleTool.circleDiv.style.clip = "rect(0px, 0px, 0px, 0px)";
558     tmpCircleTool.circleDiv.style.visibility = "hidden";   
559     clearInterval(interVal);    
560     tmpMapView.addEventHandlers();    
561 
562     tmpCircleTool.enableKeyPress();
563 
564     if(tmpCircleTool.onFinishListener!=null)
565       (tmpCircleTool.onFinishListener)();   
566     MVUtil.runListeners(tmpCircleTool.EventListeners, MVEvent.FINISH );
567     return false;
568   }), true);
569   
570   tmpDiv.onmouseout = MVUtil.closureWrapper(tmpDiv, function(evt)
571   {
572    if(tmpCircleTool.dragStatus!=3)
573      return false;
574      
575    if(!(window.ActiveXObject))
576    {
577      evt = (evt) ? evt : ((window.event) ? event : null);
578                
579      var clientx=evt.clientX;
580      var clienty=evt.clientY;
581 
582      var width = tmpMapView.getPaneWidth();
583      var height = tmpMapView.getPaneHeight();
584      var origXY = MVUtil.offsetPositionAbsolute(this);
585 
586      var OffsetDIVX = evt.pageX - origXY.x;
587      var OffsetDIVY = evt.pageY - origXY.y;
588      
589      if(OffsetDIVX<=0 || OffsetDIVX>=width || 
590      OffsetDIVY<=0 || OffsetDIVY>=height ||
591      clientx<=0 || clientx>=document.body.clientWidth ||
592      clienty<=0 || clienty>=document.body.clientHeight)
593      {     
594        return this.onmouseup(evt);
595      }
596    }
597   });
598   
599   //refresh the circle tool when refresh the ind foi
600   this.parent.indFOIsControl.attachEventListener("refreshFOI",this.refreshCircle);
601   
602   
603 }

638  *This method returns the MVSdoGeometry object that specifies a polygon that proximates 
639  *the circle on the map. 
640  *@return the MVSdoGeometry object that specifies the circle polygon on the map.
641  *@type MVSdoGeometry
642  *@see MVSdoGeometry
643  *@requires MVSdoGeometry
644 */
645 MVCircleTool.prototype.getCirclePolygon = function()
646 { 
647   if(this.dragStatus==3 && this.mRadius)
648     return MVSdoGeometry.createCirclePolygon(this.fromLon, this.fromLat, this.mRadius, this.parent.srid);
649   else if(this.sdoGeometry!=null)
650     return (this.sdoGeometry);
651   else
652     return null;
653 }

657  *@param unit a string that specifies the name of the distance unit. If omitted, the 
658  * radius is calculated in the default distance unit of the map coordinate system.
659  *@return the circle radius.
660  *@type double

662 MVCircleTool.prototype.getRadius = function(unit)
663 {
664   if(!unit)
665     return MVSdoGeometry.getDistance(this.fromLon, this.fromLat, this.toLon, this.toLat, this.parent.srid);
666   else
667     //return MVSdoGeometry.getDistance(this.fromLon, this.fromLat, this.fromLon, this.fromLat+this.mRadius, this.parent.srid, unit) ;
668 	return MVSdoGeometry.getDistance(this.fromLon, this.fromLat, this.toLon, this.toLat, this.parent.srid, unit) ;
669 }

719 /**
720  *This method attaches an event listener to the circle tool. One of the following types of event
721  *listeners can be associated with the circle tool through this method: MVEvent.START, MVEvent.FINISH and
722  MVEvent.DRAG. 
723  *<ul>
724  *<li>The MVEvent.START listener is called 
725  *when the circle tool is activated and the user starts drawing circle  
726  *by clicking the mouse left key on the map. </li>
727  *<li>The MVEvent.FINISH listener is called when the user finishes drawing 
728  *by releasing the mouse left key.</li>
729  *<li>The MVEvent.DRAG listener is called when the user drags the mouse while holding the mouse 
730  *left key.</li>
731  *</ul>
732  *<p>No parameter is passed to the listener function when it is invoked.</p>
733  *@param {String} event a String that specifies the type of event to be handled by 
734  *the listener. It can be either MVEvent.START, MVEvent.FINISH or MVEvent.DRAG.
735  *@param {Function} listener the reference to listener function.
736  *@return none
737  *@type void
738 */
739 MVCircleTool.prototype.attachEventListener = function(type, listener)
740 {
741     MVUtil.attachEventListener(this.EventListeners,type,listener)
742 }

747  *the listener. It can be either MVEvent.START, MVEvent.FINISH or MVEvent.DRAG.
