194      }
195    }
196    themebasedfoi.themeLayerControl = this ;

198    this.realWidth = 0;
199    this.realHeight = 0;
200    this.width = 0;
201    this.height = 0;

214    this.ltop = 0 ;
215    this.lDragOffsetX = 0;

217    this.centerX = 0;

259  /**
260  *@private
261  */
262  MVThemeBasedFOIControl.prototype.beforePrint = function(stylename)
263  {
264    if(this.themebasedfoi.isJDBCTheme)
265      return this.getJDBCTheme(this.themebasedfoi.theme) ;
266    else
267    {
268      var theme ="<theme name=\""+this.themebasedfoi.name+
269        "\" template_theme=\""+ MVUtil.strAfterDot(this.themebasedfoi.theme) +
270        "\" render_labels=\""+ this.themebasedfoi.renderLabels + "\" datasource=\""+
271        MVUtil.strBeforeDot(this.themebasedfoi.theme) + "\"" +
272        this.getRenderStyleParameterStr(stylename)+
273        this.getSequenceParameterStr()+">\n" ;
274      
275      if(this.themebasedfoi.queryParameter == null)
276        theme += "</theme>";
277      else
278      {
279        /* <themes>
280           <theme name="theme_us_dyn_counties">
281             <binding_parameters><parameter value="San Francisco" type="String"/></binding_parameters>
282           </theme>
283         </themes> */
284       theme +="  <binding_parameters>\n";
285       for(var i=0; i<this.themebasedfoi.queryParameterArray.length; i++)
286       {
287         var param = this.themebasedfoi.queryParameterArray[i] ;
288          theme += "<parameter " ;
289         if(param.type==undefined)
290           theme += "value=\""+param.value+"\">" ;
291         else
292         {
293           theme += "type=\"" + param.type + "\" " ;
294           if(param.type=="geometry")
295           {
296             theme += ">" + param.value.toGML() ;
297           }
298           else // array
299           {
300             theme += "elemtype=\""+ param.elemtype +"\" " ;
301             if(!param.value)
302               theme += ">" ;
303             else
304             {
305               if(param.value.substring != undefined)
306               {
307                 var val = param.value ;
308                 if(param.elemtype == "String")
309                 {
310                   var values = param.value.split(",");
311                   finalvalue = "";
312                   for(var j=0;j<values.length;j++)
313                   {
314                     if(j > 0)
315                       finalvalue += ",";
316                     var sidx = values[j].indexOf("\"") ;
317                     var eidx = values[j].lastIndexOf("\"") ;
318                     finalvalue += values[j].substring(sidx+1, eidx);
319                   }
320                   val = finalvalue ;
321                 }
322                    // old array specification
323                 theme += "value=\"" + MVUtil.strReplaceAll(val, "\"", """) + "\">" ;
324               }
325               else
326               {
327                 // new array specification
328                 theme += ">" ;
329                 for(var j=0; j<param.value.length; j++)
330                   theme += "<elem>"+param.value[j]+"</elem>" ;
331               }
332             }
333           }
334         }
335         theme += "</parameter>" ;
336       }
337       theme += "  </binding_parameters>\n";
338       theme += "</theme>";
339      }
340      return theme ;
341    }
342  }

501  /**
502  *@private
503  */
504  MVThemeBasedFOIControl.prototype.setVisible = function(par,visible)
505  {
506    if(visible)
507    {
508      if(!this.isVisible())
509      {
510        if (this.foiDisplayed)
511        {
512          if((!this.shouldUseDivFilter() && this.themebasedfoi.getFOIData()!=null &&
513           this.themebasedfoi.getFOIData()[0].gtype % 10 != 1) ||
514            this.isWholeImage() || !this.themebasedfoi.renderFOIImages)
515          {
516            if(this.themebasedfoi.clickable && this.transImageLayer)
517              this.transImageLayer.style.visibility = "visible" ;
518          }
519        }
520        this.div.style.visibility="visible" ;
521 
522        if(!this.isLoaded)
523        {
524           MVUtil.setLeftTop(this.div, Math.ceil((this.minX - this.parent.mapWindowMinLon) * this.parent.pixPerX),
525                             -Math.ceil((this.maxY - this.parent.mapWindowMaxLat) * this.parent.pixPerY));
526           this.refreshFOIs(this.parent,this.minX, this.minY, this.maxX, this.maxY, true);
527        }
528      }
529    }
530    else
531    {
532      if (this.foiDisplayed)
533      {
534        if((!this.shouldUseDivFilter() && this.themebasedfoi.getFOIData()!=null &&
535            this.themebasedfoi.getFOIData()[0].gtype % 10 != 1) ||
536            this.isWholeImage() || !this.themebasedfoi.renderFOIImages)
537        {
538          if(this.transImageLayer)
539          this.transImageLayer.style.visibility="hidden";
540          if (this.parent.infoLayer.infoWindows.length > 0)
541          {
542            for (var a = 0; a < this.parent.transparentLayer.childNodes.length; a++)
543            {
544              if (this.parent.infoLayer.infoWindows[0].nid == this.parent.transparentLayer.childNodes[a].id)
545              {
546                this.parent.infoLayer.deleteInfoWindow(this.parent.infoLayer.infoWindows[0]);
547                break;
548              }
549            }
550          }
551        }
552        else if (this.parent.infoLayer.infoWindows.length > 0)
553        {
554          for (var b = 0; b < this.div.childNodes.length; b++)
555          {
556            if (this.parent.infoLayer.infoWindows[0].nid == this.div.childNodes[b].id)
557            {
558              this.parent.infoLayer.deleteInfoWindow(this.parent.infoLayer.infoWindows[0]);
559              break;
560            }
561          }
562        }
563      }
564      this.div.style.visibility="hidden" ;
565      this.loading = false ;
566      this.parent.checkLayersAndSetLoadingIcon() ;
567    }
568  }
