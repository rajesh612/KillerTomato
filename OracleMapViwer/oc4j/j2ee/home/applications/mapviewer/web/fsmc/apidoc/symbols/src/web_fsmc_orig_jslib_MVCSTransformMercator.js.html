 41 MVCSTransformMercator.geodeticToMercator = function(lon, lat, fromSrid, toSrid, errorHandler)
 42 {
 43   var mercParams = MVCSTransformMercator.getCSParameters(toSrid) ;
 44   var geodParams = MVCSTransformMercator.getCSParameters(fromSrid) ;
 45   if(!mercParams || !geodParams)
 46   {
 47     MVi18n.reportError("MVCSTransformMercator.geodeticToMercator", "MAPVIEWER-05522", fromSrid+"->"+toSrid, errorHandler) ;
 48     return null ;
 49   }
 50   var radians = MVGeomUtil.getRadiansFromLatLon({x:lon, y:lat}) ;
 51   lon = radians.x ;
 52   lat = radians.y ;
 53   var mdFE = mercParams[3] ;
 54   var mdFN = mercParams[4] ;
 55   var k0 = mercParams[2] ;
 56   var lon0 = mercParams[1] ;
 57   var a = geodParams[0] ;
 58   var f = 1/geodParams[1] ;
 59   var e = Math.sqrt(2*f-f*f) ;
 60   var rx = mdFE + a*k0*(lon-lon0) ;
 61   var sin = Math.sin(lat) ;
 62   var tan = Math.tan(Math.PI/4 + lat/2) ;
 63   var powParam1 = (1-e*sin)/(1+e*sin);
 64   var powParam2 = e/2 ;
 65   var pow = Math.pow(powParam1,powParam2) ;
 66   var ry = mdFN + a*k0*Math.log(tan*pow);
 67   return {x:rx, y:ry};
 68 }

 70 MVCSTransformMercator.mercatorToGeodetic = function(x, y, fromSrid, toSrid, errorHandler)
 71 {
 72   var mercParams = MVCSTransformMercator.getCSParameters(fromSrid) ;
 73   var geodParams = MVCSTransformMercator.getCSParameters(toSrid) ;
 74   if(!mercParams || !geodParams)
 75   {
 76     MVi18n.reportError("MVCSTransformMercator.mercatorToGeodetic", "MAPVIEWER-05522", fromSrid+"->"+toSrid, errorHandler) ;
 77     return null ;
 78   }
 79   
 80   var mdFE = mercParams[3] ;
 81   var mdFN = mercParams[4] ;
 82   var k0 = mercParams[2] ;
 83   var lon0 = mercParams[1] ;
 84   var a = geodParams[0] ;
 85   var f = 1/geodParams[1] ;
 86   var e = Math.sqrt(2*f-f*f) ;
 87   var N = y ;
 88   var E = x ;
 89   var B = Math.E ;
 90   var t = Math.pow(B, ((mdFN-N)/(a*k0))) ;
 91   var chi = Math.PI/2 - 2*Math.atan2(t, 1.0);
 92   var pow4 = Math.pow(e,4.0);
 93   var pow6 = Math.pow(e,6.0);
 94   var pow8 = Math.pow(e,8.0);
 95   var p1 = e*e/2;
 96   var p2 = 5*pow4/24;
 97   var p3 = pow6/12 ;
 98   var p4 = 13*pow8/360;
 99   var p5 = Math.sin(2*chi) ;
100   var p6 = 7*pow4/48 ;
101   var p7 = 29*pow6/240 ;
102   var p8 = 811*pow8/11520 ;
103   var p9 = Math.sin(2*chi) ;
104   var p10 = 7*pow6/120 ;
105   var p11 = 81*pow8/1120 ;
106   var p12 = Math.sin(6*chi);
107   var p13 = 4279*pow8/161280 ;
108   var p14 = Math.sin(8*chi) ;
109   var lat = chi + (p1+p2+p3+p4)*p5 + (p6+p7+p8)*p9 + (p10+p11)*p12 + p13*p14;
110   var lon = ((E-mdFE)/(a*k0)) + lon0 ;
111   return MVGeomUtil.getLatLonFromRadians({x:lon, y:lat});
112 }

114 MVCSTransformMercator.getCSParameters = function(srid)
115 {
116   for(var i=0; i<MVCSTransformMercator.csParameters.length; i++)
117   {
118     if(MVCSTransformMercator.csParameters[i].srid == srid)
119       return MVCSTransformMercator.csParameters[i].params ;
120   }
121   return null ;
122 }
